                        #!/bin/sh
                        echo "Please guess the magic number"
                        read X
                        echo $X | grep "[^0-9]" > /dev/null 2>&1
                        if [ "$?" -eq "0" ]; then
                            echo "Sorry wanted a number"
                        else
                            if [ "$X" -eq "6" ]; then
                                echo "you got it"
                            fi
                        fi

函数


                        # common.lib
                        STD_MSG="About to rename some file"
                        rename()
                        {
                            FROM=$1
                            TO=$2
                            
                            for i in *$FROM
                            do
                                j=`basename $i $FROM`
                                mv $i ${j}$TO
                            done
                        }
                    

                        #!/bin/sh
                        . ./common.lib
                        echo $STD_MSG
                        rename .txt .bak
