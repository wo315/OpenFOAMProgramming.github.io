64Foam::scalar Foam::radiation::FresnelLaser::rho
65(
66 const scalar cosTheta
67) const
68{
69 //scalar cosTheta = cos(incidentAngle);
70
71 scalar rho =
72 0.5
73 * (
74 (1 + sqr(1 - epsilon_*cosTheta))/(1 + sqr(1 + epsilon_*cosTheta))
75 +
76 (sqr(epsilon_) - 2*epsilon_*cosTheta + 2*sqr(cosTheta))
77 /
78 (sqr(epsilon_) + 2*epsilon_*cosTheta + 2*sqr(cosTheta))
79 );
80
81 return rho;
82}
