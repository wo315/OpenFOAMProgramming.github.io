28 Foam::globalIndex
29
30Description
31 Calculates a unique integer (label so might not have enough room - 2G max)
32 for processor + local index. E.g.
33
34 globalIndex globalFaces(mesh.nFaces());
35 label globalFacei = globalFaces.toGlobal(facei);

504 // Misc low-level gather routines
505
506 //- Inplace collect in processor order on master (== procIDs[0]).
507 // Note: adjust naming?
508 template<class ProcIDsContainer, class Type>
509 static void gather
510 (
511 const labelUList& offsets,
512 const label comm,
513 const ProcIDsContainer& procIDs,
514 List<Type>& fld,
515 const int tag = UPstream::msgType(),
516 const UPstream::commsTypes ct = UPstream::commsTypes::nonBlocking
517 )
518 {
519 List<Type> allData;
520 gather(offsets, comm, procIDs, fld, allData, tag, ct);
521
522 const int masterProci = procIDs.size() ? procIDs[0] : 0;
523 if (UPstream::myProcNo(comm) == masterProci)
524 {
525 fld.transfer(allData);
526 }
527 }
