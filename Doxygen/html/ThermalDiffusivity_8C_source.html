119template<class BasicTurbulenceModel>
120Foam::tmp<Foam::volScalarField>
121Foam::ThermalDiffusivity<BasicTurbulenceModel>::alphat() const
122{
123 return tmp<volScalarField>::New
124 (
125 IOobject
126 (
127 IOobject::groupName("alphat", this->alphaRhoPhi_.group()),
128 this->runTime_.timeName(),
129 this->mesh_,
130 IOobject::NO_READ,
131 IOobject::NO_WRITE
132 ),
133 this->mesh_,
134 dimensionedScalar(dimDensity*dimViscosity, Zero)
135 );
136}
