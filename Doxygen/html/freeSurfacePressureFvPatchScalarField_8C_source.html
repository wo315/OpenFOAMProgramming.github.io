133void Foam::freeSurfacePressureFvPatchScalarField::updateCoeffs()
134{
135 if (updated())
136 {
137 return;
138 }
139
140 const fvMesh& mesh = patch().boundaryMesh().mesh();
141
142 interfaceTrackingFvMesh& itm =
143 refCast<interfaceTrackingFvMesh>
144 (
145 const_cast<dynamicFvMesh&>
146 (
147 mesh.lookupObject<dynamicFvMesh>("fvSolution")
148 )
149 );
150
151 operator==
152 (
153 pa_ + itm.freeSurfacePressureJump()
154 );
155
156 fixedValueFvPatchScalarField::updateCoeffs();
157}
