47bool Foam::functionObjects::flowType::calc()
48{
49 if (foundObject<volVectorField>(fieldName_))
50 {
51 const volVectorField& U = lookupObject<volVectorField>(fieldName_);
52 const tmp<volTensorField> tgradU(fvc::grad(U));
53 const volTensorField& gradU = tgradU();
54
55 const volScalarField magD(mag(symm(gradU)));
56 const volScalarField magOmega(mag(skew(gradU)));
57 const dimensionedScalar smallMagD("sMagD", magD.dimensions(), SMALL);
58
59 const volTensorField SSplusWW
60 (
61 (symm(gradU) & symm(gradU))
62 + (skew(gradU) & skew(gradU))
63 );
64
65 return store
66 (
67 resultName_,
68 (magD - magOmega)/(magD + magOmega + smallMagD)
69 );
70 }
71
72 return false;
73}
