43template<class Type>
44Foam::fileName Foam::surfaceWriters::ensightWriter::writeCollated
45(
46 const word& fieldName,
47 const Field<Type>& localValues
48)
49{
50 // Geometry changed since last output? Capture now before any merging.
51 const bool geomChanged = (!upToDate_);
52
53 checkOpen();
54
55 const ensight::FileName surfName(outputPath_.name());
56 const ensight::VarName varName(fieldName);
57
58
59 // Collated
60 // ========
61 // CaseFile: rootdir/surfaceName/surfaceName.case
62 // Geometry: rootdir/surfaceName/data/<index>/geometry
63 // Field: rootdir/surfaceName/data/<index>/field
64
65 // Use surface name as sub-directory for results. Eg,
66 // - SURF1/SURF1.case
67 // - SURF1/data/00000000/geometry
68 // - SURF1/data/00000000/VAR1
69 // - SURF1/data/00000000/VAR2
70
71 // Names "data" and "geometry" as per ensightCase:
72 const int maskWidth = 8;
73 const char* mask = "data/********/";
74
75
76 // Ignore the useTimeDir setting - manage ourselves
77 const fileName baseDir = outputPath_;
78
79 const word timeDir = timeName();
80 const scalar timeValue = currTime_.value();
81
82 const fileName outputFile = baseDir / surfName + ".case";
83
84 if (verbose_)
85 {
86 Info<< "Writing case file to " << outputFile << nl;
87 }
88
89 // Implicit geometry merge()
90 tmp<Field<Type>> tfield = adjustField(fieldName, mergeField(localValues));
91
92 if (verbose_)
93 {
94 Info<< endl;
95 }
96
97 // const meshedSurf& surf = surface();
98 const meshedSurfRef& surf = adjustSurface();
99
100 if (Pstream::master() || !parallel_)
101 {
102 if (!isDir(outputFile.path()))
103 {
104 mkDir(outputFile.path());
105 }
106
107 const bool stateChanged =
108 caching_.update
109 (
110 baseDir,
111 timeValue,
112 geomChanged,
113 fieldName,
114 ensightPTraits<Type>::typeName,
115 varName
116 );
117
118
119 // The most current time and geometry indices
120 const label timeIndex = caching_.latestTimeIndex();
121 const label geomIndex = caching_.latestGeomIndex();
122
123
124 // This will be used for the name of a static geometry,
125 // or just the masking part for moving geometries.
126 const fileName geometryName
127 (
128 "data"
129 / ensightCase::padded(maskWidth, geomIndex)
130 / ensightCase::geometryName
131 );
132
133
134 // Location for data (and possibly the geometry as well)
135 const fileName dataDir
136 (
137 baseDir/"data"/ensightCase::padded(maskWidth, timeIndex)
138 );
139
140 // As per mkdir -p "data/00000000"
141 mkDir(dataDir);
142
143
144 const fileName geomFile(baseDir/geometryName);
145
146 // Ensight Geometry
147 ensightOutputSurface part
148 (
149 surf.points(),
150 surf.faces(),
151 geomFile.name()
152 );
153
154 if (!exists(geomFile))
155 {
156 if (verbose_)
157 {
158 Info<< "Writing geometry to " << geomFile.name() << endl;
159 }
160
161 // Two-argument form for path-name to avoid validating base-dir
162 ensightGeoFile osGeom
163 (
164 geomFile.path(),
165 geomFile.name(),
166 writeFormat_
167 );
168 part.write(osGeom); // serial
169 }
170
171 // Write field
172 ensightFile osField
173 (
174 dataDir,
175 varName,
176 writeFormat_
177 );
178
179 if (verbose_)
180 {
181 Info<< "Writing field file to " << osField.name() << endl;
182 }
183
184 // Write field (serial only)
185 osField.write(ensightPTraits<Type>::typeName);
186 osField.newline();
187 part.writeData(osField, tfield(), this->isPointData());
188
189
190 // Update case file
191 if (stateChanged)
192 {
193 OFstream osCase(outputFile, IOstreamOption::ASCII);
194
195 // Format options
196 osCase.setf(ios_base::left);
197 osCase.setf(ios_base::scientific, ios_base::floatfield);
198 osCase.precision(5);
199
200 if (verbose_)
201 {
202 Info<< "Writing case file to " << osCase.name() << endl;
203 }
204
205 // The geometry can be any of the following:
206 // 0: constant/static
207 // 1: moving, with the same frequency as the data
208 // 2: moving, with different frequency as the data
209
210 const label tsGeom = caching_.geometryTimeset();
211
212 osCase
213 << "FORMAT" << nl
214 << "type: ensight gold" << nl
215 << nl
216 << "GEOMETRY" << nl;
217
218
219 if (tsGeom)
220 {
221 // moving
222 osCase
223 << "model: " << tsGeom << " " // time-set (1|2)
224 << mask << geometryName.name() << nl;
225 }
226 else
227 {
228 // steady
229 osCase
230 << "model: "
231 << geometryName.c_str() << nl;
232 }
233
234 osCase
235 << nl
236 << "VARIABLE" << nl;
237
238
239 for (const entry& dEntry : caching_.fieldsDict())
240 {
241 const dictionary& subDict = dEntry.dict();
242
243 const word varType(subDict.get<word>("type"));
244 const word varName
245 (
246 subDict.getOrDefault<word>
247 (
248 "name",
249 dEntry.keyword() // fieldName as fallback
250 )
251 );
252
253 osCase
254 << varType
255 <<
256 (
257 this->isPointData()
258 ? " per node: 1 " // time-set 1
259 : " per element: 1 " // time-set 1
260 )
261 << setw(15) << varName << ' '
262 << mask << ensight::FileName(varName).c_str() << nl;
263 }
264
265 osCase
266 << nl
267 << "TIME" << nl;
268
269 ensightCase::printTimeset(osCase, 1, caching_.times());
270 if (tsGeom == 2)
271 {
272 ensightCase::printTimeset
273 (
274 osCase,
275 tsGeom,
276 caching_.times(),
277 caching_.geometries()
278 );
279 }
280
281 osCase << "# end" << nl;
282 }
283
284 // Timestamp in the directory for future reference
285 {
286 OFstream timeStamp(dataDir/"time");
287 timeStamp
288 << "# timestep time" << nl
289 << dataDir.name() << ' ' << timeValue << nl;
290 }
291 }
292
293 wroteGeom_ = true;
294 return outputFile;
295}
