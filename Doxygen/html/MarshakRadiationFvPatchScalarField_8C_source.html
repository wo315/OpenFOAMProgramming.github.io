117void Foam::radiation::MarshakRadiationFvPatchScalarField::updateCoeffs()
118{
119 if (this->updated())
120 {
121 return;
122 }
123
124 // Since we're inside initEvaluate/evaluate there might be processor
125 // comms underway. Change the tag we use.
126 int oldTag = UPstream::msgType();
127 UPstream::msgType() = oldTag+1;
128
129 // Temperature field
130 const auto& Tp = patch().lookupPatchField<volScalarField>(TName_);
131
132 // Re-calc reference value
133 refValue() = 4.0*constant::physicoChemical::sigma.value()*pow4(Tp);
134
135 // Diffusion coefficient - created by radiation model's ::updateCoeffs()
136 const auto& gamma = patch().lookupPatchField<volScalarField>("gammaRad");
137
138 const boundaryRadiationProperties& boundaryRadiation =
139 boundaryRadiationProperties::New(internalField().mesh());
140
141 const tmp<scalarField> temissivity
142 (
143 boundaryRadiation.emissivity(patch().index())
144 );
145
146 const scalarField& emissivity = temissivity();
147
148 const scalarField Ep(emissivity/(2.0*(scalar(2) - emissivity)));
149
150 // Set value fraction
151 valueFraction() = 1.0/(1.0 + gamma*patch().deltaCoeffs()/Ep);
152
153 // Restore tag
154 UPstream::msgType() = oldTag;
155
156 mixedFvPatchScalarField::updateCoeffs();
157}
