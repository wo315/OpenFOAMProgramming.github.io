48Foam::dummyAgglomeration::dummyAgglomeration
49(
50 const lduMesh& mesh,
51 const dictionary& controlDict
52)
53:
54 GAMGAgglomeration(mesh, controlDict),
55 nLevels_(controlDict.get<label>("nLevels"))
56{
57 const label nCoarseCells = mesh.lduAddr().size();
58
59 for
60 (
61 label nCreatedLevels = 0;
62 nCreatedLevels < nLevels_;
63 nCreatedLevels++
64 )
65 {
66 nCells_[nCreatedLevels] = nCoarseCells;
67 restrictAddressing_.set
68 (
69 nCreatedLevels,
70 new labelField(identity(nCoarseCells))
71 );
72
73 agglomerateLduAddressing(nCreatedLevels);
74 }
75
76 // Shrink the storage of the levels to those created
77 compactLevels(nLevels_, true);
78}
