39void Foam::syncTools::combine
40(

82template<class T, class CombineOp, class TransformOp>
83void Foam::syncTools::syncPointMap
84(
85 const polyMesh& mesh,
86 Map<T>& pointValues, // from mesh point label to value
87 const CombineOp& cop,
88 const TransformOp& top
89)
90{
91 const polyBoundaryMesh& patches = mesh.boundaryMesh();
92
93 // Synchronize multiple shared points.
94 const globalMeshData& pd = mesh.globalData();
95
96 // Values on shared points. Keyed on global shared index.
97 Map<T> sharedPointValues(0);
98
99 if (pd.nGlobalPoints() > 0)
100 {
101 // meshPoint per local index
102 const labelList& sharedPtLabels = pd.sharedPointLabels();
103
104 // global shared index per local index
105 const labelList& sharedPtAddr = pd.sharedPointAddr();
106
107 sharedPointValues.resize(sharedPtAddr.size());
108
109 // Fill my entries in the shared points
110 forAll(sharedPtLabels, i)
111 {
112 const auto fnd = pointValues.cfind(sharedPtLabels[i]);
113
114 if (fnd.good())
115 {
116 combine
117 (
118 sharedPointValues,
119 cop,
120 sharedPtAddr[i], // index
121 fnd.val() // value
122 );
123 }
124 }
125 }
126
127
128 if (UPstream::parRun())
129 {
130 // Presize according to number of processor patches
131 // (global topology information may not yet be available...)
132 DynamicList<label> neighbProcs(patches.nProcessorPatches());
133 PstreamBuffers pBufs(UPstream::commsTypes::nonBlocking);
134
135 // Sample and send.
136 // Reduce communication by only sending non-zero data,
137 // but with multiply-connected processor/processor
138 // (eg, processorCyclic) also need to send zero information
139 // to keep things synchronised
140
141 // If data needs to be sent (index corresponding to neighbProcs)
142 DynamicList<bool> isActiveSend(neighbProcs.capacity());
143
144 for (const polyPatch& pp : patches)
145 {
146 const auto* ppp = isA<processorPolyPatch>(pp);
147
148 if (ppp && pp.nPoints())
149 {
150 const auto& procPatch = *ppp;
151 const label nbrProci = procPatch.neighbProcNo();
152
153 // Get data per patchPoint in neighbouring point numbers.
154 const labelList& meshPts = procPatch.meshPoints();
155 const labelList& nbrPts = procPatch.neighbPoints();
156
157 // Extract local values. Create map from nbrPoint to value.
158 // Note: how small initial size?
159 Map<T> patchInfo(meshPts.size() / 20);
160
161 forAll(meshPts, i)
162 {
163 const auto iter = pointValues.cfind(meshPts[i]);
164
165 if (iter.good())
166 {
167 patchInfo.insert(nbrPts[i], iter.val());
168 }
169 }
170
171
172 const bool hasSendData = (!patchInfo.empty());
173
174 // Neighbour connectivity (push_uniq)
175 // - record if send is required (non-empty data)
176 {
177 label nbrIndex = neighbProcs.find(nbrProci);
178 if (nbrIndex < 0)
179 {
180 nbrIndex = neighbProcs.size();
181 neighbProcs.push_back(nbrProci);
182 isActiveSend.push_back(false);
183 }
184
185 if (hasSendData)
186 {
187 isActiveSend[nbrIndex] = true;
188 }
189 }
190
191
192 // Send to neighbour
193 {
194 UOPstream toNbr(nbrProci, pBufs);
195 toNbr << patchInfo;
196 }
197 }
198 }
199
200 // Eliminate unnecessary sends
201 forAll(neighbProcs, nbrIndex)
202 {
203 if (!isActiveSend[nbrIndex])
204 {
205 pBufs.clearSend(neighbProcs[nbrIndex]);
206 }
207 }
208
209 // Limit exchange to involved procs
210 pBufs.finishedNeighbourSends(neighbProcs);
211
212
213 // Receive and combine.
214 for (const polyPatch& pp : patches)
215 {
216 const auto* ppp = isA<processorPolyPatch>(pp);
217
218 if (ppp && pp.nPoints())
219 {
220 const auto& procPatch = *ppp;
221 const label nbrProci = procPatch.neighbProcNo();
222
223 if (!pBufs.recvDataCount(nbrProci))
224 {
225 // Nothing to receive
226 continue;
227 }
228
229 Map<T> nbrPatchInfo(0);
230 {
231 UIPstream fromNbr(nbrProci, pBufs);
232 fromNbr >> nbrPatchInfo;
233 }
234
235 // Transform
236 top(procPatch, nbrPatchInfo);
237
238 const labelList& meshPts = procPatch.meshPoints();
239
240 // Only update those values which come from neighbour
241 forAllConstIters(nbrPatchInfo, nbrIter)
242 {
243 combine
244 (
245 pointValues,
246 cop,
247 meshPts[nbrIter.key()],
248 nbrIter.val()
249 );
250 }
251 }
252 }
253 }
254
255 // Do the cyclics.
256 for (const polyPatch& pp : patches)
257 {
258 const cyclicPolyPatch* cpp = isA<cyclicPolyPatch>(pp);
259
260 if (cpp && cpp->owner())
261 {
262 // Owner does all.
263
264 const cyclicPolyPatch& cycPatch = *cpp;
265 const cyclicPolyPatch& nbrPatch = cycPatch.neighbPatch();
266
267 const edgeList& coupledPoints = cycPatch.coupledPoints();
268 const labelList& meshPtsA = cycPatch.meshPoints();
269 const labelList& meshPtsB = nbrPatch.meshPoints();
270
271 // Extract local values. Create map from coupled-edge to value.
272 Map<T> half0Values(meshPtsA.size() / 20);
273 Map<T> half1Values(half0Values.size());
274
275 forAll(coupledPoints, i)
276 {
277 const edge& e = coupledPoints[i];
278
279 const auto point0Fnd = pointValues.cfind(meshPtsA[e[0]]);
280
281 if (point0Fnd.good())
282 {
283 half0Values.insert(i, *point0Fnd);
284 }
285
286 const auto point1Fnd = pointValues.cfind(meshPtsB[e[1]]);
287
288 if (point1Fnd.good())
289 {
290 half1Values.insert(i, *point1Fnd);
291 }
292 }
293
294 // Transform to receiving side
295 top(cycPatch, half1Values);
296 top(nbrPatch, half0Values);
297
298 forAll(coupledPoints, i)
299 {
300 const edge& e = coupledPoints[i];
301
302 const auto half0Fnd = half0Values.cfind(i);
303
304 if (half0Fnd.good())
305 {
306 combine
307 (
308 pointValues,
309 cop,
310 meshPtsB[e[1]],
311 *half0Fnd
312 );
313 }
314
315 const auto half1Fnd = half1Values.cfind(i);
316
317 if (half1Fnd.good())
318 {
319 combine
320 (
321 pointValues,
322 cop,
323 meshPtsA[e[0]],
324 *half1Fnd
325 );
326 }
327 }
328 }
329 }
330
331 // Synchronize multiple shared points.
332 if (pd.nGlobalPoints() > 0)
333 {
334 // meshPoint per local index
335 const labelList& sharedPtLabels = pd.sharedPointLabels();
336 // global shared index per local index
337 const labelList& sharedPtAddr = pd.sharedPointAddr();
338
339 // Reduce on master.
340
341 if (UPstream::parRun())
342 {
343 if (UPstream::master())
344 {
345 // Receive the edges using shared points from other procs
346 for (const int proci : UPstream::subProcs())
347 {
348 IPstream fromProc(UPstream::commsTypes::scheduled, proci);
349 Map<T> nbrValues(fromProc);
350
351 // Merge neighbouring values with my values
352 forAllConstIters(nbrValues, iter)
353 {
354 combine
355 (
356 sharedPointValues,
357 cop,
358 iter.key(), // edge
359 iter.val() // value
360 );
361 }
362 }
363 }
364 else
365 {
366 // Send to master
367 OPstream toMaster
368 (
369 UPstream::commsTypes::scheduled,
370 UPstream::masterNo()
371 );
372 toMaster << sharedPointValues;
373 }
374
375 // Broadcast: send merged values to all
376 Pstream::broadcast(sharedPointValues);
377 }
378
379 // Merge sharedPointValues (keyed on sharedPointAddr) into
380 // pointValues (keyed on mesh points).
381
382 // Map from global shared index to meshpoint
383 Map<label> sharedToMeshPoint(2*sharedPtAddr.size());
384 forAll(sharedPtAddr, i)
385 {
386 sharedToMeshPoint.insert(sharedPtAddr[i], sharedPtLabels[i]);
387 }
388
389 forAllConstIters(sharedToMeshPoint, iter)
390 {
391 // Do I have a value for my shared point
392 const auto sharedFnd = sharedPointValues.cfind(iter.key());
393
394 if (sharedFnd.good())
395 {
396 pointValues.set(iter.val(), sharedFnd.val());
397 }
398 }
399 }
400}

403template<class T, class CombineOp, class TransformOp>
404void Foam::syncTools::syncEdgeMap
405(
406 const polyMesh& mesh,
407 EdgeMap<T>& edgeValues,
408 const CombineOp& cop,
409 const TransformOp& top
410)
411{
412 const polyBoundaryMesh& patches = mesh.boundaryMesh();
413
414
415 // Do synchronisation without constructing globalEdge addressing
416 // (since this constructs mesh edge addressing)
417
418
419 // Swap proc patch info
420 // ~~~~~~~~~~~~~~~~~~~~
421
422 if (UPstream::parRun())
423 {
424 // Presize according to number of processor patches
425 // (global topology information may not yet be available...)
426 DynamicList<label> neighbProcs(patches.nProcessorPatches());
427 PstreamBuffers pBufs(UPstream::commsTypes::nonBlocking);
428
429 // Sample and send.
430 // Reduce communication by only sending non-zero data,
431 // but with multiply-connected processor/processor
432 // (eg, processorCyclic) also need to send zero information
433 // to keep things synchronised
434
435 // If data needs to be sent (index corresponding to neighbProcs)
436 DynamicList<bool> isActiveSend(neighbProcs.capacity());
437
438 for (const polyPatch& pp : patches)
439 {
440 const auto* ppp = isA<processorPolyPatch>(pp);
441
442 if (ppp && pp.nEdges())
443 {
444 const auto& procPatch = *ppp;
445 const label nbrProci = procPatch.neighbProcNo();
446
447 // Get data per patch edge in neighbouring edge.
448 const edgeList& edges = procPatch.edges();
449 const labelList& meshPts = procPatch.meshPoints();
450 const labelList& nbrPts = procPatch.neighbPoints();
451
452 EdgeMap<T> patchInfo(edges.size() / 20);
453
454 for (const edge& e : edges)
455 {
456 const edge meshEdge(meshPts, e);
457
458 const auto iter = edgeValues.cfind(meshEdge);
459
460 if (iter.good())
461 {
462 const edge nbrEdge(nbrPts, e);
463 patchInfo.insert(nbrEdge, iter.val());
464 }
465 }
466
467
468 const bool hasSendData = (!patchInfo.empty());
469
470 // Neighbour connectivity (push_uniq)
471 // and record if send is required (non-empty data)
472 {
473 label nbrIndex = neighbProcs.find(nbrProci);
474 if (nbrIndex < 0)
475 {
476 nbrIndex = neighbProcs.size();
477 neighbProcs.push_back(nbrProci);
478 isActiveSend.push_back(false);
479 }
480
481 if (hasSendData)
482 {
483 isActiveSend[nbrIndex] = true;
484 }
485 }
486
487
488 // Send to neighbour
489 {
490 UOPstream toNbr(nbrProci, pBufs);
491 toNbr << patchInfo;
492 }
493 }
494 }
495
496 // Eliminate unnecessary sends
497 forAll(neighbProcs, nbrIndex)
498 {
499 if (!isActiveSend[nbrIndex])
500 {
501 pBufs.clearSend(neighbProcs[nbrIndex]);
502 }
503 }
504
505 // Limit exchange to involved procs
506 pBufs.finishedNeighbourSends(neighbProcs);
507
508
509 // Receive and combine.
510 for (const polyPatch& pp : patches)
511 {
512 const auto* ppp = isA<processorPolyPatch>(pp);
513
514 if (ppp && pp.nEdges())
515 {
516 const auto& procPatch = *ppp;
517 const label nbrProci = procPatch.neighbProcNo();
518
519 if (!pBufs.recvDataCount(nbrProci))
520 {
521 // Nothing to receive
522 continue;
523 }
524
525 EdgeMap<T> nbrPatchInfo(0);
526 {
527 UIPstream fromNbr(nbrProci, pBufs);
528 fromNbr >> nbrPatchInfo;
529 }
530
531 // Apply transform to convert to this side properties.
532 top(procPatch, nbrPatchInfo);
533
534
535 // Only update those values which come from neighbour
536 const labelList& meshPts = procPatch.meshPoints();
537
538 forAllConstIters(nbrPatchInfo, nbrIter)
539 {
540 const edge& e = nbrIter.key();
541 const edge meshEdge(meshPts[e[0]], meshPts[e[1]]);
542
543 combine
544 (
545 edgeValues,
546 cop,
547 meshEdge, // edge
548 nbrIter.val() // value
549 );
550 }
551 }
552 }
553 }
554
555
556 // Swap cyclic info
557 // ~~~~~~~~~~~~~~~~
558
559 for (const polyPatch& pp : patches)
560 {
561 const cyclicPolyPatch* cpp = isA<cyclicPolyPatch>(pp);
562
563 if (cpp && cpp->owner())
564 {
565 // Owner does all.
566
567 const cyclicPolyPatch& cycPatch = *cpp;
568 const cyclicPolyPatch& nbrPatch = cycPatch.neighbPatch();
569
570 const edgeList& coupledEdges = cycPatch.coupledEdges();
571
572 const labelList& meshPtsA = cycPatch.meshPoints();
573 const edgeList& edgesA = cycPatch.edges();
574
575 const labelList& meshPtsB = nbrPatch.meshPoints();
576 const edgeList& edgesB = nbrPatch.edges();
577
578 // Extract local values. Create map from edge to value.
579 Map<T> half0Values(edgesA.size() / 20);
580 Map<T> half1Values(half0Values.size());
581
582 forAll(coupledEdges, edgei)
583 {
584 const edge& twoEdges = coupledEdges[edgei];
585
586 {
587 const edge& e0 = edgesA[twoEdges[0]];
588 const edge meshEdge0(meshPtsA[e0[0]], meshPtsA[e0[1]]);
589
590 const auto iter = edgeValues.cfind(meshEdge0);
591
592 if (iter.good())
593 {
594 half0Values.insert(edgei, iter.val());
595 }
596 }
597 {
598 const edge& e1 = edgesB[twoEdges[1]];
599 const edge meshEdge1(meshPtsB[e1[0]], meshPtsB[e1[1]]);
600
601 const auto iter = edgeValues.cfind(meshEdge1);
602
603 if (iter.good())
604 {
605 half1Values.insert(edgei, iter.val());
606 }
607 }
608 }
609
610 // Transform to this side
611 top(cycPatch, half1Values);
612 top(nbrPatch, half0Values);
613
614
615 // Extract and combine information
616
617 forAll(coupledEdges, edgei)
618 {
619 const edge& twoEdges = coupledEdges[edgei];
620
621 const auto half1Fnd = half1Values.cfind(edgei);
622
623 if (half1Fnd.good())
624 {
625 const edge& e0 = edgesA[twoEdges[0]];
626 const edge meshEdge0(meshPtsA[e0[0]], meshPtsA[e0[1]]);
627
628 combine
629 (
630 edgeValues,
631 cop,
632 meshEdge0, // edge
633 *half1Fnd // value
634 );
635 }
636
637 const auto half0Fnd = half0Values.cfind(edgei);
638
639 if (half0Fnd.good())
640 {
641 const edge& e1 = edgesB[twoEdges[1]];
642 const edge meshEdge1(meshPtsB[e1[0]], meshPtsB[e1[1]]);
643
644 combine
645 (
646 edgeValues,
647 cop,
648 meshEdge1, // edge
649 *half0Fnd // value
650 );
651 }
652 }
653 }
654 }
655
656 // Synchronize multiple shared points.
657 // Problem is that we don't want to construct shared edges so basically
658 // we do here like globalMeshData but then using sparse edge representation
659 // (EdgeMap instead of mesh.edges())
660
661 const globalMeshData& pd = mesh.globalData();
662 const labelList& sharedPtAddr = pd.sharedPointAddr();
663 const labelList& sharedPtLabels = pd.sharedPointLabels();
664
665 // 1. Create map from meshPoint to globalShared index.
666 Map<label> meshToShared(2*sharedPtLabels.size());
667 forAll(sharedPtLabels, i)
668 {
669 meshToShared.insert(sharedPtLabels[i], sharedPtAddr[i]);
670 }
671
672 // Values on shared points. From two sharedPtAddr indices to a value.
673 EdgeMap<T> sharedEdgeValues(meshToShared.size());
674
675 // From shared edge to mesh edge. Used for merging later on.
676 EdgeMap<edge> potentialSharedEdge(meshToShared.size());
677
678 // 2. Find any edges using two global shared points. These will always be
679 // on the outside of the mesh. (though might not be on coupled patch
680 // if is single edge and not on coupled face)
681 // Store value (if any) on sharedEdgeValues
682 for (label facei = mesh.nInternalFaces(); facei < mesh.nFaces(); ++facei)
683 {
684 const face& f = mesh.faces()[facei];
685
686 forAll(f, fp)
687 {
688 const label v0 = f[fp];
689 const label v1 = f[f.fcIndex(fp)];
690
691 const auto v0Fnd = meshToShared.cfind(v0);
692
693 if (v0Fnd.good())
694 {
695 const auto v1Fnd = meshToShared.cfind(v1);
696
697 if (v1Fnd.good())
698 {
699 const edge meshEdge(v0, v1);
700
701 // edge in shared point labels
702 const edge sharedEdge(*v0Fnd, *v1Fnd);
703
704 // Store mesh edge as a potential shared edge.
705 potentialSharedEdge.insert(sharedEdge, meshEdge);
706
707 const auto edgeFnd = edgeValues.cfind(meshEdge);
708
709 if (edgeFnd.good())
710 {
711 // edge exists in edgeValues. See if already in map
712 // (since on same processor, e.g. cyclic)
713 combine
714 (
715 sharedEdgeValues,
716 cop,
717 sharedEdge, // edge
718 *edgeFnd // value
719 );
720 }
721 }
722 }
723 }
724 }
725
726
727 // Now sharedEdgeValues will contain per potential sharedEdge the value.
728 // (potential since an edge having two shared points is not necessary a
729 // shared edge).
730 // Reduce this on the master.
731
732 if (UPstream::parRun())
733 {
734 if (UPstream::master())
735 {
736 // Receive the edges using shared points from other procs
737 for (const int proci : UPstream::subProcs())
738 {
739 IPstream fromProc(UPstream::commsTypes::scheduled, proci);
740 EdgeMap<T> nbrValues(fromProc);
741
742 // Merge neighbouring values with my values
743 forAllConstIters(nbrValues, iter)
744 {
745 combine
746 (
747 sharedEdgeValues,
748 cop,
749 iter.key(), // edge
750 iter.val() // value
751 );
752 }
753 }
754 }
755 else
756 {
757 // Send to master
758 {
759 OPstream toMaster
760 (
761 UPstream::commsTypes::scheduled,
762 UPstream::masterNo()
763 );
764 toMaster << sharedEdgeValues;
765 }
766 }
767
768 // Broadcast: send merged values to all
769 Pstream::broadcast(sharedEdgeValues);
770 }
771
772
773 // Merge sharedEdgeValues (keyed on sharedPointAddr) into edgeValues
774 // (keyed on mesh points).
775
776 // Loop over all my shared edges.
777 forAllConstIters(potentialSharedEdge, iter)
778 {
779 const edge& sharedEdge = iter.key();
780 const edge& meshEdge = iter.val();
781
782 // Do I have a value for the shared edge?
783 const auto sharedFnd = sharedEdgeValues.cfind(sharedEdge);
784
785 if (sharedFnd.good())
786 {
787 combine
788 (
789 edgeValues,
790 cop,
791 meshEdge, // edge
792 *sharedFnd // value
793 );
794 }
795 }
796}

878template<class T, class CombineOp, class TransformOp, class FlipOp>
879void Foam::syncTools::syncEdgeList
880(
881 const polyMesh& mesh,
882 List<T>& edgeValues,
883 const CombineOp& cop,
884 const T& nullValue,
885 const TransformOp& top,
886 const FlipOp& fop
887)
888{
889 if (edgeValues.size() != mesh.nEdges())
890 {
891 FatalErrorInFunction
892 << "Number of values " << edgeValues.size()
893 << " is not equal to the number of edges in the mesh "
894 << mesh.nEdges() << abort(FatalError);
895 }
896
897 const edgeList& edges = mesh.edges();
898 const globalMeshData& gd = mesh.globalData();
899 const labelList& meshEdges = gd.coupledPatchMeshEdges();
900 const indirectPrimitivePatch& cpp = gd.coupledPatch();

1391 (
1392 pp.start()-boundaryOffset,

1458 label face1 = nbrPatch.start()-boundaryOffset;
1459 for (label i = 0; i < patchSize; ++i)
1460 {
1461 unsigned int val0 = faceValues[face0];
1462 unsigned int val1 = faceValues[face1];
1463
1464 unsigned int t = val0;
1465 cop(t, val1);
1466 faceValues[face0] = t;
1467
1468 cop(val1, val0);
1469 faceValues[face1] = val1;
1470
1471 ++face0;
1472 ++face1;
1473 }
1474 }
1475 }
