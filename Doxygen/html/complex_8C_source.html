83Foam::Istream& Foam::operator>>(Istream& is, complex& c)
84{
85 scalar r, i;
86
87 is.readBegin("complex");
88 is >> r >> i;
89 is.readEnd("complex");
90
91 c.real(r);
92 c.imag(i);
93
