46// Templated implementation for erase() with iterator range.
47// Prefer not to expose directly.
48template<class InputIter>
49static label eraseImpl(objectRegistry& obr, InputIter first, InputIter last)
50{
51 label changed = 0;
52
53 for
54 (
55 const label nTotal = obr.size();
56 changed < nTotal && first != last; // Terminate early
57 ++first
58 )
59 {
60 if (obr.erase(*first))
61 {
62 ++changed;
63 }
64 }
65
66 return changed;
67}

228Foam::label Foam::objectRegistry::getEvent() const
229{
230 label curEvent = event_++;
231
232 if (event_ == labelMax)
233 {
234 if (objectRegistry::debug)
235 {
236 WarningInFunction
237 << "Event counter has overflowed. "
238 << "Resetting counter on all dependent objects." << nl
239 << "This might cause extra evaluations." << endl;
240 }
241
242 // Reset event counter
243 curEvent = 1;
244 event_ = 2;
245
246 // No need to reset dependent objects; overflow is now handled
247 // in regIOobject::upToDate
248 }
249
250 return curEvent;
251}

254bool Foam::objectRegistry::checkIn(regIOobject* io) const
255{
256 if (!io) return false;
257
258 objectRegistry& obr = const_cast<objectRegistry&>(*this);
259
260 if (objectRegistry::debug)
261 {
262 Pout<< "objectRegistry::checkIn : "
263 << name() << " : checking in " << io->name()
264 << " of type " << io->type()
265 << endl;
266 }
267
268 // Delete cached object if it has the same name as io, and in the
269 // cacheTemporaryObjects list
270 if (!cacheTemporaryObjects_.empty())
271 {
272 auto cacheIter = cacheTemporaryObjects_.find(io->name());
273
274 if (cacheIter.good())
275 {
276 iterator iter = obr.find(io->name());
277
278 if
279 (
280 iter.good()
281 && iter.val() != io
282 && iter.val()->ownedByRegistry()
283 )
284 {
285 if (objectRegistry::debug)
286 {
287 Pout<< "objectRegistry::checkIn : "
288 << name() << " : deleting cached object "
289 << io->name() << endl;
290 }
291
292 cacheIter.val().first() = false;
293 deleteCachedObject(iter.val());
294 }
295 }
296 }
297
298
299 bool ok = obr.insert(io->name(), io);
300
301 if (!ok && objectRegistry::debug)
302 {
303 WarningInFunction
304 << name() << " : Attempt to checkIn object with name "
305 << io->name() << " which was already checked in"
306 << endl;
307 }
308
309 return ok;
310}

313bool Foam::objectRegistry::checkOut(regIOobject* io) const
314{
315 if (!io) return false;
316
317 objectRegistry& obr = const_cast<objectRegistry&>(*this);
318
319 iterator iter = obr.find(io->name());
320
321 if (iter.good())
322 {
323 if (objectRegistry::debug)
324 {
325 Pout<< "objectRegistry::checkOut : "
326 << name() << " : checking out " << io->name()
327 << " of type " << io->type()
328 << endl;
329 }
330
331 if (iter.val() != io)
332 {
333 if (objectRegistry::debug)
334 {
335 WarningInFunction
336 << name() << " : Attempt to checkOut copy of "
337 << iter.key()
338 << endl;
339 }
340
341 return false;
342 }
343
344 return obr.erase(iter);
345 }
346
347
348 if (objectRegistry::debug)
349 {
350 Pout<< "objectRegistry::checkOut : "
351 << name() << " : could not find " << io->name() << " in registry"
352 << endl;
353 }
354
355 return false;
356}

377void Foam::objectRegistry::clear()
378{
379 // Free anything owned by the registry, but first unset both
380 // 'ownedByRegistry' and 'registered' flags to ensure that the
381 // regIOobject destructor will not affect the registry
382
383 for (iterator iter = begin(); iter != end(); ++iter)
384 {
385 regIOobject* ptr = iter.val();
386
387 if (ptr && ptr->ownedByRegistry())
388 {
389 if (objectRegistry::debug)
390 {
391 Pout<< "objectRegistry::clear : " << ptr->name() << nl;
392 }
393
394 ptr->release(true); // Relinquish ownership and registration
395 delete ptr; // Delete also clears fileHandler watches
396 }
397 }
398
399 HashTable<regIOobject*>::clear();
400}

410bool Foam::objectRegistry::erase(const iterator& iter)
411{
412 // Remove from registry - see notes in objectRegistry::clear()
413
414 if (iter.good())
415 {
416 regIOobject* ptr = const_cast<iterator&>(iter).val();
417
418 const bool ok = HashTable<regIOobject*>::erase(iter);
419
420 if (ptr && ptr->ownedByRegistry())
421 {
422 ptr->release(true); // Relinquish ownership and registration
423 delete ptr; // Delete also clears fileHandler watches
424 }
425
426 return ok;
427 }
428
429 return false;
430}

470(
471 const word& name,
472 const bool recursive
473) const
474{
475 const_iterator iter = cfind(name);
476

540 const bool writeOnProc
541) const
542{
543 bool ok = true;
544
545 for (const_iterator iter = cbegin(); iter != cend(); ++iter)
546 {
547 if (objectRegistry::debug)
548 {
549 const regIOobject& obj = *iter.val();
550
551 Pout<< "objectRegistry::write() : "
552 << name() << " : Considering writing object "
553 << iter.key() << " of type "
554 << obj.type() << " with writeOpt "
555 << static_cast<int>(obj.writeOpt())
556 << " to file " << obj.objectRelPath() << endl;
557 }
558
559 if (iter.val()->writeOpt() != IOobjectOption::NO_WRITE)
560 {
561 ok = iter.val()->writeObject(streamOpt, writeOnProc) && ok;
562 }
563 }
