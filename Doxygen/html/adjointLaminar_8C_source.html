101tmp<fvVectorMatrix> adjointLaminar::divDevReff(volVectorField& U) const
102{
103 return
104 (
105 - fvm::laplacian(nuEff(), U)
106 - fvc::div(nuEff()*dev(T(fvc::grad(U))))
107 );
108}
