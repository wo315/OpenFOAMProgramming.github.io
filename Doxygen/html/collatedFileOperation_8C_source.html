85void Foam::fileOperations::collatedFileOperation::printBanner
86(
87 const bool withRanks
88) const
89{
90 DetailInfo
91 << "I/O : " << this->type();
92
93 if (maxThreadFileBufferSize == 0)
94 {
95 DetailInfo
96 << " [unthreaded] (maxThreadFileBufferSize = 0)." << nl
97 << " Writing may be slow for large file sizes."
98 << endl;
99 }
100 else
101 {
102 DetailInfo
103 << " [threaded] (maxThreadFileBufferSize = "
104 << maxThreadFileBufferSize << ")." << nl
105 << " Requires buffer large enough to collect all data"
106 " or thread support" << nl
107 << " enabled in MPI. If MPI thread support cannot be"
108 " enabled, deactivate" << nl
109 << " threading by setting maxThreadFileBufferSize"
110 " to 0 in" << nl
111 << " OpenFOAM etc/controlDict" << endl;
112 }
113
114 if (withRanks)
115 {
116 // Information about the ranks
117 stringList hosts(Pstream::nProcs());
118 if (Pstream::master(comm_))
119 {
120 hosts[Pstream::myProcNo()] = hostName();
121 }
122 Pstream::gatherList(hosts);
123
124 DynamicList<label> offsetMaster(Pstream::nProcs());
125
126 forAll(hosts, ranki)
127 {
128 if (!hosts[ranki].empty())
129 {
130 offsetMaster.append(ranki);
131 }
132 }
133
134 if (offsetMaster.size() > 1)
135 {
136 DetailInfo
137 << "IO nodes:" << nl << '(' << nl;
138
139 offsetMaster.append(Pstream::nProcs());
140
141 for (label group = 1; group < offsetMaster.size(); ++group)
142 {
143 const label beg = offsetMaster[group-1];
144 const label end = offsetMaster[group];
145
146 DetailInfo
147 << " (" << hosts[beg].c_str() << ' '
148 << (end-beg) << ')' << nl;
149 }
150 DetailInfo
151 << ')' << nl;
152 }
153 }
154
155 //- fileModificationChecking already set by base class (masterUncollated)
156 // if (IOobject::fileModificationChecking == IOobject::timeStampMaster)
157 // {
158 // WarningInFunction
159 // << "Resetting fileModificationChecking to timeStamp" << endl;
160 // }
161 // else if (IOobject::fileModificationChecking == IOobject::inotifyMaster)
162 // {
163 // WarningInFunction
164 // << "Resetting fileModificationChecking to inotify" << endl;
165 // }

169bool Foam::fileOperations::collatedFileOperation::appendObject
170(
171 const regIOobject& io,
172 const fileName& pathName,
173 IOstreamOption streamOpt
174) const
175{
176 // Append to processorsNN/ file
177
178 const label proci = detectProcessorPath(io.objectPath());
179
180 if (debug)
181 {
182 Pout<< "collatedFileOperation::writeObject :"
183 << " For local object : " << io.name()
184 << " appending processor " << proci
185 << " data to " << pathName << endl;
186 }
187 if (proci == -1)
188 {
189 FatalErrorInFunction
190 << "Invalid processor path: " << pathName
191 << exit(FatalError);
192 }
193
194 const bool isIOmaster = fileOperation::isIOrank(proci);
195
196 // Update meta-data for current state
197 if (isIOmaster)
198 {
199 const_cast<regIOobject&>(io).updateMetaData();
200 }
201
202 // Note: cannot do append + compression. This is a limitation
203 // of ogzstream (or rather most compressed formats)
204 //
205 // File should always be created as non-atomic
206 // (consistency between append/non-append)
207
208 OFstream os
209 (
210 pathName,
211 // UNCOMPRESSED (binary only)
212 IOstreamOption(IOstreamOption::BINARY, streamOpt.version()),
213 // Append on sub-ranks
214 (isIOmaster ? IOstreamOption::NON_APPEND : IOstreamOption::APPEND)
215 );
216
217 if (!os.good())
218 {
219 FatalIOErrorInFunction(os)
220 << "Cannot open for appending"
221 << exit(FatalIOError);
222 }
223
224 if (isIOmaster)
225 {
226 decomposedBlockData::writeHeader(os, streamOpt, io);
227 }
228
229 std::streamoff blockOffset = decomposedBlockData::writeBlockEntry
230 (
231 os,
232 streamOpt,
233 io,
234 proci,
235 isIOmaster // With FoamFile header on master
236 );
237
238 return (blockOffset >= 0) && os.good();
239}

247// Construction helper: self/world/local communicator and IO ranks
248static Tuple2<label, labelList> getCommPattern()
249{
250 // Default is COMM_WORLD (single master)
251 Tuple2<label, labelList> commAndIORanks
252 (
253 UPstream::worldComm,
254 fileOperation::getGlobalIORanks()
255 );
256
257 if (UPstream::parRun() && commAndIORanks.second().size() > 1)
258 {
259 // Multiple masters: ranks for my IO range
260 commAndIORanks.first() = UPstream::allocateCommunicator
261 (
262 UPstream::worldComm,
263 fileOperation::subRanks(commAndIORanks.second())
264 );
265 }
266
267 return commAndIORanks;
268}

374bool Foam::fileOperations::collatedFileOperation::writeObject
375(
376 const regIOobject& io,
377 IOstreamOption streamOpt,
378 const bool writeOnProc
379) const
380{
381 const Time& tm = io.time();
382 const fileName& inst = io.instance();
383
384 // Update meta-data for current state
385 const_cast<regIOobject&>(io).updateMetaData();
386
387 if (inst.isAbsolute() || !tm.processorCase())
388 {
389 mkDir(io.path());
390 fileName pathName(io.objectPath());
391
392 if (debug)
393 {
394 Pout<< "collatedFileOperation::writeObject :"
395 << " For object : " << io.name()
396 << " falling back to master-only output to " << io.path()
397 << endl;
398 }
399
400 // Note: currently still NON_ATOMIC (Dec-2022)
401 masterOFstream os
402 (
403 comm_,
404 pathName,
405 streamOpt,
406 IOstreamOption::NON_APPEND,
407 writeOnProc
408 );
409
410 // If any of these fail, return
411 // (leave error handling to Ostream class)
412
413 const bool ok =
414 (
415 os.good()
416 && io.writeHeader(os)
417 && io.writeData(os)
418 );
419
420 if (ok)
421 {
422 IOobject::writeEndDivider(os);
423 }
424
425 return ok;
426 }
427 else
428 {
429 // Construct the equivalent processors/ directory
430 fileName path(processorsPath(io, inst, processorsDir(io)));
431
432 mkDir(path);
433 fileName pathName(path/io.name());
434
435 if (io.global() || io.globalObject())
436 {
437 if (debug)
438 {
439 Pout<< "collatedFileOperation::writeObject :"
440 << " For global object : " << io.name()
441 << " falling back to master-only output to " << pathName
442 << endl;
443 }
444
445 // Note: currently still NON_ATOMIC (Dec-2022)
446 masterOFstream os
447 (
448 comm_,
449 pathName,
450 streamOpt,
451 IOstreamOption::NON_APPEND,
452 writeOnProc
453 );
454
455 // If any of these fail, return
456 // (leave error handling to Ostream class)
457
458 const bool ok =
459 (
460 os.good()
461 && io.writeHeader(os)
462 && io.writeData(os)
463 );
464
465 if (ok)
466 {
467 IOobject::writeEndDivider(os);
468 }
469
470 return ok;
471 }
472 else if (!UPstream::parRun())
473 {
474 // Special path for e.g. decomposePar. Append to
475 // processorsDDD/ file
476 if (debug)
477 {
478 Pout<< "collatedFileOperation::writeObject :"
479 << " For object : " << io.name()
480 << " appending to " << pathName << endl;
481 }
482
483 return appendObject(io, pathName, streamOpt);
484 }
485 else
486 {
487 // Re-check static maxThreadFileBufferSize variable to see
488 // if needs to use threading
489 const bool useThread = (maxThreadFileBufferSize != 0);
490
491 if (debug)
492 {
493 Pout<< "collatedFileOperation::writeObject :"
494 << " For object : " << io.name()
495 << " starting collating output to " << pathName
496 << " useThread:" << useThread << endl;
497 }
498
499 if (!useThread)
500 {
501 writer_.waitAll();
502 }
503
504 // Note: currently still NON_ATOMIC (Dec-2022)
505 threadedCollatedOFstream os
506 (
507 writer_,
508 pathName,
509 streamOpt,
510 useThread
511 );
512
513 bool ok = os.good();
514
515 if (UPstream::master(comm_))
516 {
517 // Suppress comment banner
518 const bool old = IOobject::bannerEnabled(false);
519
520 ok = ok && io.writeHeader(os);
521
522 IOobject::bannerEnabled(old);
523
524 // Additional header content
525 dictionary dict;
526 decomposedBlockData::writeExtraHeaderContent
527 (
528 dict,
529 streamOpt,
530 io
531 );
532 os.setHeaderEntries(dict);
533 }
534
535 ok = ok && io.writeData(os);
536 // No end divider for collated output
537
538 return ok;
539 }
540 }

556Foam::word Foam::fileOperations::collatedFileOperation::processorsDir
557(
558 const fileName& fName
559) const
560{
561 if (UPstream::parRun())
562 {
563 const List<int>& procs(UPstream::procID(comm_));
564
565 word procDir(processorsBaseDir+Foam::name(nProcs_));
566
567 if (procs.size() != nProcs_)
568 {
569 procDir +=
570 + "_"
571 + Foam::name(procs.first())
572 + "-"
573 + Foam::name(procs.last());
574 }
575 return procDir;
576 }
577 else
578 {
579 word procDir(processorsBaseDir+Foam::name(nProcs_));
580
581 if (ioRanks_.size())
582 {
583 // Detect current processor number
584 label proci = detectProcessorPath(fName);
585
586 if (proci != -1)
587 {
588 // Find lowest io rank
589 label minProc = 0;
590 label maxProc = nProcs_-1;
591 for (const label ranki : ioRanks_)
592 {
593 if (ranki >= nProcs_)
594 {
595 break;
596 }
597 else if (ranki <= proci)
598 {
599 minProc = ranki;
600 }
601 else
602 {
603 maxProc = ranki-1;
604 break;
605 }
606 }
607 procDir +=
608 + "_"
609 + Foam::name(minProc)
610 + "-"
611 + Foam::name(maxProc);
612 }
613 }
614
615 return procDir;
616 }
