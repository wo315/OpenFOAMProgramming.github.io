187void ATCModel::computeLimiter
188(
189 volScalarField& limiter,
190 const labelList& cells,
191 const label nSmooth
192)
193{
194 // Restore values
195 limiter.primitiveFieldRef() = 1;
196 limiter.correctBoundaryConditions();
197
198 // Set to zero in predefined cells
199 for (const label celli : cells)
200 {
201 limiter[celli] = Zero;
202 }
203
204 // Correct bcs to get the correct value for boundary faces
205 limiter.correctBoundaryConditions();
206
207 // Apply "laplacian" smoother
208 const fvMesh& mesh = limiter.mesh();
209 const localMin<scalar> scheme(mesh);
210 for (label iLimit = 0; iLimit < nSmooth; ++iLimit)
211 {
212 surfaceScalarField faceLimiter
213 (
214 scheme.interpolate(limiter)
215 );
216 limiter = fvc::average(faceLimiter);
217 }
218}

221tmp<volScalarField> ATCModel::createLimiter
222(
223 const fvMesh& mesh,
224 const dictionary& dict
225)
226{
227 autoPtr<zeroATCcells> zeroType(zeroATCcells::New(mesh, dict));
228 const labelList& zeroCells = zeroType->getZeroATCcells();
229 const label nSmooth = dict.getOrDefault<label>("nSmooth", 0);
230
231 tmp<volScalarField> tlimiter
232 (
233 new volScalarField
234 (
235 IOobject
236 (
237 "limiter",
238 mesh.time().timeName(),
239 mesh,
240 IOobject::NO_READ,
241 IOobject::NO_WRITE
242 ),
243 mesh,
244 scalar(1),
245 dimless,
246 fvPatchFieldBase::zeroGradientType()
247 )
248 );
249 volScalarField& limiter = tlimiter.ref();
250
251 computeLimiter(limiter, zeroCells, nSmooth);
252
253 return tlimiter;
254}
