34Foam::extendedFaceToCellStencil::extendedFaceToCellStencil(const polyMesh& mesh)
35:
36 mesh_(mesh)
37{
38 // Check for transformation - not supported.
39 const polyBoundaryMesh& patches = mesh.boundaryMesh();
40
41 forAll(patches, patchi)
42 {
43 if (patches[patchi].coupled())
44 {
45 const coupledPolyPatch& cpp =
46 refCast<const coupledPolyPatch>(patches[patchi]);
47
48 if (!cpp.parallel() || cpp.separated())
49 {
50 FatalErrorInFunction
51 << "Coupled patches with transformations not supported."
52 << endl
53 << "Problematic patch " << cpp.name() << exit(FatalError);
54 }
55 }
56 }
57}
