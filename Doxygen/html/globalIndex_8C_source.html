33void Foam::globalIndex::reportOverflowAndExit
34(
35 const label idx,
36 const labelUList& localLens
37)
38{
39 FatalErrorInFunction
40 << "Overflow : sum of sizes exceeds labelMax ("
41 << labelMax << ") after index " << idx;
42
43 if (!localLens.empty())
44 {
45 FatalError << " of " << flatOutput(localLens);
46 }
47
48 FatalError
49 << nl
50 << "Please recompile with larger datatype for label." << nl
51 << exit(FatalError);
52}

55Foam::labelList
56Foam::globalIndex::calcOffsets
57(
58 const labelUList& localLens,
59 const bool checkOverflow
60)
61{
62 labelList values;
63
64 const label len = localLens.size();
65
66 if (len)
67 {
68 values.resize(len+1);
69
70 label start = 0;
71 for (label i = 0; i < len; ++i)
72 {
73 values[i] = start;
74 start += localLens[i];
75
76 if (checkOverflow && start < values[i])
77 {
78 reportOverflowAndExit(i, localLens);
79 }
80 }
81 values[len] = start;
82 }
83
84 return values;
85}

130Foam::CompactListList<Foam::label>
131Foam::globalIndex::bin
132(
133 const labelUList& offsets,
134 const labelUList& globalIds,
135 labelList& order,
136 DynamicList<label>& validBins
137)
138{
139 sortedOrder(globalIds, order);
140 validBins.clear();
141
142 CompactListList<label> bins;
143
144 if (globalIds.size())
145 {
146 labelList& binOffsets = bins.offsets();
147 binOffsets.resize(offsets.size(), Zero);
148
149 labelList& binValues = bins.values();
150 binValues = UIndirectList<label>(globalIds, order);
151
152 const label id = binValues[0];
153 label proci = findLower(offsets, id+1);
154
155 validBins.append(proci);
156 label binSize = 1;
157
158 for (label i = 1; i < order.size(); i++)
159 {
160 const label id = binValues[i];
161
162 if (id < offsets[proci+1])
163 {
164 binSize++;
165 }
166 else
167 {
168 // Not local. Reset proci
169 label oldProci = proci;
170 proci = findLower(offsets, id+1);
171
172 // Set offsets
173 for (label j = oldProci+1; j < proci; ++j)
174 {
175 binOffsets[j] = binOffsets[oldProci]+binSize;
176 }
177 binOffsets[proci] = i;
178 validBins.append(proci);
179 binSize = 1;
180 }
181 }
182
183 for (label j = proci+1; j < binOffsets.size(); ++j)
184 {
185 binOffsets[j] = binOffsets[proci]+binSize;
186 }
187 }
188
189 return bins;
190}

193void Foam::globalIndex::reset
194(
195 const label localSize,
196 const label comm,
197 const bool parallel
198)
199{
200 labelList localLens;
201
202 const label len = UPstream::nProcs(comm);
203
204 if (len)
205 {
206 if (parallel && UPstream::parRun()) // or UPstream::is_parallel()
207 {
208 localLens = UPstream::allGatherValues(localSize, comm);
209 }
210 else
211 {
212 // Non-parallel branch: use localSize on-proc, zero elsewhere
213
214 localLens.resize(len, Zero);
215 localLens[UPstream::myProcNo(comm)] = localSize;
216 }
217
218 reset(localLens, true); // checkOverflow = true
219 }
220 else
221 {
222 // Nothing to do
223 offsets_.clear();
224 }
225}

231 const bool checkOverflow
232)
233{
234 const label len = localLens.size();
235
236 if (len)
237 {
238 offsets_.resize_nocopy(len+1);
239
240 label start = 0;
241 for (label i = 0; i < len; ++i)
242 {
243 offsets_[i] = start;
244 start += localLens[i];
245
246 if (checkOverflow && start < offsets_[i])

249 }
250 }

252 }

327Foam::label Foam::globalIndex::maxNonLocalSize(const label proci) const
328{
329 const label len = (offsets_.size() - 1);
330
331 if (len < 1)
332 {
333 return 0;
334 }
335
336 label maxLen = 0;
337
338 for (label i=0; i < len; ++i)
339 {
340 if (i != proci)
341 {
342 const label localLen = (offsets_[i+1] - offsets_[i]);
343 maxLen = max(maxLen, localLen);
344 }
345 }
346
347 return maxLen;
348}
