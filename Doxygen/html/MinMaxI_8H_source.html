218template<class T>
219inline int Foam::MinMax<T>::compare(const T& val) const
220{
221 if (good())
222 {
223 if (max() < val)
224 {
225 return -1;
226 }
227 else if (val < min())
228 {
229 return 1;
230 }
231 }
232
233 return 0;
234}
