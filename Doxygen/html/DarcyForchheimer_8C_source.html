76void Foam::porosityModels::DarcyForchheimer::calcTransformModelData()
77{
78 // The Darcy coefficient as a tensor
79 tensor darcyCoeff(Zero);
80 darcyCoeff.xx() = dXYZ_.value().x();
81 darcyCoeff.yy() = dXYZ_.value().y();
82 darcyCoeff.zz() = dXYZ_.value().z();
83
84 // The Forchheimer coefficient as a tensor
85 // - the leading 0.5 is from 1/2*rho
86 tensor forchCoeff(Zero);
87 forchCoeff.xx() = 0.5*fXYZ_.value().x();
88 forchCoeff.yy() = 0.5*fXYZ_.value().y();
89 forchCoeff.zz() = 0.5*fXYZ_.value().z();
90
91 if (csys().uniform())
92 {
93 forAll(cellZoneIDs_, zonei)
94 {
95 D_[zonei].resize(1);
96 F_[zonei].resize(1);
97
98 D_[zonei] = csys().transform(darcyCoeff);
99 F_[zonei] = csys().transform(forchCoeff);
100 }
101 }
102 else
103 {
104 forAll(cellZoneIDs_, zonei)
105 {
106 const pointUIndList cc
107 (
108 mesh_.cellCentres(),
109 mesh_.cellZones()[cellZoneIDs_[zonei]]
110 );
111
112 D_[zonei] = csys().transform(cc, darcyCoeff);
113 F_[zonei] = csys().transform(cc, forchCoeff);
114 }
115 }
116
117
118 if
119 (
120 debug
121 && (
122 mesh_.time().writeTime()
123 || mesh_.time().timeIndex() == mesh_.time().startTimeIndex()
124 )
125 )
126 {
127 volTensorField Dout
128 (
129 IOobject
130 (
131 typeName + ":D",
132 mesh_.time().timeName(),
133 mesh_,
134 IOobject::NO_READ,
135 IOobject::NO_WRITE
136 ),
137 mesh_,
138 dimensionedTensor(dXYZ_.dimensions(), Zero)
139 );
140 volTensorField Fout
141 (
142 IOobject
143 (
144 typeName + ":F",
145 mesh_.time().timeName(),
146 mesh_,
147 IOobject::NO_READ,
148 IOobject::NO_WRITE
149 ),
150 mesh_,
151 dimensionedTensor(fXYZ_.dimensions(), Zero)
152 );
153
154
155 forAll(cellZoneIDs_, zonei)
156 {
157 const labelList& cells = mesh_.cellZones()[cellZoneIDs_[zonei]];
158
159 if (csys().uniform())
160 {
161 UIndirectList<tensor>(Dout, cells) = D_[zonei].first();
162 UIndirectList<tensor>(Fout, cells) = F_[zonei].first();
163 }
164 else
165 {
166 UIndirectList<tensor>(Dout, cells) = D_[zonei];
167 UIndirectList<tensor>(Fout, cells) = F_[zonei];
168 }
169 }
170
171 Dout.write();
172 Fout.write();
173 }
174}

177void Foam::porosityModels::DarcyForchheimer::calcForce
178(
179 const volVectorField& U,
180 const volScalarField& rho,
181 const volScalarField& mu,
182 vectorField& force
183) const
184{
185 scalarField Udiag(U.size(), Zero);
186 vectorField Usource(U.size(), Zero);
187 const scalarField& V = mesh_.V();
188
189 apply(Udiag, Usource, V, rho, mu, U);
190
191 force = Udiag*U - Usource;
192}

195void Foam::porosityModels::DarcyForchheimer::correct
196(
197 fvVectorMatrix& UEqn
198) const
199{
200 const volVectorField& U = UEqn.psi();
201 const scalarField& V = mesh_.V();
202 scalarField& Udiag = UEqn.diag();
203 vectorField& Usource = UEqn.source();
204
205 word rhoName(IOobject::groupName(rhoName_, U.group()));
206 word muName(IOobject::groupName(muName_, U.group()));
207 word nuName(IOobject::groupName(nuName_, U.group()));
208
209 if (UEqn.dimensions() == dimForce)
210 {
211 const auto& rho = mesh_.lookupObject<volScalarField>(rhoName);
212
213 if (mesh_.foundObject<volScalarField>(muName))
214 {
215 const auto& mu = mesh_.lookupObject<volScalarField>(muName);
216
217 apply(Udiag, Usource, V, rho, mu, U);
218 }
219 else
220 {
221 const auto& nu = mesh_.lookupObject<volScalarField>(nuName);
222
223 apply(Udiag, Usource, V, rho, rho*nu, U);
224 }
225 }
226 else
227 {
228 if (mesh_.foundObject<volScalarField>(nuName))
229 {
230 const auto& nu = mesh_.lookupObject<volScalarField>(nuName);
231
232 apply(Udiag, Usource, V, geometricOneField(), nu, U);
233 }
234 else
235 {
236 const auto& rho = mesh_.lookupObject<volScalarField>(rhoName);
237 const auto& mu = mesh_.lookupObject<volScalarField>(muName);
238
239 apply(Udiag, Usource, V, geometricOneField(), mu/rho, U);
240 }
241 }
242}
