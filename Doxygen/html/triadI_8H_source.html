82inline Foam::direction Foam::triad::primaryDirection(const vector& v)
83{
84 if
85 (
86 Foam::mag(v.x()) > Foam::mag(v.y())
87 && Foam::mag(v.x()) > Foam::mag(v.z())
88 )
89 {
90 return triad::X;
91 }
92 else if (Foam::mag(v.y()) > Foam::mag(v.z()))
93 {
94 return triad::Y;
95 }
96 else
97 {
98 return triad::Z;
99 }
100}

103inline Foam::vector Foam::triad::orthogonal
104(
105 const vector& v1,
106 const vector& v2
107)
108{
109 vector v3 = v1 ^ v2;
110
111 scalar magV3 = Foam::mag(v3);
112
113 if (magV3 > 0.5)
114 {
115 return v3/magV3;
116 }
117 else
118 {
119 return triad::unset[0];
120 }
121}
