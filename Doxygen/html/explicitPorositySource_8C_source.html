107bool Foam::fv::explicitPorositySource::read(const dictionary& dict)
108{
109 if (fv::cellSetOption::read(dict))
110 {
111 if (!coeffs_.readIfPresent("UNames", fieldNames_))
112 {
113 fieldNames_.resize(1);
114 fieldNames_.first() = coeffs_.getOrDefault<word>("U", "U");
115 }
116
117 fv::option::resetApplied();
118
119 return true;
120 }
121
122 return false;
123}
