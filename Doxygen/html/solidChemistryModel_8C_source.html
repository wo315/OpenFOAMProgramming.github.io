115template<class CompType, class SolidThermo>
116Foam::tmp<Foam::volScalarField>
117Foam::solidChemistryModel<CompType, SolidThermo>::Qdot() const
118{
119 tmp<volScalarField> tQdot
120 (
121 new volScalarField
122 (
123 IOobject
124 (
125 "Qdot",
126 this->mesh_.time().timeName(),
127 this->mesh_,
128 IOobject::NO_READ,
129 IOobject::AUTO_WRITE,
130 IOobject::NO_REGISTER
131 ),
132 this->mesh_,
133 dimensionedScalar(dimEnergy/dimVolume/dimTime, Zero)
134 )
135 );
136
137 if (this->chemistry_)
138 {
139 scalarField& Qdot = tQdot.ref();
140
141 forAll(Ys_, i)
142 {
143 forAll(Qdot, celli)
144 {
145 scalar hf = solidThermo_[i].Hc();
146 Qdot[celli] -= hf*RRs_[i][celli];
147 }
148 }
149 }
150
151 return tQdot;
152}
