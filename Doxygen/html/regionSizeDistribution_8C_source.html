60template<class Type>
61static Map<Type> regionSum(const regionSplit& regions, const Field<Type>& fld)
62{
63 // Per region the sum of fld
64 Map<Type> regionToSum(regions.nRegions()/Pstream::nProcs());
65
66 forAll(fld, celli)
67 {
68 const label regioni = regions[celli];
69 regionToSum(regioni, Type(Zero)) += fld[celli];
70 }
71
72 Pstream::mapCombineReduce(regionToSum, plusEqOp<Type>());
73
74 return regionToSum;
75}

77
78template<class Type>
79static List<Type> extractData(const labelUList& keys, const Map<Type>& regionData)
80{
81 List<Type> sortedData(keys.size());
82
83 forAll(keys, i)
84 {
85 sortedData[i] = regionData[keys[i]];
86 }
87 return sortedData;
88}

95void Foam::functionObjects::regionSizeDistribution::writeAlphaFields
96(
97 const regionSplit& regions,
98 const Map<label>& patchRegions,
99 const Map<scalar>& regionVolume,
100 const volScalarField& alpha
101) const
102{
103 const scalar maxDropletVol = 1.0/6.0*mathematical::pi*pow3(maxDiam_);
104
105 // Split alpha field
106 // ~~~~~~~~~~~~~~~~~
107 // Split into
108 // - liquidCore : region connected to inlet patches
109 // - per region a volume : for all other regions
110 // - backgroundAlpha : remaining alpha
111
112
113 // Construct field
114 volScalarField liquidCore
115 (
116 IOobject
117 (
118 scopedName(alphaName_ + "_liquidCore"),
119 obr_.time().timeName(),
120 obr_,
121 IOobject::NO_READ
122 ),
123 alpha,
124 fvPatchFieldBase::calculatedType()
125 );
126
127 volScalarField backgroundAlpha
128 (
129 IOobject
130 (
131 scopedName(alphaName_ + "_background"),
132 obr_.time().timeName(),
133 obr_,
134 IOobject::NO_READ
135 ),
136 alpha,
137 fvPatchFieldBase::calculatedType()
138 );
139
140
141 // Knock out any cell not in patchRegions
142 forAll(liquidCore, celli)
143 {
144 const label regioni = regions[celli];
145 if (patchRegions.found(regioni))
146 {
147 backgroundAlpha[celli] = 0;
148 }
149 else
150 {
151 liquidCore[celli] = 0;
152
153 const scalar regionVol = regionVolume[regioni];
154 if (regionVol < maxDropletVol)
155 {
156 backgroundAlpha[celli] = 0;
157 }
158 }
159 }
160 liquidCore.correctBoundaryConditions();
161 backgroundAlpha.correctBoundaryConditions();
162
163 if (log)
164 {
165 Info<< " Volume of liquid-core = "
166 << fvc::domainIntegrate(liquidCore).value()
167 << nl
168 << " Volume of background = "
169 << fvc::domainIntegrate(backgroundAlpha).value()
170 << endl;
171 }
172
173 Log << " Writing liquid-core field to " << liquidCore.name() << endl;
174 liquidCore.write();
175
176 Log<< " Writing background field to " << backgroundAlpha.name() << endl;
177 backgroundAlpha.write();
178}

181Foam::Map<Foam::label>
182Foam::functionObjects::regionSizeDistribution::findPatchRegions
183(
184 const regionSplit& regions
185) const
186{
187 // Mark all regions starting at patches
188 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
189
190 // Count number of patch faces (just for initial sizing)
191 const labelHashSet patchIDs(mesh_.boundaryMesh().patchSet(patchNames_));
192
193 label nPatchFaces = 0;
194 for (const label patchi : patchIDs)
195 {
196 nPatchFaces += mesh_.boundaryMesh()[patchi].size();
197 }
198
199
200 Map<label> patchRegions(nPatchFaces);
201 for (const label patchi : patchIDs)
202 {
203 const polyPatch& pp = mesh_.boundaryMesh()[patchi];
204
205 // Collect all regions on the patch
206 const labelList& faceCells = pp.faceCells();
207
208 for (const label celli : faceCells)
209 {
210 patchRegions.insert
211 (
212 regions[celli],
213 Pstream::myProcNo() // dummy value
214 );
215 }
216 }
217
218
219 // Make sure all the processors have the same set of regions
220 Pstream::mapCombineReduce(patchRegions, minEqOp<label>());
221
222 return patchRegions;
223}

226Foam::tmp<Foam::scalarField>
227Foam::functionObjects::regionSizeDistribution::divide
228(
229 const scalarField& num,
230 const scalarField& denom
231)
232{
233 auto tresult = tmp<scalarField>::New(num.size());
234 auto& result = tresult.ref();
235
236 forAll(denom, i)
237 {
238 if (denom[i] != 0)
239 {
240 result[i] = num[i]/denom[i];
241 }
242 else
243 {
244 result[i] = 0;
245 }
246 }
247 return tresult;
248}

251void Foam::functionObjects::regionSizeDistribution::writeGraphs
252(
253 const word& fieldName, // name of field
254 const scalarField& sortedField, // per region field data
255
256 const labelList& indices, // index of bin for each region
257 const scalarField& binCount, // per bin number of regions
258 const coordSet& coords // graph data for bins
259) const
260{
261 if (Pstream::master())
262 {
263 // Calculate per-bin average
264 scalarField binSum(nBins_, Zero);
265 forAll(sortedField, i)
266 {
267 binSum[indices[i]] += sortedField[i];
268 }
269
270 scalarField binAvg(divide(binSum, binCount));
271
272 // Per bin deviation
273 scalarField binSqrSum(nBins_, Zero);
274 forAll(sortedField, i)
275 {
276 binSqrSum[indices[i]] += Foam::sqr(sortedField[i]);
277 }
278 scalarField binDev
279 (
280 sqrt(divide(binSqrSum, binCount) - Foam::sqr(binAvg))
281 );
282
283
284 auto& writer = formatterPtr_();
285
286 word outputName;
287 if (writer.buffering())
288 {
289 outputName =
290 (
291 coords.name()
292 + coordSetWriter::suffix
293 (
294 wordList
295 ({
296 fieldName + "_sum",
297 fieldName + "_avg",
298 fieldName + "_dev"
299 })
300 )
301 );
302 }
303 else
304 {
305 outputName = coords.name();
306 }
307
308 writer.open
309 (
310 coords,
311 (baseTimeDir() / outputName)
312 );
313
314 Log << " Writing distribution of "
315 << fieldName << " to " << writer.path() << endl;
316
317 writer.write(fieldName + "_sum", binSum);
318 writer.write(fieldName + "_avg", binAvg);
319 writer.write(fieldName + "_dev", binDev);
320 writer.close(true);
321 }
322}

352

354 binCount, // per bin number of regions
355 coords // graph data for bins

430bool Foam::functionObjects::regionSizeDistribution::write()
431{
432 Log << type() << " " << name() << " write:" << nl;
433
434 tmp<volScalarField> talpha;
435 talpha.cref(obr_.cfindObject<volScalarField>(alphaName_));
436 if (talpha)
437 {
438 Log << " Looking up field " << alphaName_ << endl;
439 }
440 else
441 {
442 Info<< " Reading field " << alphaName_ << endl;
443 talpha.reset
444 (
445 new volScalarField
446 (
447 IOobject
448 (
449 alphaName_,
450 mesh_.time().timeName(),
451 mesh_,
452 IOobject::MUST_READ,
453 IOobject::NO_WRITE
454 ),
455 mesh_
456 )
457 );
458 }
459 const auto& alpha = talpha();
460
461 Log << " Volume of alpha = "
462 << fvc::domainIntegrate(alpha).value()
463 << endl;
464
465 const scalar meshVol = gSum(mesh_.V());
466 const scalar maxDropletVol = 1.0/6.0*mathematical::pi*pow3(maxDiam_);
467 const scalar delta = (maxDiam_-minDiam_)/nBins_;
468
469 Log << " Mesh volume = " << meshVol << nl
470 << " Maximum droplet diameter = " << maxDiam_ << nl
471 << " Maximum droplet volume = " << maxDropletVol
472 << endl;
473
474
475 // Determine blocked faces
476 boolList blockedFace(mesh_.nFaces(), false);
477 // label nBlocked = 0;
478
479 {
480 for (label facei = 0; facei < mesh_.nInternalFaces(); facei++)
481 {
482 scalar ownVal = alpha[mesh_.faceOwner()[facei]];
483 scalar neiVal = alpha[mesh_.faceNeighbour()[facei]];
484
485 if
486 (
487 (ownVal < threshold_ && neiVal > threshold_)
488 || (ownVal > threshold_ && neiVal < threshold_)
489 )
490 {
491 blockedFace[facei] = true;
492 // ++nBlocked;
493 }
494 }
495
496 // Block coupled faces
497 forAll(alpha.boundaryField(), patchi)
498 {
499 const fvPatchScalarField& fvp = alpha.boundaryField()[patchi];
500 if (fvp.coupled())
501 {
502 tmp<scalarField> townFld(fvp.patchInternalField());
503 tmp<scalarField> tnbrFld(fvp.patchNeighbourField());
504 const auto& ownFld = townFld();
505 const auto& nbrFld = tnbrFld();
506
507 label start = fvp.patch().patch().start();
508
509 forAll(ownFld, i)
510 {
511 scalar ownVal = ownFld[i];
512 scalar neiVal = nbrFld[i];
513
514 if
515 (
516 (ownVal < threshold_ && neiVal > threshold_)
517 || (ownVal > threshold_ && neiVal < threshold_)
518 )
519 {
520 blockedFace[start+i] = true;
521 // ++nBlocked;
522 }
523 }
524 }
525 }
526 }
527
528
529 regionSplit regions(mesh_, blockedFace);
530
531 Log << " Determined " << regions.nRegions()
532 << " disconnected regions" << endl;
533
534
535 if (debug)
536 {
537 volScalarField region
538 (
539 IOobject
540 (
541 "region",
542 mesh_.time().timeName(),
543 mesh_,
544 IOobject::NO_READ,
545 IOobject::NO_WRITE
546 ),
547 mesh_,
548 dimensionedScalar(dimless, Zero)
549 );
550 Info<< " Dumping region as volScalarField to " << region.name()
551 << endl;
552
553 forAll(regions, celli)
554 {
555 region[celli] = regions[celli];
556 }
557 region.correctBoundaryConditions();
558 region.write();
559 }
560
561
562 // Determine regions connected to supplied patches
563 Map<label> patchRegions(findPatchRegions(regions));
564
565
566 // Sum all regions
567 const scalarField alphaVol(alpha.primitiveField()*mesh_.V());
568 Map<scalar> allRegionVolume(regionSum(regions, mesh_.V()));
569 Map<scalar> allRegionAlphaVolume(regionSum(regions, alphaVol));
570 Map<label> allRegionNumCells
571 (
572 regionSum
573 (
574 regions,
575 labelField(mesh_.nCells(), 1.0)
576 )
577 );
578
579 if (debug)
580 {
581 Info<< " " << token::TAB << "Region"
582 << token::TAB << "Volume(mesh)"
583 << token::TAB << "Volume(" << alpha.name() << "):"
584 << token::TAB << "nCells"
585 << nl;
586 scalar meshSumVol = 0.0;
587 scalar alphaSumVol = 0.0;
588 label nCells = 0;
589
590 auto vIter = allRegionVolume.cbegin();
591 auto aIter = allRegionAlphaVolume.cbegin();
592 auto numIter = allRegionNumCells.cbegin();
593 for
594 (
595 ;
596 vIter.good() && aIter.good();
597 ++vIter, ++aIter, ++numIter
598 )
599 {
600 Info<< " " << token::TAB << vIter.key()
601 << token::TAB << vIter()
602 << token::TAB << aIter()
603 << token::TAB << numIter()
604 << nl;
605
606 meshSumVol += vIter();
607 alphaSumVol += aIter();
608 nCells += numIter();
609 }
610 Info<< " " << token::TAB << "Total:"
611 << token::TAB << meshSumVol
612 << token::TAB << alphaSumVol
613 << token::TAB << nCells
614 << endl;
615 }
616
617
618 if (log)
619 {
620 Info<< " Patch connected regions (liquid core):" << nl;
621 Info<< token::TAB << " Region"
622 << token::TAB << "Volume(mesh)"
623 << token::TAB << "Volume(" << alpha.name() << "):"
624 << nl;
625
626 forAllConstIters(patchRegions, iter)
627 {
628 const label regioni = iter.key();
629 Info<< " " << token::TAB << regioni
630 << token::TAB << allRegionVolume[regioni]
631 << token::TAB << allRegionAlphaVolume[regioni] << nl;
632
633 }
634 Info<< endl;
635 }
636
637 if (log)
638 {
639 Info<< " Background regions:" << nl;
640 Info<< " " << token::TAB << "Region"
641 << token::TAB << "Volume(mesh)"
642 << token::TAB << "Volume(" << alpha.name() << "):"
643 << nl;
644
645 auto vIter = allRegionVolume.cbegin();
646 auto aIter = allRegionAlphaVolume.cbegin();
647
648 for
649 (
650 ;
651 vIter.good() && aIter.good();
652 ++vIter, ++aIter
653 )
654 {
655 if
656 (
657 !patchRegions.found(vIter.key())
658 && vIter() >= maxDropletVol
659 )
660 {
661 Info<< " " << token::TAB << vIter.key()
662 << token::TAB << vIter()
663 << token::TAB << aIter() << nl;
664 }
665 }
666 Info<< endl;
667 }
668
669
670
671 // Split alpha field
672 // ~~~~~~~~~~~~~~~~~
673 // Split into
674 // - liquidCore : region connected to inlet patches
675 // - per region a volume : for all other regions
676 // - backgroundAlpha : remaining alpha
677 writeAlphaFields(regions, patchRegions, allRegionVolume, alpha);
678
679
680 // Extract droplet-only allRegionVolume, i.e. delete liquid core
681 // (patchRegions) and background regions from maps.
682 // Note that we have to use mesh volume (allRegionVolume) and not
683 // allRegionAlphaVolume since background might not have alpha in it.
684 // Deleting regions where the volume-alpha-weighted is lower than
685 // threshold
686 forAllIters(allRegionVolume, vIter)
687 {
688 const label regioni = vIter.key();
689 if
690 (
691 patchRegions.found(regioni)
692 || vIter() >= maxDropletVol
693 || (allRegionAlphaVolume[regioni]/vIter() < threshold_)
694 )
695 {
696 allRegionVolume.erase(vIter);
697 allRegionAlphaVolume.erase(regioni);
698 allRegionNumCells.erase(regioni);
699 }
700 }
701
702 if (allRegionVolume.size())
703 {
704 // Construct mids of bins for plotting
705 pointField xBin(nBins_, Zero);
706
707 {
708 scalar x = 0.5*delta;
709 for (point& p : xBin)
710 {
711 p.x() = x;
712 x += delta;
713 }
714 }
715
716 const coordSet coords("diameter", "x", xBin, mag(xBin));
717
718
719 // Get in region order the alpha*volume and diameter
720 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
721
722 const labelList sortedRegions = allRegionAlphaVolume.sortedToc();
723
724 scalarField sortedVols
725 (
726 extractData(sortedRegions, allRegionAlphaVolume)
727 );
728
729 vectorField centroids(sortedVols.size(), Zero);
730
731 // Check if downstream bins are calculated
732 if (isoPlanes_)
733 {
734 vectorField alphaDistance
735 (
736 (alpha.primitiveField()*mesh_.V())
737 *(mesh_.C().primitiveField() - origin_)()
738 );
739
740 Map<vector> allRegionAlphaDistance
741 (
742 regionSum
743 (
744 regions,
745 alphaDistance
746 )
747 );
748
749 // 2. centroid
750 vectorField sortedMoment
751 (
752 extractData(sortedRegions, allRegionAlphaDistance)
753 );
754
755 centroids = sortedMoment/sortedVols + origin_;
756
757 // Bin according to centroid
758 scalarField distToPlane((centroids - origin_) & direction_);
759
760 vectorField radialDistToOrigin
761 (
762 (centroids - origin_) - (distToPlane*direction_)
763 );
764
765 const scalar deltaX = maxDownstream_/nDownstreamBins_;
766 labelList downstreamIndices(distToPlane.size(), -1);
767 forAll(distToPlane, i)
768 {
769 if
770 (
771 (mag(radialDistToOrigin[i]) < maxDiameter_)
772 && (distToPlane[i] < maxDownstream_)
773 )
774 {
775 downstreamIndices[i] = distToPlane[i]/deltaX;
776 }
777 }
778
779 scalarField binDownCount(nDownstreamBins_, Zero);
780 forAll(distToPlane, i)
781 {
782 if (downstreamIndices[i] != -1)
783 {
784 binDownCount[downstreamIndices[i]] += 1.0;
785 }
786 }
787
788 // Write
789 if (Pstream::master())
790 {
791 // Construct mids of bins for plotting
792 pointField xBin(nDownstreamBins_, Zero);
793
794 {
795 scalar x = 0.5*deltaX;
796 for (point& p : xBin)
797 {
798 p.x() = x;
799 x += deltaX;
800 }
801 }
802
803 const coordSet coords("distance", "x", xBin, mag(xBin));
804
805 auto& writer = formatterPtr_();
806 writer.nFields(1);
807
808 writer.open
809 (
810 coords,
811 writeFile::baseTimeDir() / (coords.name() + "_isoPlanes")
812 );
813
814 writer.write("isoPlanes", binDownCount);
815 writer.close(true);
816 }
817
818 // Write to log
819 if (log)
820 {
821 Info<< " Iso-planes Bins:" << nl
822 << " " << token::TAB << "Bin"
823 << token::TAB << "Min distance"
824 << token::TAB << "Count:"
825 << nl;
826
827 scalar delta = 0.0;
828 forAll(binDownCount, bini)
829 {
830 Info<< " " << token::TAB << bini
831 << token::TAB << delta
832 << token::TAB << binDownCount[bini] << nl;
833 delta += deltaX;
834 }
835 Info<< endl;
836
837 }
838 }
839
840 // Calculate the diameters
841 scalarField sortedDiameters(sortedVols.size());
842 forAll(sortedDiameters, i)
843 {
844 sortedDiameters[i] = Foam::cbrt
845 (
846 sortedVols[i]
847 *6/constant::mathematical::pi
848 );
849 }
850
851 // Determine the bin index for all the diameters
852 labelList indices(sortedDiameters.size());
853 forAll(sortedDiameters, i)
854 {
855 indices[i] = (sortedDiameters[i]-minDiam_)/delta;
856 }
857
858 // Calculate the counts per diameter bin
859 scalarField binCount(nBins_, Zero);
860 forAll(sortedDiameters, i)
861 {
862 binCount[indices[i]] += 1.0;
863 }
864
865 // Write counts
866 if (Pstream::master())
867 {
868 auto& writer = formatterPtr_();
869 writer.nFields(1);
870
871 writer.open
872 (
873 coords,
874 writeFile::baseTimeDir() / (coords.name() + "_count")
875 );
876
877 writer.write("count", binCount);
878 writer.close(true);
879 }
880
881 // Write to log
882 if (log)
883 {
884 Info<< " Bins:" << nl
885 << " " << token::TAB << "Bin"
886 << token::TAB << "Min diameter"
887 << token::TAB << "Count:"
888 << nl;
889
890 scalar diam = 0.0;
891 forAll(binCount, bini)
892 {
893 Info<< " " << token::TAB << bini
894 << token::TAB << diam
895 << token::TAB << binCount[bini] << nl;
896
897 diam += delta;
898 }
899
900 Info<< endl;
901 }
902
903
904 // Write average and deviation of droplet volume.
905 writeGraphs
906 (
907 "volume", // name of field
908 sortedVols, // per region field data
909
910 indices, // per region the bin index
911 binCount, // per bin number of regions
912 coords // graph data for bins
913 );
914
915 // Collect some more fields
916 {
917 for
918 (
919 const word& fldName
920 : obr_.sortedNames<volScalarField>(fields_)
921 )
922 {
923 Log << " Scalar field " << fldName << endl;
