36void Foam::IOobjectList::checkObjectOrder
37(
38 const UPtrList<const IOobject>& objs,
39 bool syncPar
40)
41{
42 if (syncPar && Pstream::parRun())
43 {
44 wordList objectNames(objs.size());
45
46 auto iter = objectNames.begin();
47
48 for (const IOobject& io : objs)
49 {
50 *iter = io.name(); // nameOp<IOobject>()
51 ++iter;
52 }
53
54 checkNameOrder(objectNames, syncPar);
55 }
56}

59void Foam::IOobjectList::checkNameOrder
60(
61 const wordList& objectNames,
62 bool syncPar
63)
64{
65 if (syncPar && Pstream::parRun())
66 {
67 wordList masterNames;
68 if (Pstream::master())
69 {
70 masterNames = objectNames;
71 }
72 Pstream::broadcast(masterNames);
73
74 if (objectNames != masterNames)
75 {
76 FatalErrorInFunction
77 << "Objects not synchronised across processors." << nl
78 << "Master has " << flatOutput(masterNames) << nl
79 << "Processor " << Pstream::myProcNo()
80 << " has " << flatOutput(objectNames) << endl
81 << exit(FatalError);
82 }
83 }
84}

87void Foam::IOobjectList::syncNames(wordList& objNames)
88{
89 if (Pstream::parRun())
90 {
91 // Synchronize names
92 Pstream::combineReduce(objNames, ListOps::uniqueEqOp<word>());
93 }
94
95 Foam::sort(objNames); // Consistent order
96}

101Foam::IOobjectList::IOobjectList
102(
103 const objectRegistry& db,
104 const fileName& instance,
105 const fileName& local,
106 IOobjectOption ioOpt
107)
108:
109 HashPtrTable<IOobject>()
110{
111 word newInstance;
112 fileNameList objNames = fileHandler().readObjects
113 (
114 db,
115 instance,
116 local,
117 newInstance
118 );
119
120 for (const auto& objName : objNames)
121 {
122 auto objectPtr = autoPtr<IOobject>::New
123 (
124 objName,
125 newInstance,
126 local,
127 db,
128 ioOpt
129 );
130
131 bool ok = false;
132 const bool oldThrowingIOerr = FatalIOError.throwing(true);
133
134 try
135 {
136 // Use object with local scope and current instance (no searching)
137 ok = objectPtr->typeHeaderOk<regIOobject>(false, false);
138 }
139 catch (const Foam::IOerror& err)
140 {
141 Warning << err << nl << endl;
142 }
143
144 FatalIOError.throwing(oldThrowingIOerr);
145
146 if (ok)
147 {
148 insert(objectPtr->name(), objectPtr);
149 }
150 }
151}

156const Foam::IOobject* Foam::IOobjectList::cfindObject
157(
158 const word& objName
159) const
160{
161 // Like HashPtrTable::get(), or lookup() with a nullptr
162 const IOobject* io = nullptr;
163
164 const const_iterator iter(cfind(objName));
165 if (iter.good())
166 {
167 io = iter.val();
168 }
169
170 if (IOobject::debug)
171 {
172 if (io)
173 {
174 InfoInFunction << "Found " << objName << endl;
175 }
176 else
177 {
178 InfoInFunction << "Could not find " << objName << endl;
179 }
180 }
181
182 return io;
183}

346{
347 os << nl << list.size() << nl << token::BEGIN_LIST << nl;
348
349 forAllConstIters(list, iter)
350 {
351 os << iter.key() << token::SPACE
352 << iter.val()->headerClassName() << nl;
353 }
