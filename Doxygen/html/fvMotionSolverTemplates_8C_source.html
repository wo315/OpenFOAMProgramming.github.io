34template<class Type>
35Foam::wordList Foam::fvMotionSolver::cellMotionBoundaryTypes
36(
37 const typename GeometricField<Type, pointPatchField, pointMesh>::
38 Boundary& pmUbf
39) const
40{
41 wordList cmUbf = pmUbf.types();
42
43 // Remove global patches from the end of the list
44 cmUbf.setSize(fvMesh_.boundary().size());
45
46 forAll(cmUbf, patchi)
47 {
48 if (isA<fixedValuePointPatchField<Type>>(pmUbf[patchi]))
49 {
50 cmUbf[patchi] = cellMotionFvPatchField<Type>::typeName;
51 }
52
53 if (debug)
54 {
55 Pout<< "Patch:" << fvMesh_.boundary()[patchi].patch().name()
56 << " pointType:" << pmUbf.types()[patchi]
57 << " cellType:" << cmUbf[patchi] << endl;
58 }
59 }
60
61 return cmUbf;
62}
