102template<class... Args>
103inline Foam::vtk::formatter& Foam::vtk::formatter::xmlComment
104(
105 const std::string& text,
106 Args&&... args
107)
108{
109 if (canWriteToplevel("xml comment"))
110 {
111 indent();
112 os_ << "<!--";
113
114 if (sizeof...(Args))
115 {
116 os_ << nl;
117
118 xmlCommentLoop(text, std::forward<Args>(args)...);
119
120 indent(); indent(2);
121 }
122 else
123 {
124 os_ << ' ' << text << ' ';
125 }
126
127 os_ << "-->" << nl;
128 }
129
130 return *this;
131}
