43{
44 HashTable<const filmModelType*> models
45 = db().time().lookupClass<filmModelType>();
46
47 forAllConstIters(models, iter)

49 if (iter()->regionMesh().name() == filmRegionName_)

60

246void filmPyrolysisRadiativeCoupledMixedFvPatchScalarField::updateCoeffs()
247{
248 if (updated())
249 {
250 return;
251 }
252
253 // Get the coupling information from the mappedPatchBase
254 const mappedPatchBase& mpp =
255 refCast<const mappedPatchBase>(patch().patch());
256
257 const label patchi = patch().index();
258 const label nbrPatchi = mpp.samplePolyPatch().index();
259 const polyMesh& mesh = patch().boundaryMesh().mesh();
260 const polyMesh& nbrMesh = mpp.sampleMesh();
261 const fvPatch& nbrPatch =
262 refCast<const fvMesh>(nbrMesh).boundary()[nbrPatchi];
263
264 scalarField intFld(patchInternalField());
265
266 const auto& nbrField =
267 refCast
268 <
269 const filmPyrolysisRadiativeCoupledMixedFvPatchScalarField
270 >
271 (
272 nbrPatch.lookupPatchField<volScalarField>(TnbrName_)
273 );
274
275 // Swap to obtain full local values of neighbour internal field
276 scalarField nbrIntFld(nbrField.patchInternalField());
277 mpp.distribute(nbrIntFld);
278
279 scalarField& Tp = *this;
280
281 const scalarField K(this->kappa(*this));
282 const scalarField nbrK(nbrField.kappa(*this));
283
284 // Swap to obtain full local values of neighbour K*delta
285 scalarField KDeltaNbr(nbrK*nbrPatch.deltaCoeffs());
286 mpp.distribute(KDeltaNbr);
287
288 scalarField myKDelta(K*patch().deltaCoeffs());
289
290 scalarList Tfilm(patch().size(), Zero);
291 scalarList htcwfilm(patch().size(), Zero);
292 scalarList filmDelta(patch().size(), Zero);
293
294 const pyrolysisModelType& pyrolysis = pyrModel();
295 const filmModelType& film = filmModel();
296
297 // Obtain Rad heat (qr)
298 scalarField qr(patch().size(), Zero);
299
300 label coupledPatchi = -1;
301 if (pyrolysisRegionName_ == mesh.name())
302 {
303 coupledPatchi = patchi;
304 if (qrName_ != "none")
305 {
306 qr = nbrPatch.lookupPatchField<volScalarField>(qrName_);
307 mpp.distribute(qr);
308 }
309 }
310 else if (pyrolysis.primaryMesh().name() == mesh.name())
311 {
312 coupledPatchi = nbrPatch.index();
313 if (qrName_ != "none")
314 {
315 qr = patch().lookupPatchField<volScalarField>(qrName_);
316 }
317 }
318 else
319 {
320 FatalErrorInFunction
321 << type() << " condition is intended to be applied to either the "
322 << "primary or pyrolysis regions only"
323 << exit(FatalError);
324 }
325
326 const label filmPatchi = pyrolysis.nbrCoupledPatchID(film, coupledPatchi);
327
328 const scalarField htcw(film.htcw().h()().boundaryField()[filmPatchi]);
329
330 // Obtain htcw
331 htcwfilm =
332 pyrolysis.mapRegionPatchField
333 (
334 film,
335 coupledPatchi,
336 filmPatchi,
337 htcw,
338 true
339 );
340
341
342 // Obtain Tfilm at the boundary through Ts.
343 // NOTE: Tf is not good as at the boundary it will retrieve Tp
344 Tfilm = film.Ts().boundaryField()[filmPatchi];
345 film.toPrimary(filmPatchi, Tfilm);
346
347 // Obtain delta
348 filmDelta =
349 pyrolysis.mapRegionPatchField<scalar>
350 (
351 film,
352 "deltaf",
353 coupledPatchi,
354 true
355 );
356
357 // Estimate wetness of the film (1: wet , 0: dry)
358 scalarField ratio
359 (
360 min
361 (
362 max
363 (
364 (filmDelta - filmDeltaDry_)/(filmDeltaWet_ - filmDeltaDry_),
365 scalar(0)
366 ),
367 scalar(1)
368 )
369 );
370
371 scalarField qConv(ratio*htcwfilm*(Tfilm - Tp)*convectiveScaling_);
372
373 scalarField qRad((1.0 - ratio)*qr);
374
375 scalarField alpha(KDeltaNbr - (qRad + qConv)/Tp);
376
377 valueFraction() = alpha/(alpha + (1.0 - ratio)*myKDelta);
378
379 refValue() = ratio*Tfilm + (1.0 - ratio)*(KDeltaNbr*nbrIntFld)/alpha;
380
381 mixedFvPatchScalarField::updateCoeffs();
382
383 if (debug)
384 {
385 scalar Qc = gSum(qConv*patch().magSf());
386 scalar qr = gSum(qRad*patch().magSf());
387 scalar Qt = gSum((qConv + qRad)*patch().magSf());
388
389 Info<< mesh.name() << ':'
390 << patch().name() << ':'
391 << this->internalField().name() << " <- "
392 << nbrMesh.name() << ':'
393 << nbrPatch.name() << ':'
394 << this->internalField().name() << " :" << nl
395 << " convective heat[W] : " << Qc << nl
396 << " radiative heat [W] : " << qr << nl
397 << " total heat [W] : " << Qt << nl
398 << " wall temperature "
399 << " min:" << gMin(*this)
400 << " max:" << gMax(*this)
401 << " avg:" << gAverage(*this)
402 << endl;
403 }
404}
