113(
114 const exponentialSolidTransport<Thermo>& ct
115)
116{
117 scalar Y1 = this->Y();
118
119 Y1 /= this->Y();
120 scalar Y2 = ct.Y()/this->Y();
121
122 kappa0_ = Y1*kappa0_ + Y2*ct.kappa0_;
123 n0_ = (Y1*n0_ + Y2*ct.n0_);
124 Tref_ = (Y1*Tref_ + Y2*ct.Tref_);
