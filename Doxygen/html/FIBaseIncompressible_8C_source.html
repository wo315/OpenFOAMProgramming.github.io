127void FIBase::accumulateIntegrand(const scalar dt)
128{
129 // Accumulate multiplier of grad(dxdb)
130 gradDxDbMult_ += computeGradDxDbMultiplier()().T()*dt;
131
132 // Accumulate multiplier of div(dxdb)
133 PtrList<objective>& functions(objectiveManager_.getObjectiveFunctions());
134 for (objective& func : functions)
135 {
136 if (func.hasDivDxDbMult())
137 {
138 divDxDbMult_ +=
139 func.weight()*func.divDxDbMultiplier().primitiveField()*dt;
140 }
141 }
142
143 // Terms from fvOptions
144 fv::options::New(this->mesh_).postProcessSens
145 (
146 optionsDxDbMult_, adjointVars_.solverName()
147 );
148
149 // Accumulate source for the adjoint to the eikonal equation
150 if (includeDistance_)
151 {
152 eikonalSolver_->accumulateIntegrand(dt);
153 }
154
155 // Accumulate direct sensitivities
156 accumulateDirectSensitivityIntegrand(dt);
157
158 // Accumulate sensitivities due to boundary conditions
159 accumulateBCSensitivityIntegrand(dt);
160}

163void FIBase::clearSensitivities()
164{
165 gradDxDbMult_ = dimensionedTensor(gradDxDbMult_.dimensions(), Zero);
166 divDxDbMult_ = Zero;
167 optionsDxDbMult_ = vector::zero;
168
169 if (includeDistance_)
170 {
171 eikonalSolver_->reset();
172 }
173
174 shapeSensitivities::clearSensitivities();
175}
