36template<class T>
37T Foam::ReadHex(ISstream& is)
38{
39 // The char '0' == 0
40 static constexpr int offsetZero = int('0');
41
42 // The char 'A' (or 'a') == 10
43 static constexpr int offsetAlpha = int('A') - 10;
44
45 char c = 0;
46
47 // Get next non-whitespace character
48 while (is.get(c) && isspace(c))
49 {}
50
51 T result = 0;
52 do
53 {
54 if (isspace(c) || c == 0) break;
55
56 if (!isxdigit(c))
57 {
58 FatalIOErrorInFunction(is)
59 << "Illegal hex digit: '" << c << "'"
60 << exit(FatalIOError);
61 }
62
63 result <<= 4;
64
65 if (isdigit(c))
66 {
67 result += int(c) - offsetZero;
68 }
69 else
70 {
71 result += toupper(c) - offsetAlpha;
72 }
73 } while (is.get(c));
74
75 return result;
76}
