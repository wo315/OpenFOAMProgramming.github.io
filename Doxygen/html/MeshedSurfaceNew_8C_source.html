35template<class Face>
36Foam::autoPtr<Foam::MeshedSurface<Face>>
37Foam::MeshedSurface<Face>::New
38(
39 const fileName& name,
40 const word& fileType,
41 bool mandatory
42)
43{
44 const word ext(name.ext());
45
46 if (fileType.empty())
47 {
48 // Handle empty/missing type
49
50 if (ext.empty())
51 {
52 FatalErrorInFunction
53 << "Cannot determine format from filename" << nl
54 << " " << name << nl
55 << exit(FatalError);
56 }
57
58 return New(name, ext, mandatory);
59 }
60 else if (fileType == "gz")
61 {
62 // Degenerate call
63 return New(name.lessExt(), name.stem().ext(), mandatory);
64 }
65 else if (ext == "gz")
66 {
67 // Handle trailing "gz" on file name
68 return New(name.lessExt(), fileType, mandatory);
69 }
70
71 // if (check && !exists(name))
72 // {
73 // FatalErrorInFunction
74 // << "No such file " << name << nl
75 // << exit(FatalError);
76 // }
77
78
79 DebugInFunction
80 << "Construct MeshedSurface (" << fileType << ")\n";
81
82 auto* ctorPtr = fileExtensionConstructorTable(fileType);
83
84 if (ctorPtr)
85 {
86 return autoPtr<MeshedSurface<Face>>(ctorPtr(name));
87 }
88
89
90 // Delegate to friend if possible
91 const wordHashSet delegate(FriendType::readTypes());
92
93 if (delegate.found(fileType))
94 {
95 // OK, can create indirectly
96 auto surf = autoPtr<MeshedSurface<Face>>::New();
97 surf->transfer(*FriendType::New(name, fileType));
98
99 return surf;
100 }
101 else if (mandatory)
102 {
103 FatalErrorInFunction
104 << "Unknown surface format " << fileType << nl << nl
105 << "Valid types:" << nl
106 << flatOutput((delegate | readTypes()).sortedToc()) << nl
107 << exit(FatalError);
108 }
109
110 // Failed, but was optional
111 return nullptr;
112}

115template<class Face>
116Foam::autoPtr<Foam::MeshedSurface<Face>>
117Foam::MeshedSurface<Face>::New(const fileName& name)
118{
119 if (name.has_ext("gz"))
120 {
121 // Handle trailing "gz" on file name
122 return New(name.lessExt(), name.stem().ext());
123 }
124
125 return New(name, name.ext());
126}
