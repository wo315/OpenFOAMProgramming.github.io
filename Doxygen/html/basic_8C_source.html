68template<class Type>
69void Foam::AveragingMethods::Basic<Type>::updateGrad()
70{
71 GeometricField<Type, fvPatchField, volMesh> tempData
72 (
73 IOobject
74 (
75 "BasicAverage::Data",
76 this->mesh_,
77 IOobject::NO_READ,
78 IOobject::NO_WRITE,
79 IOobject::NO_REGISTER
80 ),
81 this->mesh_,
82 dimensioned<Type>(dimless, Zero),
83 fvPatchFieldBase::zeroGradientType()
84 );
85 tempData.primitiveFieldRef() = data_;
86 tempData.correctBoundaryConditions();
87 dataGrad_ = fvc::grad(tempData)->primitiveField();
88}
