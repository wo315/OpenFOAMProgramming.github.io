49tmp<volScalarField> qZeta::fMu() const
50{
51 const volScalarField Rt(q_*k_/(2.0*nu()*zeta_));
52
53 if (anisotropic_)
54 {
55 return exp((-scalar(2.5) + Rt/20.0)/pow3(scalar(1) + Rt/130.0));
56 }
57 else
58 {
59 return
60 exp(-6.0/sqr(scalar(1) + Rt/50.0))
61 *(scalar(1) + 3.0*exp(-Rt/10.0));
62 }
63}

65
66tmp<volScalarField> qZeta::f2() const
67{
68 tmp<volScalarField> Rt = q_*k_/(2.0*nu()*zeta_);
69 return scalar(1) - 0.3*exp(-sqr(Rt));
70}

73void qZeta::correctNut()
74{
75 nut_ = Cmu_*fMu()*sqr(k_)/epsilon_;
76 nut_.correctBoundaryConditions();

240void qZeta::correct()
241{
242 if (!turbulence_)
243 {
244 return;
245 }
246
247 eddyViscosity<incompressible::RASModel>::correct();
248
249 volScalarField G(GName(), nut_/(2.0*q_)*2*magSqr(symm(fvc::grad(U_))));
250 const volScalarField E(nu()*nut_/q_*fvc::magSqrGradGrad(U_));
251
252 // Zeta equation
253 tmp<fvScalarMatrix> zetaEqn
254 (
255 fvm::ddt(zeta_)
256 + fvm::div(phi_, zeta_)
257 - fvm::laplacian(DzetaEff(), zeta_)
258 ==
259 (2.0*C1_ - 1)*G*zeta_/q_
260 - fvm::SuSp((2.0*C2_*f2() - dimensionedScalar(1.0))*zeta_/q_, zeta_)
261 + E
262 );
263
264 zetaEqn.ref().relax();
265 solve(zetaEqn);
266 bound(zeta_, zetaMin_);
267
268
269 // q equation
270 tmp<fvScalarMatrix> qEqn
271 (
272 fvm::ddt(q_)
273 + fvm::div(phi_, q_)
274 - fvm::laplacian(DqEff(), q_)
275 ==
276 G - fvm::Sp(zeta_/q_, q_)
277 );
278
279 qEqn.ref().relax();
280 solve(qEqn);
281 bound(q_, qMin_);
282
283
284 // Re-calculate k and epsilon
285 k_ = sqr(q_);
286 k_.correctBoundaryConditions();
287
288 epsilon_ = 2*q_*zeta_;
289 epsilon_.correctBoundaryConditions();
290
291 correctNut();
292}
