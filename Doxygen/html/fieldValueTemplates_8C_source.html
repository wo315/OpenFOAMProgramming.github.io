35template<class Type>
36void Foam::functionObjects::fieldValue::combineFields(Field<Type>& field)
37{
38 if (Pstream::parRun())
39 {
40 List<Field<Type>> allValues(Pstream::nProcs());
41 allValues[Pstream::myProcNo()] = field;
42
43 Pstream::allGatherList(allValues);
44
45 field =
46 ListListOps::combine<Field<Type>>
47 (
48 allValues,
49 accessOp<Field<Type>>()
50 );
51 }
52}

58 combineFields(field.ref());
