40 auto& fld = tfld.ref();

42 forAll(pointToVerts_, i)
43 {
44 const edge& verts = pointToVerts_[i];
45 Type& val = fld[i];
46
47 scalar s0;
48 Type v0;
49 {
50 label idx = verts.first();
51 if (idx < mesh_.nPoints())
52 {
53 // Point index
54 s0 = pVals_[idx];
55 v0 = pointData[idx];
56 }
57 else
58 {
59 // Cell index
60 idx -= mesh_.nPoints();
61 s0 = cVals_[idx];

63 }
64 }
65
66 scalar s1;
67 Type v1;
68 {
69 label idx = verts.second();
70 if (idx == verts.first())
71 {
72 // Duplicate index (ie, snapped)
73 val = v0;
74 continue;
75 }
76 else if (idx < mesh_.nPoints())
77 {
78 // Point index
79 s1 = pVals_[idx];
80 v1 = pointData[idx];
81 }
82 else
83 {
84 // Cell index
85 idx -= mesh_.nPoints();
86 s1 = cVals_[idx];
87 v1 = cellData[idx];
88 }
89 }
90
91 const scalar d = s1-s0;
92 if (mag(d) > VSMALL)
93 {
94 const scalar s = (iso_-s0)/d;
95 val = s*v1+(1.0-s)*v0;
96 }
97 else
98 {
99 val = 0.5*(v0+v1);
100 }
101 }
102
103 return tfld;
104}
