40)
41{

44 const word methodName(reductionDict.get<word>("method"));

46 Info<< "Selecting chemistry reduction method " << methodName << endl;

48 const word methodTypeName

50 methodName

52 );
53
54 const auto& cnstrTable = *(dictionaryConstructorTablePtr_);

56 auto* ctorPtr = cnstrTable.lookup(methodTypeName, nullptr);

62 constexpr const int nCmpt = 7;

73 DynamicList<wordList> validCmpts;

80 "reactionThermo",
81 "transport",
82 "thermo",
83 "equationOfState",
84 "specie",
85 "energy"

99 validCmpts.append(std::move(cmpts));
100 }

120 FatalError
