85void Foam::adjointOutletPressureFvPatchScalarField::updateCoeffs()
86{
87 if (updated())
88 {
89 return;
90 }
91
92 const auto& phip = patch().lookupPatchField<surfaceScalarField>("phi");
93 const auto& phiap = patch().lookupPatchField<surfaceScalarField>("phia");
94 const auto& Up = patch().lookupPatchField<volVectorField>("U");
95 const auto& Uap = patch().lookupPatchField<volVectorField>("Ua");
96
97 operator==((phiap/patch().magSf() - 1.0)*phip/patch().magSf() + (Up & Uap));
98
99 fixedValueFvPatchScalarField::updateCoeffs();
100}
