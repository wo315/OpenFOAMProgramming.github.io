95void Foam::energyJumpAMIFvPatchScalarField::updateCoeffs()
96{
97 if (this->updated())
98 {
99 return;
100 }
101
102 if (this->cyclicAMIPatch().owner())
103 {
104 const basicThermo& thermo = basicThermo::lookupThermo(*this);
105 label patchID = patch().index();
106
107 const scalarField& pp = thermo.p().boundaryField()[patchID];
108 const fixedJumpAMIFvPatchScalarField& TbPatch =
109 refCast<const fixedJumpAMIFvPatchScalarField>
110 (
111 thermo.T().boundaryField()[patchID]
112 );
113
114 fixedJumpAMIFvPatchScalarField& Tbp =
115 const_cast<fixedJumpAMIFvPatchScalarField&>(TbPatch);
116
117 // force update of jump
118 Tbp.evaluate(Pstream::commsTypes::blocking);
119
120 const labelUList& faceCells = this->patch().faceCells();
121
122 jump_ =
123 thermo.he(pp, Tbp+Tbp.jump(), faceCells)
124 - thermo.he(pp, Tbp, faceCells);
125 }
126
127 fixedJumpAMIFvPatchField<scalar>::updateCoeffs();
128}
