450template<class IntType>
451inline bool Foam::IntRange<IntType>::contains(const IntType value)
452const noexcept
453{
454 // Avoid overflow risk:
455 // (value < (start_ + size_)) --> ((value - start_) < size_)
456
457 return (size_ && start_ <= value && (value - start_) < size_);
458}
