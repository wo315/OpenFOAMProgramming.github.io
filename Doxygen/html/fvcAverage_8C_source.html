45template<class Type>
46tmp<GeometricField<Type, fvPatchField, volMesh>>
47average
48(
49 const GeometricField<Type, fvsPatchField, surfaceMesh>& ssf
50)
51{
52 const fvMesh& mesh = ssf.mesh();
53
54 tmp<GeometricField<Type, fvPatchField, volMesh>> taverage
55 (
56 new GeometricField<Type, fvPatchField, volMesh>
57 (
58 IOobject
59 (
60 "average("+ssf.name()+')',
61 ssf.instance(),
62 mesh,
63 IOobject::NO_READ,
64 IOobject::NO_WRITE
65 ),
66 mesh,
67 ssf.dimensions()
68 )
69 );
70 GeometricField<Type, fvPatchField, volMesh>& av = taverage.ref();
71
72 av.primitiveFieldRef() =
73 (
74 surfaceSum(mesh.magSf()*ssf)().primitiveField()
75 /surfaceSum(mesh.magSf())().primitiveField()
76 );
77
78 typename GeometricField<Type, fvPatchField, volMesh>::
79 Boundary& bav = av.boundaryFieldRef();
80
81 forAll(bav, patchi)
82 {
83 bav[patchi] = ssf.boundaryField()[patchi];
84 }
85
86 av.correctBoundaryConditions();
87
88 return taverage;
