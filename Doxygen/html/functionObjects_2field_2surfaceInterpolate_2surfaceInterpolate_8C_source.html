92bool Foam::functionObjects::surfaceInterpolate::write()
93{
94 Log << " functionObjects::" << type() << " " << name()
95 << " writing interpolated surface fields:" << nl;
96
97 forAll(fieldSet_, i)
98 {
99 const word& fieldName = fieldSet_[i].second();
100
101 const regIOobject* ioptr = obr_.findObject<regIOobject>(fieldName);
102
103 if (ioptr)
104 {
105 Log << " " << fieldName << nl;
106 ioptr->write();
107 }
108 else
109 {
110 WarningInFunction
111 << "Unable to find field " << fieldName
112 << " in the mesh database" << endl;
113 }
114 }
115
116 Log << endl;
117
118 return true;
119}
