46void pointNoise::filterTimeData
47(
48 const scalarField& t0,
49 const scalarField& p0,
50 scalarField& t,
51 scalarField& p
52) const
53{
54 DynamicList<scalar> tf(t0.size());
55 DynamicList<scalar> pf(t0.size());
56
57 forAll(t0, timeI)
58 {
59 if (t0[timeI] >= startTime_)
60 {
61 tf.append(t0[timeI]);
62 pf.append(p0[timeI]);
63 }
64 }
65
66 t.transfer(tf);
67 p.transfer(pf);
68}

71void pointNoise::processData
72(
73 const label dataseti,
74 const Function1Types::CSV<scalar>& data
75)
76{
77 if (!Pstream::master())
78 {
79 // Only ever called on master, report if we have odd logic...
80 WarningInFunction
81 << "Currently only to be called from master process..." << endl;
82 return;
83 }
84
85 Info<< "Reading data file: "
86 << fileObr_.time().relativePath(data.fName()) << endl;
87
88 const word fNameBase(data.fName().stem());
89
90 // Time and pressure history data
91 scalarField t, p;
92 filterTimeData(data.x(), data.y(), t, p);
93
94 Info<< " read " << t.size() << " values" << nl << endl;
95
96 if (!validateBounds(p))
97 {
98 Info<< "No noise data generated" << endl;
99 return;
100 }
101
102 Info<< "Creating noise FFT" << endl;
103
104 const scalar deltaT = checkUniformTimeStep(t);
105
106 // Apply conversions
107 p *= rhoRef_;
108 p -= average(p);
109
110 // Determine the windowing
111 windowModelPtr_->validate(t.size());
112 const windowModel& win = windowModelPtr_();
113 const scalar deltaf = 1.0/(deltaT*win.nSamples());
114 const fileName outDir(baseFileDir(dataseti)/fNameBase);
115
116
117 // Narrow band data
118 // ----------------
119
120 scalarField f(uniformFrequencies(deltaT, true));
121
122 // RMS pressure [Pa]
123 if (writePrmsf_)
124 {
125 auto filePtr = newFile(outDir/"Prms_f");
126 auto& os = filePtr();
127
128 Info<< " Writing " << os.name() << endl;
129
130 writeFileHeader(os, "f [Hz]", "Prms(f) [Pa]");
131 writeFreqDataToFile(os, f, RMSmeanPf(p));
132 }
133
134 // PSD [Pa^2/Hz]
135 const scalarField PSDf(this->PSDf(p, deltaT));
136
137 if (writePSDf_)
138 {
139 auto filePtr = newFile(outDir/"PSD_f");
140 auto& os = filePtr();
141
142 Info<< " Writing " << os.relativeName() << endl;
143
144 writeFileHeader(os, "f [Hz]", "PSD(f) [PaPa_Hz]");
145 writeFreqDataToFile(os, f, PSDf);
146 }
147
148 // PSD [dB/Hz]
149 if (writePSD_)
150 {
151 auto filePtr = newFile(outDir/"PSD_dB_Hz_f");
152 auto& os = filePtr();
153
154 Info<< " Writing " << os.relativeName() << endl;
155
156 writeFileHeader(os, "f [Hz]", "PSD(f) [dB_Hz]");
157 writeFreqDataToFile(os, f, PSD(PSDf));
158 }
159
160 // SPL [dB]
161 if (writeSPL_)
162 {
163 auto filePtr = newFile(outDir/"SPL_dB_f");
164 auto& os = filePtr();
165
166 Info<< " Writing " << os.relativeName() << endl;
167
168 writeFileHeader
169 (
170 os,
171 "f [Hz]",
172 "SPL(f) [" + weightingTypeNames_[SPLweighting_] + "]"
173 );
174 writeFreqDataToFile(os, f, SPL(PSDf*deltaf, f));
175 }
176
177 if (writeOctaves_)
178 {
179 labelList octave13BandIDs;
180 scalarField octave13FreqCentre;
181 setOctaveBands
182 (
183 f,
184 fLower_,
185 fUpper_,
186 3,
187 octave13BandIDs,
188 octave13FreqCentre
189 );
190
191
192 // 1/3 octave data
193 // ---------------
194
195 // Integrated PSD = P(rms)^2 [Pa^2]
196 scalarField Prms13f(octaves(PSDf, f, octave13BandIDs));
197
198 auto filePtr = newFile(outDir/"SPL13_dB_fm");
199 auto& os = filePtr();
200
201 Info<< " Writing " << os.relativeName() << endl;
202
203 writeFileHeader
204 (
205 os,
206 "fm [Hz]",
207 "SPL(fm) [" + weightingTypeNames_[SPLweighting_] + "]"
208 );
209 writeFreqDataToFile
210 (
211 os,
212 octave13FreqCentre,
213 SPL(Prms13f, octave13FreqCentre)
214 );
215 }
216}

240void pointNoise::calculate()
241{
242 // Point data only handled by master
243 if (!Pstream::master())
244 {
245 return;
246 }
247
248 forAll(inputFileNames_, filei)
249 {
250 fileName fName = inputFileNames_[filei];
251 fName.expand();
252
253 if (!fName.isAbsolute())
254 {
255 fName = argList::envGlobalPath()/fName;
256 }
257
258 Function1Types::CSV<scalar> data("pressure", dict_, nullptr, fName);
259 processData(filei, data);
260 }
261}

264bool pointNoise::read(const dictionary& dict)
265{
266 if (noiseModel::read(dict))
267 {
268 if (!dict.readIfPresent("files", inputFileNames_))
269 {
270 inputFileNames_.resize(1);
271
272 // Note: lookup uses same keyword as used by the CSV constructor
273 dict.readEntry("file", inputFileNames_.first());
274 }
275
276 return true;
277 }
278
279 return false;
280}
