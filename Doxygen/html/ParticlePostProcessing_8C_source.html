113template<class CloudType>
114void Foam::ParticlePostProcessing<CloudType>::postPatch
115(
116 const parcelType& p,
117 const polyPatch& pp,
118 bool&
119)
120{
121 if (!collector_.isPatch())
122 {
123 return;
124 }
125
126 const label patchi = pp.index();
127 const label localPatchi = collector_.IDs().find(patchi);
128
129 if (header_.empty())
130 {
131 OStringStream data;
132 p.writeProperties(data, fields_, " ", true);
133 header_ = data.str();
134 }
135
136 if (localPatchi != -1 && data_[localPatchi].size() < maxStoredParcels_)
137 {
138 times_[localPatchi].append(this->owner().time().value());
139
140 OStringStream data;
141 data<< Pstream::myProcNo();
142 p.writeProperties(data, fields_, " ", false);
143
144 data_[localPatchi].append(data.str());
145 }
146}

149template<class CloudType>
150void Foam::ParticlePostProcessing<CloudType>::postFace
151(
152 const parcelType& p,
153 bool&
154)
155{
156 if (collector_.isPatch())
157 {
158 return;
159 }
160
161 const labelList& IDs = collector_.IDs();
162 const List<boundBox>& BBs = collector_.BBs();
163 const faceZoneMesh& fzm = this->owner().mesh().faceZones();
164
165 if (header_.empty())
166 {
167 OStringStream data;
168 p.writeProperties(data, fields_, " ", true);
169 header_ = data.str();
170 }
171
172 forAll(IDs, i)
173 {
174 if (!BBs[i].contains(p.position()))
175 {
176 // Quick reject if the particle is not in the face zone bound box
177 continue;
178 }
179
180 const label zonei = IDs[i];
181 const label localFacei = fzm[zonei].find(p.face());
182
183 if (localFacei != -1 && data_[localFacei].size() < maxStoredParcels_)
184 {
185 times_[i].append(this->owner().time().value());
186
187 OStringStream data;
188 data<< Pstream::myProcNo();
189 p.writeProperties(data, fields_, " ", false);
190
191 data_[i].append(data.str());
192 }
193 }
194}

197template<class CloudType>
198void Foam::ParticlePostProcessing<CloudType>::write()
199{
200 const wordList& names = collector_.names();
201
202 forAll(names, i)
203 {
204 List<scalarList> procTimes(Pstream::nProcs());
205 procTimes[Pstream::myProcNo()] = times_[i];
206 Pstream::gatherList(procTimes);
207
208 List<List<string>> procData(Pstream::nProcs());
209 procData[Pstream::myProcNo()] = data_[i];
210 Pstream::gatherList(procData);
211
212 Pstream::combineReduce
213 (
214 header_,
215 [](string& x, const string& y)
216 {
217 if (y.size() > x.size())
218 {
219 x = y;
220 }
221 }
222 );
223
224 if (Pstream::master())
225 {
226 List<string> globalData;
227 globalData = ListListOps::combine<List<string>>
228 (
229 procData,
230 accessOp<List<string>>()
231 );
232
233 scalarList globalTimes;
234 globalTimes = ListListOps::combine<scalarList>
235 (
236 procTimes,
237 accessOp<scalarList>()
238 );
239
240 if (this->writeToFile())
241 {
242 autoPtr<OFstream> osPtr = this->newFileAtTime
243 (
244 names[i],
245 this->owner().time().value()
246 );
247 OFstream& os = osPtr.ref();
248
249 writeFileHeader(os);
250
251 const labelList indices(sortedOrder(globalTimes));
252 forAll(globalTimes, j)
253 {
254 const label datai = indices[j];
255
256 os << globalTimes[datai] << tab
257 << globalData[datai].c_str()
258 << nl;
259 }
260 }
261 }
262
263 times_[i].clearStorage();
264 data_[i].clearStorage();
265 }
266}
