105template<class Type>
106Foam::tmp<Foam::Field<Type>> Foam::wedgeFaPatchField<Type>::snGrad() const
107{
108 const Field<Type> pif (this->patchInternalField());
109
110 return
111 (
112 transform(refCast<const wedgeFaPatch>(this->patch()).faceT(), pif)
113 - pif
114 )*(0.5*this->patch().deltaCoeffs());
115}

118template<class Type>
119void Foam::wedgeFaPatchField<Type>::evaluate(const Pstream::commsTypes)
120{
121 if (!this->updated())
122 {
123 this->updateCoeffs();
124 }
125
126 faPatchField<Type>::operator==
127 (
128 transform
129 (
130 refCast<const wedgeFaPatch>(this->patch()).edgeT(),
131 this->patchInternalField()
132 )
133 );
134}

137template<class Type>
138Foam::tmp<Foam::Field<Type>>
139Foam::wedgeFaPatchField<Type>::snGradTransformDiag() const
140{
141 const diagTensor diagT =
142 0.5*diag(I - refCast<const wedgeFaPatch>(this->patch()).faceT());
143
144 const vector diagV(diagT.xx(), diagT.yy(), diagT.zz());
145
146 return tmp<Field<Type>>
147 (
148 new Field<Type>
149 (
150 this->size(),
151 transformMask<Type>
152 (
153 pow
154 (
155 diagV,
156 pTraits<typename powProduct<vector, pTraits<Type>::rank>
157 ::type>::zero
158 )
159 )
160 )
161 );
162}
