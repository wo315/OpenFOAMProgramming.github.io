30inline Foam::label
31Foam::ensightFaces::add(const elemType etype, label id, bool flip)
32{
33 // Linear addressing location
34 const label index = offsets_[etype] + sizes_[etype]++;
35
36 addressing()[index] = id;
37
38 if (flipMap_.size())
39 {
40 flipMap_[index] = flip;
41 }
42
43 return index;
44}
