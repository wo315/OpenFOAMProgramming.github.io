113template<class Type>
114void Foam::mappedFixedValueFvPatchField<Type>::updateCoeffs()
115{
116 if (this->updated())
117 {
118 return;
119 }
120
121 this->operator==(this->mappedField());
122
123 if (debug)
124 {
125 Info<< "mapped on field:"
126 << this->internalField().name()
127 << " patch:" << this->patch().name()
128 << " avg:" << gAverage(*this)
129 << " min:" << gMin(*this)
130 << " max:" << gMax(*this)
131 << endl;
132 }
133
134 fixedValueFvPatchField<Type>::updateCoeffs();
135}
