56void ShihQuadraticKE::correctNonlinearStress(const volTensorField& gradU)
57{
58 volSymmTensorField S(symm(gradU));
59 volTensorField W(skew(gradU));
60
61 volScalarField sBar((k_/epsilon_)*sqrt(2.0)*mag(S));
62 volScalarField wBar((k_/epsilon_)*sqrt(2.0)*mag(W));
63
64 volScalarField Cmu((2.0/3.0)/(Cmu1_ + sBar + Cmu2_*wBar));
65
66 nut_ = Cmu*sqr(k_)/epsilon_;
67 nut_.correctBoundaryConditions();
68
69 nonlinearStress_ =
70 k_*sqr(k_/epsilon_)/(Cbeta_ + pow3(sBar))
71 *(
72 Cbeta1_*dev(innerSqr(S))
73 + Cbeta2_*twoSymm(S&W)
74 + Cbeta3_*devSymm(W&W)
75 );
76}

256void ShihQuadraticKE::correct()
257{
258 if (!turbulence_)
259 {
260 return;
261 }
262
263 nonlinearEddyViscosity<incompressible::RASModel>::correct();
264
265 tmp<volTensorField> tgradU = fvc::grad(U_);
266 const volTensorField& gradU = tgradU();
267
268 volScalarField G
269 (
270 GName(),
271 (nut_*twoSymm(gradU) - nonlinearStress_) && gradU
272 );
273
274
275 // Update epsilon and G at the wall
276 epsilon_.boundaryFieldRef().updateCoeffs();
277 // Push any changed cell values to coupled neighbours
278 epsilon_.boundaryFieldRef().evaluateCoupled<coupledFvPatch>();
279
280 // Dissipation equation
281 tmp<fvScalarMatrix> epsEqn
282 (
283 fvm::ddt(epsilon_)
284 + fvm::div(phi_, epsilon_)
285 - fvm::laplacian(DepsilonEff(), epsilon_)
286 ==
287 Ceps1_*G*epsilon_/k_
288 - fvm::Sp(Ceps2_*epsilon_/k_, epsilon_)
289 );
290
291 epsEqn.ref().relax();
292 epsEqn.ref().boundaryManipulate(epsilon_.boundaryFieldRef());
293 solve(epsEqn);
294 bound(epsilon_, epsilonMin_);
295
296
297 // Turbulent kinetic energy equation
298 tmp<fvScalarMatrix> kEqn
299 (
300 fvm::ddt(k_)
301 + fvm::div(phi_, k_)
302 - fvm::laplacian(DkEff(), k_)
303 ==
304 G
305 - fvm::Sp(epsilon_/k_, k_)
306 );
307
308 kEqn.ref().relax();
309 solve(kEqn);
310 bound(k_, kMin_);
311
312
313 // Re-calculate viscosity and non-linear stress
314 correctNonlinearStress(gradU);
315}
