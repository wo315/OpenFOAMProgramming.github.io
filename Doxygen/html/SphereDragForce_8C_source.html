33template<class CloudType>
34Foam::scalar Foam::SphereDragForce<CloudType>::CdRe(const scalar Re) const
35{
36 // (AOB:Eq. 35)
37 if (Re > 1000.0)
38 {
39 return 0.424*Re;
40 }
41
42 return 24.0*(1.0 + (1.0/6.0)*pow(Re, 2.0/3.0));
43}
