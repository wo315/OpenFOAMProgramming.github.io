89bool Foam::functionObjects::fieldValue::read(const dictionary& dict)
90{
91 if (fvMeshFunctionObject::read(dict) && writeFile::read(dict))
92 {
93 if (dict != dict_)
94 {
95 dict_ = dict;
96 }
97
98 dict.readEntry("writeFields", writeFields_);
99 scaleFactor_ = dict.getOrDefault<scalar>("scaleFactor", 1.0);
100 dict.readEntry("fields", fields_);
101
102 return true;
103 }
104
105 return false;
106}
