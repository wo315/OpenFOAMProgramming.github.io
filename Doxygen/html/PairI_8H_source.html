30template<class T>
31inline int Foam::Pair<T>::compare(const Pair<T>& a, const Pair<T>& b)
32{
33 if (a.first() == b.first() && a.second() == b.second())
34 {
35 return 1;
36 }
37 if (a.first() == b.second() && a.second() == b.first())
38 {
39 return -1;
40 }
41
42 return 0;
43}
