61Foam::pointZoneSet::pointZoneSet
62(
63 const polyMesh& mesh,
64 const word& name,
65 IOobjectOption::readOption rOpt,
66 IOobjectOption::writeOption wOpt
67)
68:
69 pointSet(mesh, name, 1024), // do not read pointSet
70 mesh_(mesh),
71 addressing_()
72{
73 const pointZoneMesh& pointZones = mesh.pointZones();
74 label zoneID = pointZones.findZoneID(name);
75
76 if
77 (
78 IOobjectOption::isReadRequired(rOpt)
79 || (IOobjectOption::isReadOptional(rOpt) && zoneID != -1)
80 )
81 {
82 const pointZone& fz = pointZones[zoneID];
83 addressing_ = fz;
84 }
85
86 updateSet();
87
88 check(mesh.nPoints());
89}

126void Foam::pointZoneSet::invert(const label maxLen)
127{
128 // Count
129 label n = 0;
130
131 for (label pointi = 0; pointi < maxLen; ++pointi)
132 {
133 if (!found(pointi))
134 {
135 ++n;
136 }
137 }
138
139 // Fill
140 addressing_.setSize(n);
141 n = 0;
142
143 for (label pointi = 0; pointi < maxLen; ++pointi)
144 {
145 if (!found(pointi))
146 {
147 addressing_[n] = pointi;
148 ++n;
149 }
150 }
151 updateSet();
152}

229bool Foam::pointZoneSet::writeObject
230(
231 IOstreamOption streamOpt,
232 const bool writeOnProc
233) const
234{
235 // Write shadow pointSet
236 word oldTypeName = typeName;
237 const_cast<word&>(type()) = pointSet::typeName;
238 bool ok = pointSet::writeObject(streamOpt, writeOnProc);
239 const_cast<word&>(type()) = oldTypeName;
240
241 // Modify pointZone
242 pointZoneMesh& pointZones = const_cast<polyMesh&>(mesh_).pointZones();
243 label zoneID = pointZones.findZoneID(name());
244
245 if (zoneID == -1)
246 {
247 zoneID = pointZones.size();
248
249 pointZones.setSize(zoneID+1);
250 pointZones.set
251 (
252 zoneID,
253 new pointZone
254 (
255 name(),
256 addressing_,
257 zoneID,
258 pointZones
259 )
260 );
261 }
262 else
263 {
264 pointZones[zoneID] = addressing_;
265 }
266 pointZones.clearAddressing();
267
268 return ok && pointZones.write(writeOnProc);
269}

272void Foam::pointZoneSet::updateMesh(const mapPolyMesh& morphMap)
273{
274 // pointZone
275 labelList newAddressing(addressing_.size());
276
277 label n = 0;
278 for (const label pointi : addressing_)
279 {
280 const label newPointi = morphMap.reversePointMap()[pointi];
281 if (newPointi >= 0)
282 {
283 newAddressing[n] = newPointi;
284 ++n;
285 }
286 }
287 newAddressing.resize(n);
288
289 addressing_.transfer(newAddressing);
290
291 updateSet();
292}
