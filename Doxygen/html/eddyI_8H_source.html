108Foam::boundBox Foam::eddy::bounds(const bool global) const
109{
110 boundBox bb;
111
112 if (global)
113 {
114 bb.min() = Rpg_ & -sigma_;
115 bb.max() = Rpg_ & sigma_;
116 }
117 else
118 {
119 bb.min() = -sigma_;
120 bb.max() = sigma_;
121 }
122
123 return bb;
124}
