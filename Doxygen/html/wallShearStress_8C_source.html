62void Foam::functionObjects::wallShearStress::calcShearStress
63(
64 const volSymmTensorField& Reff,
65 volVectorField& shearStress
66)
67{
68 shearStress.dimensions().reset(Reff.dimensions());
69
70 for (const label patchi : patchSet_)
71 {
72 vectorField& ssp = shearStress.boundaryFieldRef()[patchi];
73 const vectorField& Sfp = mesh_.Sf().boundaryField()[patchi];
74 const scalarField& magSfp = mesh_.magSf().boundaryField()[patchi];
75 const symmTensorField& Reffp = Reff.boundaryField()[patchi];
76
77 ssp = (-Sfp/magSfp) & Reffp;
78 }
79}

123bool Foam::functionObjects::wallShearStress::read(const dictionary& dict)
124{
125 fvMeshFunctionObject::read(dict);
126 writeFile::read(dict);
127
128 writeFields_ = true; // May change in the future
129 dict.readIfPresent("writeFields", writeFields_);
130
131 const polyBoundaryMesh& pbm = mesh_.boundaryMesh();
132
133 patchSet_ =
134 mesh_.boundaryMesh().patchSet
135 (
136 dict.getOrDefault<wordRes>("patches", wordRes())
137 );
138
139 Info<< type() << " " << name() << ":" << nl;
140
141 if (patchSet_.empty())
142 {
143 forAll(pbm, patchi)
144 {
145 if (isA<wallPolyPatch>(pbm[patchi]))
146 {
147 patchSet_.insert(patchi);
148 }
149 }
150
151 Info<< " processing all wall patches" << nl << endl;
152 }
153 else
154 {
155 Info<< " processing wall patches: " << nl;
156 labelHashSet filteredPatchSet;
157 for (const label patchi : patchSet_)
158 {
159 if (isA<wallPolyPatch>(pbm[patchi]))
160 {
161 filteredPatchSet.insert(patchi);
162 Info<< " " << pbm[patchi].name() << endl;
163 }
164 else
165 {
166 WarningInFunction
167 << "Requested wall shear stress on non-wall boundary "
168 << "type patch: " << pbm[patchi].name() << endl;
169 }
170 }
171
172 Info<< endl;
173
174 patchSet_ = filteredPatchSet;
175 }
176
177 return true;
178}

181bool Foam::functionObjects::wallShearStress::execute()
182{
183 auto& wallShearStress =
184 mesh_.lookupObjectRef<volVectorField>(scopedName(typeName));
185
186 // Compressible
187 {
188 typedef compressible::turbulenceModel turbType;
189
190 const turbType* modelPtr =
191 findObject<turbType>(turbulenceModel::propertiesName);
192
193 if (modelPtr)
194 {
195 calcShearStress(modelPtr->devRhoReff(), wallShearStress);
196 return true;
197 }
198 }
199
200 // Incompressible
201 {
202 typedef incompressible::turbulenceModel turbType;
203
204 const turbType* modelPtr =
205 findObject<turbType>(turbulenceModel::propertiesName);
206
207 if (modelPtr)
208 {
209 calcShearStress(modelPtr->devReff(), wallShearStress);
210 return true;
211 }
212 }
213
214 FatalErrorInFunction
215 << "Unable to find turbulence model in the "
216 << "database" << exit(FatalError);
217
218 return false;
219}

222bool Foam::functionObjects::wallShearStress::write()
223{
224 const auto& wallShearStress =
225 obr_.lookupObject<volVectorField>(scopedName(typeName));
226
227 Log << type() << ' ' << name() << " write:" << nl;
228
229 if (writeFields_)
230 {
231 Log << " writing field " << wallShearStress.name() << endl;
232 wallShearStress.write();
233 }
234
235 const fvPatchList& patches = mesh_.boundary();
236
237 for (const label patchi : patchSet_)
238 {
239 const fvPatch& pp = patches[patchi];
240
241 const vectorField& ssp = wallShearStress.boundaryField()[patchi];
242
243 const vector minSsp = gMin(ssp);
244 const vector maxSsp = gMax(ssp);
245
246 if (UPstream::master())
247 {
248 writeCurrentTime(file());
249
250 file()
251 << token::TAB << pp.name()
252 << token::TAB << minSsp
253 << token::TAB << maxSsp
254 << endl;
255 }
256
257 Log << " min/max(" << pp.name() << ") = "
258 << minSsp << ", " << maxSsp << endl;
259 }
260
261 return true;
262}
