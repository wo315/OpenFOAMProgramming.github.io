73void Foam::fv::buoyancyEnergy::addSup
74(
75 const volScalarField& rho,
76 fvMatrix<scalar>& eqn,
77 const label fieldi
78)
79{
80 const uniformDimensionedVectorField& g =
81 meshObjects::gravity::New(mesh_.time());
82
83 const auto& U = mesh_.lookupObject<volVectorField>(UName_);
84
85 eqn += rho*(U&g);
