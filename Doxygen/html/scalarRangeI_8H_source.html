150inline Foam::scalar Foam::scalarRange::value() const
151{
152 switch (type_)
153 {
154 case scalarRange::EQ: // For equals, min and max are identical
155 case scalarRange::GE:
156 case scalarRange::GT:
157 return min_;
158
159 case scalarRange::LE:
160 case scalarRange::LT:
161 return max_;
162
163 case scalarRange::GE_LE:
164 // Multiply before adding to avoid possible overflow
165 return (0.5 * min_) + (0.5 * max_);
166
167 default: // NONE or ALWAYS. Zero is reasonable dummy value.
168 return 0;
169 }
170}

173inline bool Foam::scalarRange::contains(const scalar val) const
174{
175 switch (type_)
176 {
177 case scalarRange::EQ: return equal(val, min_);
178 case scalarRange::GE: return (val >= min_);
179 case scalarRange::GT: return (val > min_);
180 case scalarRange::LE: return (val <= max_);
181 case scalarRange::LT: return (val < max_);
182 case scalarRange::GE_LE: return (val >= min_ && val <= max_);
183 case scalarRange::ALWAYS: return true;
184 default: return false;
185 }
186}
