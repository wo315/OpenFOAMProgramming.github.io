41Foam::direction Foam::componentVelocityMotionSolver::cmpt
42(
43 const word& cmptName
44) const
45{
46 if (cmptName == "x")
47 {
48 return vector::X;
49 }
50 else if (cmptName == "y")
51 {
52 return vector::Y;
53 }
54 else if (cmptName == "z")
55 {
56 return vector::Z;
57 }
58 else
59 {
60 FatalErrorInFunction
61 << "Given component name " << cmptName << " should be x, y or z"
62 << exit(FatalError);
63
64 return 0;
65 }
66}
