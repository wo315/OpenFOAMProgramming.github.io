120template<class Type>
121Foam::tmp<Foam::Field<Type>> Foam::wedgeFvPatchField<Type>::snGrad() const
122{
123 const Field<Type> pif(this->patchInternalField());
124
125 return
126 (
127 transform(refCast<const wedgeFvPatch>(this->patch()).cellT(), pif) - pif
128 )*(0.5*this->patch().deltaCoeffs());

151template<class Type>
152Foam::tmp<Foam::Field<Type>>
153Foam::wedgeFvPatchField<Type>::snGradTransformDiag() const
154{

157
158 const vector diagV(diagT.xx(), diagT.yy(), diagT.zz());
159
160 return tmp<Field<Type>>

163 (
164 this->size(),
165 transformMask<Type>
166 (
167 pow
168 (
169 diagV,
170 pTraits<typename powProduct<vector, pTraits<Type>::rank>
171 ::type>::zero
172 )
173 )
174 )
