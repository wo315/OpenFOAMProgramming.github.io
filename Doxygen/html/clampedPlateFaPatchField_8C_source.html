108template<>
109void clampedPlateFaPatchField<scalar>::evaluate(const Pstream::commsTypes)
110{
111 if (!this->updated())
112 {
113 this->updateCoeffs();
114 }
115
116 Field<scalar>::operator=(Zero);
117
118 auto& iF = const_cast<Field<scalar>&>(this->primitiveField());
119
120 for (const label facei : this->patch().edgeFaces())
121 {
122 iF[facei] = Zero;
123 }
124
125 faPatchField<scalar>::evaluate();
126}
