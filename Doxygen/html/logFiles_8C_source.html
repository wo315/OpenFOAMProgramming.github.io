35void Foam::functionObjects::logFiles::createFiles()
36{
37 if (Pstream::master())
38 {
39 const word startTimeName =
40 fileObr_.time().timeName(fileObr_.time().startTime().value());
41
42 forAll(names_, i)
43 {
44 if (!filePtrs_.set(i))

50 }
51 }
52}
