33template<class Type>
34void Foam::IOField<Type>::readFromStream(const bool readOnProc)
35{
36 Istream& is = readStream(typeName, readOnProc);
37
38 if (readOnProc)
39 {
40 is >> *this;
41 }
42 close();
43}

46template<class Type>
47bool Foam::IOField<Type>::readContents()
48{
49 if (isReadRequired() || (isReadOptional() && headerOk()))
50 {
51 readFromStream();
52 return true;
53 }
54
55 return false;

184template<class Type>
185Foam::Field<Type> Foam::IOField<Type>::readContents(const IOobject& io)
186{
187 IOobject rio(io, IOobjectOption::NO_REGISTER);
188 if (rio.readOpt() == IOobjectOption::MUST_READ_IF_MODIFIED)
189 {
190 rio.readOpt(IOobjectOption::MUST_READ);
191 }
192
193 IOField<Type> reader(rio);
194
195 return Field<Type>(std::move(static_cast<Field<Type>&>(reader)));
196}
