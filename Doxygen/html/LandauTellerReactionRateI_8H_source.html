63inline Foam::scalar Foam::LandauTellerReactionRate::operator()
64(
65 const scalar p,
66 const scalar T,
67 const scalarField&
68) const
69{
70 scalar lta = A_;
71
72 if (mag(beta_) > VSMALL)
73 {
74 lta *= pow(T, beta_);
75 }
76
77 scalar expArg = 0.0;
78
79 if (mag(Ta_) > VSMALL)
80 {
81 expArg -= Ta_/T;
82 }
83
84 if (mag(B_) > VSMALL)
85 {
86 expArg += B_/cbrt(T);
87 }
88
89 if (mag(C_) > VSMALL)
90 {
91 expArg += C_/pow(T, 2.0/3.0);
92 }
93
94 if (mag(expArg) > VSMALL)
95 {
96 lta *= exp(expArg);
97 }
98
99 return lta;
100}
