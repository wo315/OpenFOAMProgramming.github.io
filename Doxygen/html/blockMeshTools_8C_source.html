36static inline const Foam::entry* resolveLabel(const entry& e, const label val)
37{
38 if (e.isStream())
39 {
40 const tokenList& toks = e.stream();
41
42 if (!toks.empty() && toks[0].isLabel(val))
43 {
44 return &e;
45 }
46 }
47
48 return nullptr;
49}

56void Foam::blockMeshTools::read
57(
58 Istream& is,
59 label& val,
60 const dictionary& dict
61)
62{
63 token t(is);
64 if (t.isLabel())
65 {
66 val = t.labelToken();
67 }
68 else if (t.isWord())
69 {
70 const word& varName = t.wordToken();
71 const entry* eptr =
72 dict.findScoped(varName, keyType::REGEX_RECURSIVE);
73
74 if (eptr)
75 {
76 // Read as label
77 val = Foam::readLabel(eptr->stream());
78 }
79 else
80 {
81 FatalIOErrorInFunction(is)
82 << "Undefined variable "
83 << varName << ". Valid variables are " << dict
84 << exit(FatalIOError);
85 }
86 }
87 else
88 {
89 FatalIOErrorInFunction(is)
90 << "Illegal token " << t.info()
91 << " when trying to read label"
92 << exit(FatalIOError);
93 }
94
95 is.fatalCheck(FUNCTION_NAME);
96}

111void Foam::blockMeshTools::write
112(
113 Ostream& os,
114 const label val,
115 const dictionary& dict
116)
117{
118 for (const entry& e : dict)
119 {
120 const entry* eptr = resolveLabel(e, val);
121 if (eptr)
122 {
123 os << eptr->keyword();
124 return;
125 }
126 }
127 os << val;
128}
