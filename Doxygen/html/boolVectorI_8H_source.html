81inline unsigned int Foam::boolVector::count(const bool on) const
82{
83 unsigned int total = 0;
84
85 for (const bool val : *this)
86 {
87 if (val) ++total;
88 }
89
90 if (!on)
91 {
92 // Return the number of bits that are OFF.
93 return (3u - total);
94 }
95
96 return total;
97}
