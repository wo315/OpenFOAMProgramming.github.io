75template<class Type>
76Foam::tmp<Foam::Field<Type>>
77Foam::sampledDistanceSurface::sampleOnIsoSurfacePoints
78(
79 const interpolation<Type>& interpolator
80) const
81{
82 if (!this->hasIsoSurface())
83 {
84 FatalErrorInFunction
85 << "cannot call without an iso-surface" << nl
86 << exit(FatalError);
87 }
88
89 // Assume volPointInterpolation for the point field!
90 const auto& volFld = interpolator.psi();
91
92 tmp<VolumeField<Type>> tvolFld(volFld);
93 tmp<PointField<Type>> tpointFld;
94
95 // Interpolated point field
96 tpointFld.reset
97 (
98 volPointInterpolation::New(tvolFld().mesh()).interpolate(tvolFld())
99 );
100
101 if (average_)
102 {
103 tvolFld.reset(pointAverage(tpointFld()));
104 }
105
106 return this->isoSurfaceInterpolate(tvolFld(), tpointFld());
107}
