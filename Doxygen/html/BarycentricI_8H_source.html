68template<class Cmpt>
69inline Cmpt Foam::Barycentric<Cmpt>::inner(const Barycentric<Cmpt>& b2) const
70{
71 const Barycentric<Cmpt>& b1 = *this;
72
73 return (b1.a()*b2.a() + b1.b()*b2.b() + b1.c()*b2.c() + b1.d()*b2.d());
74}
