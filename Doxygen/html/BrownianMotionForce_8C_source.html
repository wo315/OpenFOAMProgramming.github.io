39template<class CloudType>
40Foam::tmp<Foam::volScalarField>
41Foam::BrownianMotionForce<CloudType>::kModel() const
42{
43 const objectRegistry& obr = this->owner().mesh();
44 const word turbName =
45 IOobject::groupName
46 (
47 turbulenceModel::propertiesName,
48 this->owner().U().group()
49 );
50
51 const turbulenceModel* turb = obr.findObject<turbulenceModel>(turbName);
52
53 if (turb)
54 {
55 return turb->k();
56 }
57
58 FatalErrorInFunction
59 << "Turbulence model not found in mesh database" << nl
60 << "Database objects include: " << obr.sortedToc()
61 << abort(FatalError);
62
63 return nullptr;
64}

112template<class CloudType>
113void Foam::BrownianMotionForce<CloudType>::cacheFields(const bool store)
114{
115 if (turbulence_)
116 {
117 if (store)
118 {
119 tmp<volScalarField> tk = kModel();
120 if (tk.movable())
121 {
122 // Take ownership
123 kPtr_ = tk.ptr();
124 ownK_ = true;
125 }
126 else
127 {
128 kPtr_ = &tk.cref();
129 ownK_ = false;
130 }
131 }
132 else
133 {
134 if (ownK_)
135 {
136 deleteDemandDrivenData(kPtr_);
137 ownK_ = false;
138 }
139 }
140 }
141}

144template<class CloudType>
145Foam::forceSuSp Foam::BrownianMotionForce<CloudType>::calcCoupled
146(
147 const typename CloudType::parcelType& p,
148 const typename CloudType::parcelType::trackingData& td,
149 const scalar dt,
150 const scalar mass,
151 const scalar Re,
152 const scalar muc
153) const
154{
155 forceSuSp value(Zero);
156
157 const scalar dp = p.d();
158 const scalar Tc = td.Tc();
159
160 const scalar alpha = 2.0*lambda_/dp;
161 const scalar cc = 1.0 + alpha*(1.257 + 0.4*exp(-1.1/alpha));
162
163 // Boltzmann constant
164 const scalar kb = physicoChemical::k.value();
165
166 scalar f = 0;
167 if (turbulence_)
168 {
169 const label celli = p.cell();
170 const volScalarField& k = *kPtr_;
171 const scalar kc = k[celli];
172 const scalar Dp = kb*Tc*cc/(3*mathematical::pi*muc*dp);
173 f = sqrt(2.0*sqr(kc)*sqr(Tc)/(Dp*dt));
174 }
175 else
176 {
177 const scalar s0 =
178 216*muc*kb*Tc/(sqr(mathematical::pi)*pow5(dp)*sqr(p.rho())*cc);
179 f = mass*sqrt(mathematical::pi*s0/dt);
180 }
181
182
183 // To generate a cubic distribution (3 independent directions) :
184 // const scalar sqrt2 = sqrt(2.0);
185 // for (direction dir = 0; dir < vector::nComponents; dir++)
186 // {
187 // const scalar x = rndGen_.sample01<scalar>();
188 // const scalar eta = sqrt2*Math::erfInv(2*x - 1.0);
189 // value.Su()[dir] = f*eta;
190 // }
191
192
193 // To generate a spherical distribution:
194
195 Random& rnd = this->owner().rndGen();
196
197 const scalar theta = rnd.sample01<scalar>()*twoPi;
198 const scalar u = 2*rnd.sample01<scalar>() - 1;
199
200 const scalar a = sqrt(1 - sqr(u));
201 const vector dir(a*cos(theta), a*sin(theta), u);
202
203 value.Su() = f*mag(rnd.GaussNormal<scalar>())*dir;
204
205 return value;
206}
