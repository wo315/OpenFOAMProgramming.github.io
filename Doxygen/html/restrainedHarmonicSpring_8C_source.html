75scalar restrainedHarmonicSpring::energy(const vector r) const
76{
77 scalar magR = mag(r);
78
79 if (magR < rR_)
80 {
81 return 0.5 * springConstant_ * magSqr(r);
82 }
83 else
84 {
85 return 0.5 * springConstant_ * rR_ * rR_
86 + springConstant_ * rR_ * (magR - rR_);
87 }
88}

90
91vector restrainedHarmonicSpring::force(const vector r) const
92{
93 scalar magR = mag(r);
94
95 if (magR < rR_)
96 {
97 return -springConstant_ * r;
98 }
99 else
100 {
101 return -springConstant_ * rR_ * r / magR;
102 }
103}
