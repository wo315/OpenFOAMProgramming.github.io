100void Foam::adjointWallVelocityFvPatchVectorField::manipulateMatrix
101(
102 fvMatrix<vector>& matrix
103)
104{
105 addProfiling
106 (
107 adjointWallVelocityFvPatchVectorField,
108 "adjointWallVelocityFvPatchVectorField::manipulateMatrix"
109 );
110 // Grab ref to the diagonal matrix
111 vectorField& source = matrix.source();
112
113 // Define boundary condition type
114 typedef Foam::nutUSpaldingWallFunctionFvPatchScalarField
115 SAwallFunctionPatchField;
116
117 if
118 (
119 isA<SAwallFunctionPatchField>(boundaryContrPtr_->turbulentDiffusivity())
120 && patch().size() != 0
121 )
122 {
123 const tmp<vectorField> tnf = patch().nf();
124 const vectorField& nf = tnf();
125 const scalarField& magSf = patch().magSf();
126
127 const fvPatchField<vector>& Up = boundaryContrPtr_->Ub();
128 const fvPatchField<vector>& Uap = *this;
129
130 const vectorField Uc(Up.patchInternalField());
131 const vectorField Uc_t(Uc - (Uc & nf)*nf);
132
133 // By convention, tf has the direction of the tangent PRIMAL velocity
134 // at the first cell off the wall
135 const vectorField tf(Uc_t/mag(Uc_t));
136
137 tmp<scalarField> tnuw = boundaryContrPtr_->momentumDiffusion();
138 const scalarField& nuw = tnuw();
139 tmp<scalarField> tnu = boundaryContrPtr_->laminarDiffusivity();
140 const scalarField& nu = tnu();
141 tmp<scalarField> tyC = boundaryContrPtr_->wallDistance();
142 const scalarField& yC = tyC();
143
144 const scalarField magGradU(mag(Up.snGrad()));
145 const scalarField vtau(sqrt(nuw*magGradU));
146 const scalarField uPlus(mag(Uc)/vtau);
147 const scalarField yPlus(yC*vtau/nu);
148 const scalarField kUu(min(kappa_*uPlus, scalar(50)));
149 const scalarField auxA((kappa_/E_)*(exp(kUu)-1 - kUu - 0.5*kUu*kUu));
150 const scalarField auxB(-(1 + auxA)/(yPlus + uPlus*(1 + auxA)));
151
152 // Tangent components are according to tf
153 tmp<vectorField> tsource = boundaryContrPtr_->normalVelocitySource();
154 const scalarField rt(tsource() & tf);
155 const scalarField Uap_t(Uap & tf);
156
157 forAll(Up, faceI)
158 {
159 label cellI = patch().faceCells()[faceI];
160 source[cellI] +=
161 2*auxB[faceI]*vtau[faceI]*((rt[faceI] + Uap_t[faceI]))
162 *(Uc[faceI]/mag(Uc[faceI]))*magSf[faceI];
163 }
164 }
165}

168void Foam::adjointWallVelocityFvPatchVectorField::updateCoeffs()
169{
170 if (updated())
171 {
172 return;
173 }
174
175 const fvPatchField<vector>& Up = boundaryContrPtr_->Ub();
176
177 // Patch geometry
178 tmp<vectorField> tnf = patch().nf();
179 const vectorField& nf = tnf();
180
181 // Internal fields
182 vectorField Uac(this->patchInternalField());
183 vectorField Uc(Up.patchInternalField());
184
185 // Tangent vector based on the direction of Vc
186 vectorField Uc_t(Uc - (Uc & nf)*nf);
187 vectorField tf1(Uc_t/mag(Uc_t));
188
189 // Tangent vector as the cross product of tf1 x nf
190 vectorField tf2((tf1 ^ nf)/mag(tf1 ^ nf));
191
192 // Normal adjoint component comes from the objective function
193 tmp<vectorField> tsource = boundaryContrPtr_->normalVelocitySource();
194 vectorField Uan(-(tsource() & nf)*nf);
195
196 // Tangential adjoint velocity in the t1 direction depends on the primal
197 // wall function used
198 vectorField Uap_t1(patch().size(), Zero);
199 typedef Foam::nutUSpaldingWallFunctionFvPatchScalarField
200 SAwallFunctionPatchField;
201
202 const fvPatchScalarField& nutb = boundaryContrPtr_->turbulentDiffusivity();
203 if (isA<SAwallFunctionPatchField>(nutb))
204 {
205 Uap_t1 = (Uac & tf1)*tf1;
206 // leaving out second term for now
207 //- (1./delta)*((gradUaC & nf) & tf1)*tf1;
208 }
209 else
210 {
211 Uap_t1 = - (tsource() & tf1)*tf1;
212 }
213
214 // Adjoint velocity in the t2 direction
215 vectorField Uap_t2(-(tsource() & tf2)*tf2);
216
217 operator==(Uan + Uap_t1 + Uap_t2);
218
219 fixedValueFvPatchVectorField::updateCoeffs();
220}
