106 ),

108{
109 if (!isA<mappedPatchBase>(this->patch().patch()))

111 FatalErrorInFunction

113 << "\n for patch " << p.name()
114 << " of field " << internalField().name()
115 << " in file " << internalField().objectPath()
116 << exit(FatalError);
117 }
118
119 WarningInFunction
120 << "This BC has been superseded by "
121 << "compressible::turbulentTemperatureRadCoupledMixed "
122 << "which has more functionalities and it can handle "
123 << "the assemble coupled option for energy. "
124 << endl;
125
126 // Read list of layers
127 if (dict.readIfPresent("thicknessLayers", thicknessLayers_))
128 {
129 dict.readEntry("kappaLayers", kappaLayers_);
130 }
131 // Read single additional PatchFunction1
132 thicknessLayer_ = PatchFunction1<scalar>::NewIfPresent
133 (
134 p.patch(),
135 "thicknessLayer",
136 dict
137 );
138 kappaLayer_ = PatchFunction1<scalar>::NewIfPresent
139 (
140 p.patch(),
141 "kappaLayer",
142 dict
143 );
144
145
146 this->readValueEntry(dict, IOobjectOption::MUST_READ);
147
148 if (this->readMixedEntries(dict))
149 {
150 // Full restart
151 }
152 else
153 {
154 // Start from user entered data. Assume fixedValue.
155 refValue() = *this;
156 refGrad() = Zero;
157 valueFraction() = 1.0;
158 }
159
160// This blocks (crashes) with more than two worlds!
161//

263tmp<Foam::scalarField>
264turbulentTemperatureCoupledBaffleMixedFvPatchScalarField::kappa
265(
266 const scalarField& Tp
267) const
268{
269 // Get kappa from relevant thermo
270 tmp<scalarField> tk(temperatureCoupledBase::kappa(Tp));
271
272 // Optionally modify with explicit resistance
273 if (thicknessLayer_ || thicknessLayers_.size())
274 {
275 scalarField KDelta(tk*patch().deltaCoeffs());
276
277 // Harmonic averaging of kappa*deltaCoeffs
278 {
279 KDelta = 1.0/KDelta;
280 if (thicknessLayer_)
281 {
282 const scalar t = db().time().timeOutputValue();
283 KDelta +=
284 thicknessLayer_().value(t)
285 /kappaLayer_().value(t);
286 }
287 if (thicknessLayers_.size())
288 {
289 forAll(thicknessLayers_, iLayer)
290 {
291 KDelta += thicknessLayers_[iLayer]/kappaLayers_[iLayer];
292 }
293 }
294 KDelta = 1.0/KDelta;
295 }
296
297 // Update kappa from KDelta
298 tk = KDelta/patch().deltaCoeffs();
299 }
300
301 return tk;
302}

305void turbulentTemperatureCoupledBaffleMixedFvPatchScalarField::updateCoeffs()
306{
307 if (updated())
308 {
309 return;
310 }
311
312 // Since we're inside initEvaluate/evaluate there might be processor
313 // comms underway. Change the tag we use.
314 const int oldTag = UPstream::msgType();
315 UPstream::msgType() = oldTag+1;
316
317 // Get the coupling information from the mappedPatchBase
318 const mappedPatchBase& mpp =
319 mappedPatchFieldBase<scalar>::mapper
320 (
321 patch(),
322 this->internalField()
323 );
324
325 const scalarField& Tp = *this;
326 const scalarField kappaTp(kappa(Tp));
327 const tmp<scalarField> myKDelta = kappaTp*patch().deltaCoeffs();
328
329
330 scalarField nbrIntFld;
331 scalarField nbrKDelta;
332 if (mpp.sameWorld())
333 {
334 // Same world so lookup
335 const auto& nbrMesh = refCast<const fvMesh>(mpp.sampleMesh());
336 const label nbrPatchID = mpp.samplePolyPatch().index();
337 const auto& nbrPatch = nbrMesh.boundary()[nbrPatchID];
338
339 const turbulentTemperatureCoupledBaffleMixedFvPatchScalarField&
340 nbrField =
341 refCast
342 <
343 const turbulentTemperatureCoupledBaffleMixedFvPatchScalarField
344 >
345 (
346 nbrPatch.lookupPatchField<volScalarField>(TnbrName_)
347 );
348
349 // Swap to obtain full local values of neighbour K*delta
350 nbrIntFld = nbrField.patchInternalField();
351 nbrKDelta = nbrField.kappa(nbrField)*nbrPatch.deltaCoeffs();
352 }
353 else
354 {
355 // Different world so use my region,patch. Distribution below will
356 // do the reordering.
357 nbrIntFld = patchInternalField();
358 nbrKDelta = myKDelta.ref();
359 }
360 distribute(this->internalField().name() + "_value", nbrIntFld);
361 distribute(this->internalField().name() + "_weights", nbrKDelta);
362
363
364 // Both sides agree on
365 // - temperature : (myKDelta*fld + nbrKDelta*nbrFld)/(myKDelta+nbrKDelta)
366 // - gradient : (temperature-fld)*delta
367 // We've got a degree of freedom in how to implement this in a mixed bc.
368 // (what gradient, what fixedValue and mixing coefficient)
369 // Two reasonable choices:
370 // 1. specify above temperature on one side (preferentially the high side)
371 // and above gradient on the other. So this will switch between pure
372 // fixedvalue and pure fixedgradient
373 // 2. specify gradient and temperature such that the equations are the
374 // same on both sides. This leads to the choice of
375 // - refGradient = zero gradient
376 // - refValue = neighbour value
377 // - mixFraction = nbrKDelta / (nbrKDelta + myKDelta())
378
379 this->refValue() = nbrIntFld;
380 this->refGrad() = Zero;
381 this->valueFraction() = nbrKDelta/(nbrKDelta + myKDelta());
382
383 mixedFvPatchScalarField::updateCoeffs();
384
385 if (debug)
386 {
387 scalar Q = gSum(kappaTp*patch().magSf()*snGrad());
388
389 Info<< patch().boundaryMesh().mesh().name() << ':'
390 << patch().name() << ':'
391 << this->internalField().name() << " <- "
392 << mpp.sampleRegion() << ':'
393 << mpp.samplePatch() << ':'
394 << this->internalField().name() << " :"
395 << " heat transfer rate:" << Q
396 << " walltemperature "
397 << " min:" << gMin(*this)
398 << " max:" << gMax(*this)
399 << " avg:" << gAverage(*this)
400 << endl;
401 }
402
403 // Restore tag
404 UPstream::msgType() = oldTag;
405}
