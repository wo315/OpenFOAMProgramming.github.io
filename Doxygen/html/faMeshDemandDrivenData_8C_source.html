47// Define an area-weighted normal for three points (defining a triangle)
48// (p0, p1, p2) are the base, first, second points respectively
49//
50// From the original Tukovic code:
51//
52// vector n = (d1^d2)/mag(d1^d2);
53// scalar sinAlpha = mag(d1^d2)/(mag(d1)*mag(d2));
54// scalar w = sinAlpha/(mag(d1)*mag(d2));
55//
56// ie, normal weighted by area, sine angle and inverse distance squared.
57// - area : larger weight for larger areas
58// - sin : lower weight for narrow angles (eg, shards)
59// - inv distance squared : lower weights for distant points
60//
61// The above refactored, with 0.5 for area:
62//
63// (d1 ^ d2) / (2 * magSqr(d1) * magSqr(d2))
64
65static inline vector areaInvDistSqrWeightedNormal
66(
67 const vector& a,
68 const vector& b
69)
70{
71 const scalar s(2*magSqr(a)*magSqr(b));
72
73 return s < ROOTVSMALL ? Zero : (a ^ b) / s;
74}

77// The area normal for the face dual (around base-point)
78// described by the right/left edge points and the centre point
79//
80// The adjustment for 1/2 edge point (the dual point) is done internally
81static inline vector areaInvDistSqrWeightedNormalDualEdge
82(
83 const point& basePoint,
84 const point& rightPoint,
85 const point& leftPoint,
86 const point& centrePoint
87)
88{
89 const vector mid(centrePoint - basePoint);
90
91 return
92 (
93 areaInvDistSqrWeightedNormal
94 (
95 0.5*(rightPoint - basePoint), // vector to right 1/2 edge
96 mid
97 )
98 + areaInvDistSqrWeightedNormal
99 (
100 mid,
101 0.5*(leftPoint - basePoint) // vector to left 1/2 edge
102 )
103 );
104}

107// Weighted area normal for the face triangle (around base-point)
108// to the edge points and the centre point.
109// Used for example for area-weighted edge normals
110// ---------------------------------------------------------------------------
111// own |
112// * | Don't bother trying to split the triangle into
113// /.\ | sub-triangles. Planar anyhow and skew weighting
114// / . \ | is less relevant here.
115// / . \ |
116// / . \ | Note: need negative for neighbour side orientation.
117// e0 *----:----* e1 |
118// ---------------------------------------------------------------------------
119
120static inline vector areaInvDistSqrWeightedNormalFaceTriangle
121(
122 const linePointRef& edgeLine,
123 const point& faceCentre
124)
125{
126 return
127 (
128 areaInvDistSqrWeightedNormal
129 (
130 (edgeLine.a() - faceCentre), // From centre to right edge
131 (edgeLine.b() - faceCentre) // From centre to left edge
132 )
133 );
134}

156// Calculate transform tensor with reference vector (unitAxis1)
157// and direction vector (axis2).
158//
159// This is nearly identical to the meshTools axesRotation
160// with an E3_E1 transformation with the following exceptions:
161//
162// - axis1 (e3 == unitAxis1): is already normalized (unit vector)
163// - axis2 (e1 == dirn): no difference
164// - transformation is row-vectors, not column-vectors
165static inline tensor rotation_e3e1
166(
167 const vector& unitAxis1,
168 vector dirn
169)
170{
171 dirn.removeCollinear(unitAxis1);
172 dirn.normalise();
173
174 // Set row vectors
175 return tensor
176 (
177 dirn,
178 (unitAxis1^dirn),
179 unitAxis1
180 );
181}

289//- Calculate the 'Le' vector from faceCentre to edge centre
290// using the edge normal to correct for curvature
291//
292// Normalise and rescaled to the edge length
293static inline vector calcLeVector
294(
295 const point& faceCentre,
296 const linePointRef& edgeLine,
297 const vector& edgeNormal // (unit or area normal)
298)
299{
300 const vector centreToEdge(edgeLine.centre() - faceCentre);
301
302 vector leVector(edgeLine.vec() ^ edgeNormal);
303
304 scalar s(mag(leVector));
305
306 if (s < ROOTVSMALL)
307 {
308 // The calculated edgeNormal somehow degenerate and thus a
309 // bad cross-product?
310 // Revert to basic centre -> edge
311
312 leVector = centreToEdge;
313 leVector.removeCollinear(edgeLine.unitVec());
314 s = mag(leVector);
315
316 if (s < ROOTVSMALL)
317 {
318 // Unlikely that this should happen
319 return Zero;
320 }
321
322 leVector *= edgeLine.mag()/s;
323 }
324 else
325 {
326 // The additional orientation is probably unnecessary
327 leVector *= edgeLine.mag()/s * sign(leVector & centreToEdge);
328 }
329
330 return leVector;
331}

339// Impose a minimum (edge) vector length
340// - disallow any mag(vec) < SMALL
341static inline void imposeMinVectorLength(vectorField& fld)
342{
343 // If it is small, no orientation information reasonably possible
344 // so use a vector(1,1,1) * sqrt(1/3)*min-length
345
346 // sqrt(1/3) = 0.5773502691896257, but slightly rounded down
347 const vector minVector(vector::uniform(0.57735*SMALL));
348 const scalar minLenSqr(SMALL*SMALL);
349
350 for (vector& v : fld)
351 {
352 if (v.magSqr() < minLenSqr)
353 {
354 v = minVector;
355 }
356 }
357}

364Foam::tmp<Foam::vectorField> Foam::faMesh::calcRawEdgeNormals(int order) const
365{
366 // Return edge normals with flat boundary addressing
367 auto tedgeNormals = tmp<vectorField>::New(nEdges_);
368 auto& edgeNormals = tedgeNormals.ref();
369
370 // Need face centres
371 const areaVectorField& fCentres = areaCentres();
372
373 // Also need local points
374 const pointField& localPoints = points();
375
376
377 if (order < 0)
378 {
379 // geometryOrder (-1): no communication
380
381 // Simple (primitive) edge normal calculations.
382 // These are primarly designed to avoid any communication
383 // but are thus necessarily inconsistent across processor boundaries!
384
385 WarningInFunction
386 << "Using geometryOrder < 0 : "
387 "simplified edge area-normals, without processor connectivity"
388 << endl;
389
390 // Internal (edge normals) - contributions from owner/neighbour
391 for (label edgei = 0; edgei < nInternalEdges_; ++edgei)
392 {
393 const linePointRef edgeLine(edges_[edgei].line(localPoints));
394
395 edgeNormals[edgei] =
396 (
397 areaNormalFaceTriangle
398 (
399 edgeLine,
400 fCentres[edgeOwner()[edgei]]
401 )
402 // NB: reversed sign (edge orientation flipped for neighbour)
403 - areaNormalFaceTriangle
404 (
405 edgeLine,
406 fCentres[edgeNeighbour()[edgei]]
407 )
408 );
409 }
410
411 // Boundary (edge normals). Like above, but only has owner
412 for (label edgei = nInternalEdges_; edgei < nEdges_; ++edgei)
413 {
414 const linePointRef edgeLine(edges_[edgei].line(localPoints));
415
416 edgeNormals[edgei] =
417 (
418 areaNormalFaceTriangle
419 (
420 edgeLine,
421 fCentres[edgeOwner()[edgei]]
422 )
423 );
424 }
425 }
426 else
427 {
428 // geometryOrder (0) - no communication
429 // otherwise with communication
430
431 if (order < 1)
432 {
433 WarningInFunction
434 << "Using geometryOrder == 0 : "
435 "weighted edge normals, without processor connectivity"
436 << endl;
437 }
438
439 // Internal (edge normals)
440 // - area-weighted contributions from owner/neighbour
441 for (label edgei = 0; edgei < nInternalEdges_; ++edgei)
442 {
443 const linePointRef edgeLine(edges_[edgei].line(localPoints));
444
445 edgeNormals[edgei] =
446 (
447 areaInvDistSqrWeightedNormalFaceTriangle
448 (
449 edgeLine,
450 fCentres[edgeOwner()[edgei]]
451 )
452 // NB: reversed sign (edge orientation flipped for neighbour)
453 - areaInvDistSqrWeightedNormalFaceTriangle
454 (
455 edgeLine,
456 fCentres[edgeNeighbour()[edgei]]
457 )
458 );
459 }
460
461 // Boundary (edge normals). Like above, but only has owner
462 for (label edgei = nInternalEdges_; edgei < nEdges_; ++edgei)
463 {
464 const linePointRef edgeLine(edges_[edgei].line(localPoints));
465
466 edgeNormals[edgei] =
467 (
468 areaInvDistSqrWeightedNormalFaceTriangle
469 (
470 edgeLine,
471 fCentres[edgeOwner()[edgei]]
472 )
473 );
474 }
475 }
476
477
478 // Boundary edge corrections
479 bitSet nbrBoundaryAdjust;
480
481 // Processor-processor first (for convenience)
482 if (order >= 1)
483 {
484 DynamicList<UPstream::Request> requests
485 (
486 2*(boundary().size() - boundary().nNonProcessor())
487 );
488
489 PtrList<vectorField> nbrEdgeNormals(boundary().size());
490
491 // Prefer our own slicing into the raw list (instead of patchSlice).
492 // Although patchSlice does properly handle the start() offset in
493 // the presence of emptyPaPatch, we are slightly paranoia,
494 // and want to avoid kicking off the patchStarts() data...
495
496 label patchOffset = nInternalEdges_;
497
498 forAll(boundary(), patchi)
499 {
500 const faPatch& fap = boundary()[patchi];
501
502 auto edgeNorms = edgeNormals.slice(patchOffset, fap.nEdges());
503 patchOffset += edgeNorms.size();
504
505 const auto* fapp = isA<processorFaPatch>(fap);
506
507 if (fapp)
508 {
509 const label nbrProci = fapp->neighbProcNo();
510
511 if (UPstream::parRun() && !edgeNorms.empty())
512 {
513 nbrEdgeNormals.emplace(patchi, edgeNorms.size());
514
515 // Recv accumulated weighted edge normals
516 UIPstream::read
517 (
518 requests.emplace_back(),
519 nbrProci,
520 nbrEdgeNormals[patchi]
521 );
522
523 // Send accumulated weighted edge normals
524 UOPstream::write
525 (
526 requests.emplace_back(),
527 nbrProci,
528 edgeNorms
529 );
530 }
531 }
532 else if (isA<wedgeFaPatch>(fap))
533 {
534 // Correct wedge edges
535 const auto& wedgePatch = refCast<const wedgeFaPatch>(fap);
536
537 const vector wedgeNorm
538 (
539 transform
540 (
541 wedgePatch.edgeT(),
542 wedgePatch.centreNormal()
543 ).normalise()
544 );
545
546 for (vector& e : edgeNorms)
547 {
548 e.removeCollinear(wedgeNorm);
549 }
550 }
551 // TBD:
556 else if (correctPatchPointNormals(patchi) && !fap.coupled())
557 {
558 // Neighbour correction requested
559 nbrBoundaryAdjust.set(patchi);
560 }
561 }
562
563 // Wait for outstanding requests
564 UPstream::waitRequests(requests);
565
566 if (!requests.empty())
567 {
568 // Add in received values
569
570 patchOffset = nInternalEdges_;
571
572 forAll(boundary(), patchi)
573 {
574 const faPatch& fap = boundary()[patchi];
575
576 auto edgeNorms = edgeNormals.slice(patchOffset, fap.nEdges());
577 patchOffset += edgeNorms.size();
578
579 if (nbrEdgeNormals.set(patchi))
580 {
581 const vectorField& nbrNorms = nbrEdgeNormals[patchi];
582
583 forAll(edgeNorms, patchEdgei)
584 {
585 edgeNorms[patchEdgei] += nbrNorms[patchEdgei];
586 }
587 }
588 }
589 }
590 }
591
592
593 // Apply boundary edge corrections
594 if
595 (
596 order >= 1
597 && hasPatchPointNormalsCorrection()
598 && returnReduceOr(nbrBoundaryAdjust.any())
599 )
600 {
601 DebugInFunction
602 << "Apply " << nbrBoundaryAdjust.count()
603 << " boundary neighbour corrections" << nl;
604
605 // Collect face normals, per boundary edge
606
607 (void)this->haloFaceNormals();
608
609 // Using our own slicing (instead of patchSlice) - see above
610 label patchOffset = nInternalEdges_;
611
612 for (const label patchi : nbrBoundaryAdjust)
613 {
614 const faPatch& fap = boundary()[patchi];
615
616 auto edgeNorms = edgeNormals.slice(patchOffset, fap.nEdges());
617 patchOffset += edgeNorms.size();
618
619 if (fap.ngbPolyPatchIndex() < 0)
620 {
621 FatalErrorInFunction
622 << "Neighbour polyPatch index is not defined "
623 << "for faPatch " << fap.name()
624 << abort(FatalError);
625 }
626
627 // Apply the correction
628
629 // Note from Zeljko Tukovic:
630 //
631 // This posibility is used for free-surface tracking
632 // calculations to enforce 90 deg contact angle between
633 // finite-area mesh and neighbouring polyPatch. It is very
634 // important for curvature calculation to have correct normal
635 // at contact line points.
636
637 vectorField nbrNorms(this->haloFaceNormals(patchi));
638
639 const label nPatchEdges = min(edgeNorms.size(), nbrNorms.size());
640
641 for (label patchEdgei = 0; patchEdgei < nPatchEdges; ++patchEdgei)
642 {
643 edgeNorms[patchEdgei].removeCollinear(nbrNorms[patchEdgei]);
644 }
645 }
646 }
647
648
649 // Remove collinear components and normalise
650
651 forAll(edgeNormals, edgei)
652 {
653 const linePointRef edgeLine(edges_[edgei].line(localPoints));
654
655 edgeNormals[edgei].removeCollinear(edgeLine.unitVec());
656 edgeNormals[edgei].normalise();
657 }
658
659 imposeMinVectorLength(edgeNormals);
660
661 return tedgeNormals;
662}

665void Foam::faMesh::calcLe() const
666{
667 DebugInFunction
668 << "Calculating local edges" << endl;
669
670 if (LePtr_)
671 {
672 FatalErrorInFunction
673 << "LePtr_ already allocated"
674 << abort(FatalError);
675 }
676
677 LePtr_ =
678 new edgeVectorField
679 (
680 IOobject
681 (
682 "Le",
683 mesh().pointsInstance(),
684 faMesh::meshSubDir,
685 faMesh::thisDb(),
686 IOobject::NO_READ,
687 IOobject::NO_WRITE,
688 IOobject::NO_REGISTER
689 ),
690 *this,
691 dimLength
692 // -> calculatedType()
693 );
694 edgeVectorField& Le = *LePtr_;
695
696 // Need face centres
697 const areaVectorField& fCentres = areaCentres();
698
699 // Also need local points
700 const pointField& localPoints = points();
701
702
703 // The edgeAreaNormals _may_ use communication (depends on geometryOrder)
704
705 {
706 const edgeVectorField& edgeNormals = edgeAreaNormals();
707
708 // Internal (edge vector)
709 {
710 vectorField& fld = Le.primitiveFieldRef();
711 for (label edgei = 0; edgei < nInternalEdges_; ++edgei)
712 {
713 fld[edgei] = calcLeVector
714 (
715 fCentres[edgeOwner()[edgei]],
716 edges_[edgei].line(localPoints),
717 edgeNormals[edgei]
718 );
719 }
720 }
721
722 // Boundary (edge vector)
723 forAll(boundary(), patchi)
724 {
725 const faPatch& fap = boundary()[patchi];
726 vectorField& pfld = Le.boundaryFieldRef()[patchi];
727
728 const vectorField& bndEdgeNormals =
729 edgeNormals.boundaryField()[patchi];
730
731 label edgei = fap.start();
732
733 forAll(pfld, patchEdgei)
734 {
735 pfld[patchEdgei] = calcLeVector
736 (
737 fCentres[edgeOwner()[edgei]],
738 edges_[edgei].line(localPoints),
739 bndEdgeNormals[patchEdgei]
740 );
741
742 ++edgei;
743 }
744 }
745 }
746
747
748 // Impose a minimum (edge) vector length
749
750 // Internal (edge vector)
751 {
752 imposeMinVectorLength(Le.primitiveFieldRef());
753 }
754
755 // Boundary (edge vector)
756 for (vectorField& pfld : Le.boundaryFieldRef())
757 {
758 imposeMinVectorLength(pfld);
759 }
760}

1492// Point normal calculations
1493// ~~~~~~~~~~~~~~~~~~~~~~~~~
1494
1495// Revised method (general)
1496// ------------------------
1497//
1498// - For each patch face obtain a centre point (mathematical avg)
1499// and use that to define the face dual as a pair of triangles:
1500// - tri1: base-point / mid-side of right edge / face centre
1501// - tri2: base-point / face centre / mid-side of left edge
1502//
1503// - Walk all face points, inserting directly into the corresponding
1504// locations. No distinction between internal or boundary points (yet).
1505//
1506// Revised method (boundary correction)
1507// ------------------------------------
1508//
1509// - correct wedge directly, use processor patch information to exchange
1510// the current summed values. [No change from original].
1511//
1512// - explicit correction of other boundaries.
1513// Use the new boundary halo information for the face normals.
1514// Calculate the equivalent face-point normals locally and apply
1515// correction as before.
1516
1517void Foam::faMesh::calcPointAreaNormals(vectorField& result) const
1518{
1519 DebugInFunction
1520 << "Calculating pointAreaNormals : face dual method" << endl;
1521
1522 result.resize_nocopy(nPoints());
1523 result = Zero;
1524
1525 const pointField& points = patch().localPoints();
1526 const faceList& faces = patch().localFaces();
1527
1528
1529 // Loop over all faces
1530
1531 for (const face& f : faces)
1532 {
1533 const label nVerts(f.size());
1534
1535 point centrePoint(Zero);
1536 for (const label fp : f)
1537 {
1538 centrePoint += points[fp];
1539 }
1540 centrePoint /= nVerts;
1541
1542 for (label i = 0; i < nVerts; ++i)
1543 {
1544 const label pt0 = f.thisLabel(i); // base
1545
1546 result[pt0] +=
1547 areaInvDistSqrWeightedNormalDualEdge
1548 (
1549 points[pt0], // base
1550 points[f.nextLabel(i)], // right
1551 points[f.prevLabel(i)], // left
1552 centrePoint
1553 );
1554 }
1555 }
1556
1557
1558 // Boundary edge corrections
1559 bitSet nbrBoundaryAdjust;
1560
1561 forAll(boundary(), patchi)
1562 {
1563 const faPatch& fap = boundary()[patchi];

1567 // Correct wedge points
1568
1569 const auto& wedgePatch = refCast<const wedgeFaPatch>(fap);
1570
1571 const labelList& patchPoints = wedgePatch.pointLabels();
1572
1573 const vector N
1574 (
1575 transform
1576 (
1577 wedgePatch.edgeT(),
1578 wedgePatch.centreNormal()
1579 ).normalise()
1580 );
1581
1582 for (const label pti : patchPoints)
1583 {
1584 result[pti].removeCollinear(N);
1585 }
1586
1587 // Axis point correction
1588 if (wedgePatch.axisPoint() > -1)
1589 {
1590 result[wedgePatch.axisPoint()] =
1591 wedgePatch.axis()
1592 *(
1593 wedgePatch.axis()
1594 &result[wedgePatch.axisPoint()]
1595 );
1596 }
1597 }
1598 else if (Pstream::parRun() && isA<processorFaPatch>(fap))

1613 UIndirectList<vector>(result, patchPoints)
1614 );

1618 (
1619 UPstream::commsTypes::blocking,

1622 );
1623 }

1629 UIPstream::read
1630 (
1631 UPstream::commsTypes::blocking,

1633 patchPointNormals
1634 );
1635 }
1636
1637 for (const label pti : nonGlobalPatchPoints)
1638 {

1640 patchPointNormals[nbrPatchPoints[pti]];

1642 }

1648 else if (correctPatchPointNormals(patchi) && !fap.coupled())

1650 // Neighbour correction requested

1652 }
1653 }

1665 );

1670 Zero
1671 );
1672
1673 forAll(addr, i)
1674 {
1675 gpNormals[addr[i]] += spNormals[i];
1676 }
1677
1678 Pstream::combineReduce(gpNormals, plusEqOp<vectorField>());
1679
1680 // Extract local data
1681 forAll(addr, i)
1682 {
1683 spNormals[i] = gpNormals[addr[i]];
1684 }
1685
1686 forAll(spNormals, pointI)
1687 {
1688 result[spLabels[pointI]] = spNormals[pointI];
1689 }
1690 }
1691
1692
1693 if
1694 (
1695 hasPatchPointNormalsCorrection()
1696 && returnReduceOr(nbrBoundaryAdjust.any())
1697 )
1698 {
1699 DebugInFunction
1700 << "Apply " << nbrBoundaryAdjust.count()
1701 << " boundary neighbour corrections" << nl;

1704

1707 const auto& haloNormals = this->haloFaceNormals();
