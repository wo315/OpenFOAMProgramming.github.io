30inline char* Foam::CStringList::stringCopy(char *dest, const std::string& src)
31{
32 // Like string::copy() but without extra checks
33 const size_t len = src.length();
34 for (size_t i = 0; i < len; ++i)
35 {
36 *dest = src[i];
37 ++dest;
38 }
39 *(dest++) = '\0';
40
41 return dest;
42}
