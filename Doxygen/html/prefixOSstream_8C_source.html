35inline void Foam::prefixOSstream::checkWritePrefix()
36{
37 if (printPrefix_ && !prefix_.empty())
38 {
39 OSstream::write(prefix_.c_str());
40 printPrefix_ = false;
41 }
42}

75Foam::Ostream& Foam::prefixOSstream::write(const char c)
76{
77 checkWritePrefix();
78 OSstream::write(c);
79
80 if (c == token::NL)
81 {
82 printPrefix_ = true;
83 }
84
85 return *this;
86}

89Foam::Ostream& Foam::prefixOSstream::write(const char* str)
90{
91 checkWritePrefix();
92 OSstream::write(str);
93
94 const size_t len = strlen(str);
95 if (len && str[len-1] == token::NL)
96 {
97 printPrefix_ = true;
98 }
99
100 return *this;
101}
