57template<class CloudType>
58void Foam::WeberNumberReacting<CloudType>::postEvolve
59(
60 const typename parcelType::trackingData& td
61)
62{
63 const auto& c = this->owner();
64
65 auto* resultPtr = c.template getObjectPtr<IOField<scalar>>("We");
66
67 if (!resultPtr)
68 {
69 resultPtr = new IOField<scalar>
70 (
71 IOobject
72 (
73 "We",
74 c.time().timeName(),
75 c,
76 IOobject::NO_READ,
77 IOobject::NO_WRITE,
78 IOobject::REGISTER
79 )
80 );
81
82 resultPtr->store();
83 }
84 auto& We = *resultPtr;
85
86 We.resize(c.size());
87
88 const auto& thermo = c.db().template lookupObject<SLGThermo>("SLGThermo");
89 const auto& liquids = thermo.liquids();
90
91 const auto& UInterp = td.UInterp();
92 const auto& pInterp = td.pInterp();
93 const auto& rhoInterp = td.rhoInterp();
94
95 label parceli = 0;
96 forAllConstIters(c, parcelIter)
97 {
98 const parcelType& p = parcelIter();
99
100 const auto& coords = p.coordinates();
101 const auto& tetIs = p.currentTetIndices();
102
103 const vector Uc(UInterp.interpolate(coords, tetIs));
104
105 const scalar pc =
106 max
107 (
108 pInterp.interpolate(coords, tetIs),
109 c.constProps().pMin()
110 );
111
112 const scalar rhoc(rhoInterp.interpolate(coords, tetIs));
113 const scalarField X(liquids.X(p.YLiquid()));
114 const scalar sigma = liquids.sigma(pc, p.T(), X);
115
116 We[parceli++] = rhoc*magSqr(p.U() - Uc)*p.d()/sigma;
117 }
118
119 const bool haveParticles = c.size();
120 if (c.time().writeTime() && returnReduceOr(haveParticles))
121 {
122 We.write(haveParticles);
123 }
124}
