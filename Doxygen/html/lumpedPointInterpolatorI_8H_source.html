169template<class T>
170inline T Foam::lumpedPointInterpolator::interpolate(const UList<T>& input) const
171{
172 if (nearest_ == -1)
173 {
174 return Zero;
175 }
176 else if (next1_ == -1 || next1_ == nearest_)
177 {
178 return input[nearest_];
179 }
180
181 return
182 (
183 input[nearest_] * (1-weight1_-weight2_)
184 + input[next1_] * (weight1_)
185 + input[next2_] * (weight2_)
186 );
187}
