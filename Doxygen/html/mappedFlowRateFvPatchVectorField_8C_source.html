107void Foam::mappedFlowRateFvPatchVectorField::updateCoeffs()
108{
109 if (updated())
110 {
111 return;
112 }
113
114 // Since we're inside initEvaluate/evaluate there might be processor
115 // comms underway. Change the tag we use.
116 int oldTag = UPstream::msgType();
117 UPstream::msgType() = oldTag+1;
118
119 // Get the coupling information from the mappedPatchBase
120 const mappedPatchBase& mpp = refCast<const mappedPatchBase>
121 (
122 patch().patch()
123 );
124 const polyMesh& nbrMesh = mpp.sampleMesh();
125 const fvPatch& nbrPatch = refCast<const fvMesh>
126 (
127 nbrMesh
128 ).boundary()[mpp.samplePolyPatch().index()];
129
130 scalarList phi
131 (
132 nbrPatch.lookupPatchField<surfaceScalarField>(nbrPhiName_)
133 );
134
135 mpp.distribute(phi);
136
137
138 const surfaceScalarField& phiName =
139 db().lookupObject<surfaceScalarField>(phiName_);
140
141 scalarField U(-phi/patch().magSf());
142
143 vectorField n(patch().nf());
144
145 if (phiName.dimensions() == dimVolume/dimTime)
146 {
147 // volumetric flow-rate
148 operator==(n*U);
149 }
150 else if (phiName.dimensions() == dimMass/dimTime)
151 {
152 const auto& rhop = patch().lookupPatchField<volScalarField>(rhoName_);
153
154 // mass flow-rate
155 operator==(n*U/rhop);
156
157 if (debug)
158 {
159 scalar phi = gSum(rhop*(*this) & patch().Sf());
160 Info<< patch().boundaryMesh().mesh().name() << ':'
161 << patch().name() << ':'
162 << this->internalField().name() << " <- "
163 << nbrMesh.name() << ':'
164 << nbrPatch.name() << ':'
165 << this->internalField().name() << " :"
166 << " mass flux[Kg/s]:" << -phi
167 << endl;
168 }
169 }
170 else
171 {
172 FatalErrorInFunction
173 << "dimensions of " << phiName_ << " are incorrect" << nl
174 << " on patch " << this->patch().name()
175 << " of field " << this->internalField().name()
176 << " in file " << this->internalField().objectPath()
177 << nl << exit(FatalError);
178 }
179
180 // Restore tag
181 UPstream::msgType() = oldTag;
182
183 fixedValueFvPatchField<vector>::updateCoeffs();
184}
