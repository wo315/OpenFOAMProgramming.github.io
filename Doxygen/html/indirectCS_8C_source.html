97void Foam::coordSystem::indirect::writeEntry
98(
99 const word& keyword,
100 Ostream& os
101) const
102{
103 if (!valid())
104 {
105 return;
106 }
107
108 const bool subDict = !keyword.empty();
109
110 if (subDict)
111 {
112 os.beginBlock(keyword);
113
114 os.writeEntry("type", type());
115 os.writeEntry("name", name());
116
117 os.endBlock();
118 }
119}
