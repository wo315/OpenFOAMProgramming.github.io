109void Foam::freestreamPressureFvPatchScalarField::updateCoeffs()
110{
111 if (updated())
112 {
113 return;
114 }
115
116 const Field<vector>& Up =
117 patch().template lookupPatchField<volVectorField>(UName_);
118
119 valueFraction() = 0.5 + 0.5*(Up & patch().nf())/mag(Up);
120
121 mixedFvPatchField<scalar>::updateCoeffs();
122}

125void Foam::freestreamPressureFvPatchScalarField::write(Ostream& os) const
126{
127 fvPatchField<scalar>::write(os);
128 os.writeEntryIfDifferent<word>("U", "U", UName_);
129 freestreamValue().writeEntry("freestreamValue", os);
130 fvPatchField<scalar>::writeValueEntry(os);
131}
