32void Foam::SIBS::polyExtrapolate
33(
34 const label iest,
35 const scalar xest,
36 const scalarField& yest,
37 scalarField& yz,
38 scalarField& dy,
39 scalarField& x,
40 scalarRectangularMatrix& d
41) const
42{
43 label n = yz.size();
44
45 x[iest] = xest;
46
47 for (label j=0; j<n; j++)
48 {
49 dy[j] = yz[j] = yest[j];
50 }
51
52 if (iest == 0)
53 {
54 for (label j=0; j<n; j++)
55 {
56 d(j, 0) = yest[j];
57 }
58 }
59 else
60 {
61 scalarField c(yest);
62
63 for (label k1=0; k1<iest; k1++)
64 {
65 scalar delta = 1.0/(x[iest - k1 - 1] - xest);
66 scalar f1 = xest*delta;
67 scalar f2 = x[iest - k1 - 1]*delta;
68
69 for (label j=0; j<n; j++)
70 {
71 scalar q = d[j][k1];
72 d[j][k1] = dy[j];
73 delta = c[j] - q;
74 dy[j] = f1*delta;
75 c[j] = f2*delta;
76 yz[j] += dy[j];
77 }
78 }
79
80 for (label j=0; j<n; j++)
81 {
82 d[j][iest] = dy[j];
83 }
84 }
85}
