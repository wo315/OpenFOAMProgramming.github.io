60std::string Foam::clock::dateTime()

64
65 std::ostringstream os;
66 os
67 << std::setfill('0')
68 << std::setw(4) << curr->tm_year + 1900
69 << '-' << std::setw(2) << curr->tm_mon + 1
70 << '-' << std::setw(2) << curr->tm_mday

73 << ':' << std::setw(2) << curr->tm_min
