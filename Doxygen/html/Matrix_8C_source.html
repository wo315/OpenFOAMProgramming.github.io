34template<class Form, class Type>
35template<class ListType>
36Foam::tmp<Foam::Field<Type>> Foam::Matrix<Form, Type>::AmulImpl
37(
38 const ListType& x
39) const
40{
41 const Matrix<Form, Type>& mat = *this;
42
43 #ifdef FULLDEBUG
44 if (mat.n() != x.size())
45 {
46 FatalErrorInFunction
47 << "Attempt to multiply incompatible Matrix and Vector:" << nl
48 << "Matrix : (" << mat.m() << ", " << mat.n() << ')' << nl
49 << "Matrix columns != Vector size (" << x.size() << ')' << nl
50 << abort(FatalError);
51 }
52 #endif
53
54 auto tresult = tmp<Field<Type>>::New(mat.m(), Zero);
55 auto& result = tresult.ref();
56
57 for (label i = 0; i < mat.m(); ++i)
58 {
59 for (label j = 0; j < mat.n(); ++j)
60 {
61 result[i] += mat(i, j)*x[j];
62 }
63 }
64
65 return tresult;
66}

69template<class Form, class Type>
70template<class ListType>
71Foam::tmp<Foam::Field<Type>> Foam::Matrix<Form, Type>::TmulImpl
72(
73 const ListType& x
74) const
75{
76 const Matrix<Form, Type>& mat = *this;
77
78 #ifdef FULLDEBUG
79 if (mat.m() != x.size())
80 {
81 FatalErrorInFunction
82 << "Attempt to multiply incompatible Matrix and Vector:" << nl
83 << "Matrix : (" << mat.m() << ", " << mat.n() << ')' << nl
84 << "Matrix rows != Vector size (" << x.size() << ')' << nl
85 << abort(FatalError);
86 }
87 #endif
88
89 auto tresult = tmp<Field<Type>>::New(mat.n(), Zero);
90 auto& result = tresult.ref();
91
92 for (label i = 0; i < mat.m(); ++i)
93 {
94 const Type& val = x[i];
95 for (label j = 0; j < mat.n(); ++j)
96 {
97 result[j] += val*mat(i, j);
98 }
99 }
100
101 return tresult;
102}

324template<class Form, class Type>
325void Foam::Matrix<Form, Type>::resize(const label m, const label n)
326{
327 if (m == mRows_ && n == nCols_)
328 {
329 return;
330 }
331
332 Matrix<Form, Type> newMatrix(m, n, Zero);
333
334 const label mrow = min(m, mRows_);
335 const label ncol = min(n, nCols_);
336
337 for (label i = 0; i < mrow; ++i)
338 {
339 for (label j = 0; j < ncol; ++j)
340 {
341 newMatrix(i, j) = (*this)(i, j);
342 }
343 }
344
345 transfer(newMatrix);
346}

349template<class Form, class Type>
350void Foam::Matrix<Form, Type>::resize_nocopy(const label mrow, const label ncol)
351{
352 if (mrow == mRows_ && ncol == nCols_)
353 {
354 return;
355 }
356
357 const label oldLen = (mRows_ * nCols_);
358
359 const label newLen = (mrow * ncol);
360
361 if (oldLen == newLen)
362 {
363 // Shallow resize is enough
364 mRows_ = mrow;
365 nCols_ = ncol;
366 }
367 else
368 {
369 this->clear();
370
371 mRows_ = mrow;
372 nCols_ = ncol;
373
374 this->doAlloc();
375 }
376}
