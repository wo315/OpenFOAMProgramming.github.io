86bool Foam::SIMPLEControl::read()
87{
88 simpleControl::read();
89 solverControl::read();
90 readIters();
91
92 if (average_ && averageStartIter_ > nIters_)
93 {
94 WarningInFunction
95 << "Average start iteration is larger than nIter in solver "
96 << solver_.solverName() << nl
97 << tab << "Disabling averaging ..." << nl
98 << endl;
99 average_ = false;
100 }
101
102 return true;
103}
