84{

202Foam::functionObjects::energyTransport::energyTransport
203(
204 const word& name,
205 const Time& runTime,
206 const dictionary& dict
207)
208:
209 fvMeshFunctionObject(name, runTime, dict),
210 fieldName_(dict.getOrDefault<word>("field", "T")),
211 phiName_(dict.getOrDefault<word>("phi", "phi")),
212 rhoName_(dict.getOrDefault<word>("rho", "rho")),
213 nCorr_(0),
214 schemesField_("unknown-schemesField"),
215 fvOptions_(mesh_),
216 multiphaseThermo_(dict.subOrEmptyDict("phaseThermos")),
217 Cp_("Cp", dimEnergy/dimMass/dimTemperature, 0, dict),
218 kappa_
219 (
220 "kappa",
221 dimEnergy/dimTime/dimLength/dimTemperature,
222 0,
223 dict
224 ),
225 rho_("rhoInf", dimDensity, 0, dict),
226 Prt_("Prt", dimless, 1, dict),
227 rhoCp_
228 (
229 IOobject
230 (
231 "rhoCp",
232 mesh_.time().timeName(),
233 mesh_,
234 IOobject::NO_READ,
235 IOobject::NO_WRITE,
236 IOobject::NO_REGISTER
237 ),
238 mesh_,
239 dimensionedScalar(dimEnergy/dimTemperature/dimVolume, Zero)
240 )
241{
242 read(dict);
243
244 // If the flow is multiphase
245 if (!multiphaseThermo_.empty())
246 {
247 Cps_.setSize(multiphaseThermo_.size());
248 kappas_.setSize(Cps_.size());
249 phaseNames_.setSize(Cps_.size());
250
251 label phasei = 0;
252 forAllConstIters(multiphaseThermo_, iter)
253 {
254 const word& key = iter().keyword();
255
256 const dictionary* subDictPtr = multiphaseThermo_.findDict(key);
257
258 if (!subDictPtr)
259 {
260 FatalErrorInFunction
261 << "Found non-dictionary entry " << iter()
262 << " in top-level dictionary " << multiphaseThermo_
263 << exit(FatalError);
264 }
265
266 const dictionary& dict = *subDictPtr;
267
268 phaseNames_[phasei] = key;
269
270 Cps_.set
271 (
272 phasei,
273 new dimensionedScalar
274 (
275 "Cp",
276 dimEnergy/dimMass/dimTemperature,
277 dict
278 )
279 );
280
281 kappas_.set
282 (
283 phasei,
284 new dimensionedScalar //[J/m/s/K]
285 (
286 "kappa",
287 dimEnergy/dimTime/dimLength/dimTemperature,
288 dict
289 )
290 );
291
292 ++phasei;
293 }
294
295 phases_.setSize(phaseNames_.size());
296 forAll(phaseNames_, i)
297 {
298 phases_.set
299 (
300 i,
301 mesh_.getObjectPtr<volScalarField>(phaseNames_[i])
302 );
303 }
304
305 rhoCp_ = rho()*Cp();
306 rhoCp_.oldTime();
307 }
308 else
309 {
310 if (Cp_.value() == 0.0 || kappa_.value() == 0.0)
311 {
312 FatalErrorInFunction
313 << " Multiphase thermo dictionary not found and Cp/kappa "
314 << " for single phase are zero. Please entry either"
315 << exit(FatalError);
316 }
317
318 }
319
320 // Force creation of transported field so any BCs using it can
321 // look it up
322 volScalarField& s = transportedField();
323 s.correctBoundaryConditions();
324}

335bool Foam::functionObjects::energyTransport::read(const dictionary& dict)
336{
337 fvMeshFunctionObject::read(dict);
338
339 dict.readIfPresent("phi", phiName_);
340 dict.readIfPresent("rho", rhoName_);
341
342 schemesField_ = dict.getOrDefault("schemesField", fieldName_);
343
344 dict.readIfPresent("nCorr", nCorr_);
345
346 if (dict.found("fvOptions"))
347 {
348 fvOptions_.reset(dict.subDict("fvOptions"));
349 }
350
351 return true;
352}

355bool Foam::functionObjects::energyTransport::execute()
356{
357 volScalarField& s = transportedField();
358
359 Log << type() << " execute: " << s.name() << endl;
360
361 const surfaceScalarField& phi =
362 mesh_.lookupObject<surfaceScalarField>(phiName_);
363
364 // Calculate the diffusivity
365 const volScalarField kappaEff("kappaEff", this->kappaEff());
366
367 word divScheme("div(phi," + schemesField_ + ")");
368 word laplacianScheme
369 (
370 "laplacian(kappaEff," + schemesField_ + ")"
371 );
372
373 // Set under-relaxation coeff
374 scalar relaxCoeff = 0.0;
375 if (mesh_.relaxEquation(schemesField_))
376 {
377 relaxCoeff = mesh_.equationRelaxationFactor(schemesField_);
378 }
379
380 if (phi.dimensions() == dimMass/dimTime)
381 {
382 rhoCp_ = rho()*Cp();
383 const surfaceScalarField rhoCpPhi(fvc::interpolate(Cp())*phi);
384
385 for (label i = 0; i <= nCorr_; i++)
386 {
387 fvScalarMatrix sEqn
388 (
389 fvm::ddt(rhoCp_, s)
390 + fvm::div(rhoCpPhi, s, divScheme)
391 - fvm::Sp(fvc::ddt(rhoCp_) + fvc::div(rhoCpPhi), s)
392 - fvm::laplacian(kappaEff, s, laplacianScheme)
393 ==
394 fvOptions_(rhoCp_, s)
395 );
396
397 sEqn.relax(relaxCoeff);
398
399 fvOptions_.constrain(sEqn);
400
401 sEqn.solve(mesh_.solverDict(schemesField_));
402 }
403 }
404 else if (phi.dimensions() == dimVolume/dimTime)
405 {
406 dimensionedScalar rhoCp(rho_*Cp_);
407
408 const surfaceScalarField CpPhi(rhoCp*phi);
409
410 auto trhoCp = tmp<volScalarField>::New
411 (
412 IOobject
413 (
414 "trhoCp",
415 mesh_.time().timeName(),
416 mesh_,
417 IOobject::NO_READ,
418 IOobject::NO_WRITE
419 ),
420 mesh_,
421 rhoCp
422 );
423
424 for (label i = 0; i <= nCorr_; i++)
425 {
426 fvScalarMatrix sEqn
427 (
428 fvm::ddt(rhoCp, s)
429 + fvm::div(CpPhi, s, divScheme)
430 - fvm::laplacian(kappaEff, s, laplacianScheme)
431 ==
432 fvOptions_(trhoCp.ref(), s)
433 );
434
435 sEqn.relax(relaxCoeff);
436
437 fvOptions_.constrain(sEqn);
438
439 sEqn.solve(mesh_.solverDict(schemesField_));
440 }
441 }
442 else
443 {
444 FatalErrorInFunction
445 << "Incompatible dimensions for phi: " << phi.dimensions() << nl
446 << "Dimensions should be " << dimMass/dimTime << " or "
447 << dimVolume/dimTime << exit(FatalError);
448 }
449
450 Log << endl;
451
452 return true;
453}
