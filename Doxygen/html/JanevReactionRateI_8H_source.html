60inline Foam::scalar Foam::JanevReactionRate::operator()
61(
62 const scalar p,
63 const scalar T,
64 const scalarField&
65) const
66{
67 scalar lta = A_;
68
69 if (mag(beta_) > VSMALL)
70 {
71 lta *= pow(T, beta_);
72 }
73
74 scalar expArg = 0.0;
75
76 if (mag(Ta_) > VSMALL)
77 {
78 expArg -= Ta_/T;
79 }
80
81 scalar lnT = log(T);
82
83 for (int n=0; n<nb_; n++)
84 {
85 expArg += b_[n]*pow(lnT, n);
86 }
87
88 lta *= exp(expArg);
89
90 return lta;
91}
