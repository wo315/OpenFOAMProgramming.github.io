61Foam::tmp<Foam::FieldField<Foam::Field, Foam::scalar>>
62Foam::heatTransferCoeffModel::q() const
63{
64 const auto& T = mesh_.lookupObject<volScalarField>(TName_);
65 const volScalarField::Boundary& Tbf = T.boundaryField();
66
67 auto tq = tmp<FieldField<Field, scalar>>::New(Tbf.size());
68 auto& q = tq.ref();
69
70 forAll(q, patchi)
71 {
72 q.set(patchi, new Field<scalar>(Tbf[patchi].size(), Zero));
73 }
74
75 typedef compressible::turbulenceModel cmpTurbModel;
76
77 if (mesh_.foundObject<cmpTurbModel>(cmpTurbModel::propertiesName))
78 {
79 const auto& turb =
80 mesh_.lookupObject<cmpTurbModel>(cmpTurbModel::propertiesName);
81
82 // Note: calling he(p,T) instead of he()
83 const volScalarField he(turb.transport().he(turb.transport().p(), T));
84 const volScalarField::Boundary& hebf = he.boundaryField();
85
86 const volScalarField alphaEff(turb.alphaEff());
87 const volScalarField::Boundary& alphaEffbf = alphaEff.boundaryField();
88
89 for (const label patchi : patchSet_)
90 {
91 q[patchi] = alphaEffbf[patchi]*hebf[patchi].snGrad();
92 }
93 }
94 else if (mesh_.foundObject<fluidThermo>(fluidThermo::dictName))
95 {
96 const auto& thermo =
97 mesh_.lookupObject<fluidThermo>(fluidThermo::dictName);
98
99 // Note: calling he(p,T) instead of he()
100 const volScalarField he(thermo.he(thermo.p(), T));
101 const volScalarField::Boundary& hebf = he.boundaryField();
102
103 const volScalarField& alpha(thermo.alpha());
104 const volScalarField::Boundary& alphabf = alpha.boundaryField();
105
106 for (const label patchi : patchSet_)
107 {
108 q[patchi] = alphabf[patchi]*hebf[patchi].snGrad();
109 }
110 }
111 else
112 {
113 FatalErrorInFunction
114 << "Unable to find a valid thermo model to evaluate q. " << nl
115 << "Database contents are: " << mesh_.objectRegistry::sortedToc()
116 << exit(FatalError);
117 }
118
119 // Add radiative heat flux contribution if present
120
121 const auto* qrPtr = mesh_.cfindObject<volScalarField>(qrName_);
122
123 if (qrPtr)
124 {
125 const volScalarField::Boundary& qrbf = qrPtr->boundaryField();
126
127 for (const label patchi : patchSet_)
128 {
129 q[patchi] += qrbf[patchi];
130 }
131 }
132
133 return tq;
134}
