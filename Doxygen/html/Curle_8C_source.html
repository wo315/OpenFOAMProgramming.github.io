79bool Foam::functionObjects::Curle::read(const dictionary& dict)
80{
81 if (!(fvMeshFunctionObject::read(dict) && writeFile::read(dict)))
82 {
83 return false;
84 }
85
86 dict.readIfPresent("p", pName_);
87
88 patchSet_ = mesh_.boundaryMesh().patchSet(dict.get<wordRes>("patches"));
89
90 if (patchSet_.empty())
91 {
92 WarningInFunction
93 << "No patches defined"
94 << endl;
95
96 return false;
97 }
98
99 const modeType inputMode = modeTypeNames_.get("input", dict);
100
101 switch (inputMode)
102 {
103 case modeType::point:
104 {
105 observerPositions_ = dict.get<List<point>>("observerPositions");
106 break;
107 }
108 case modeType::surface:
109 {
110 const fileName fName = (dict.get<fileName>("surface")).expand();
111 inputSurface_ = MeshedSurface<face>(fName);
112
113 observerPositions_ = inputSurface_.Cf();
114 }
115 }
116
117 if (observerPositions_.empty())
118 {
119 WarningInFunction
120 << "No observer positions defined"
121 << endl;
122
123 return false;
124 }
125
126 const auto outputMode = modeTypeNames_.get("output", dict);
127
128 switch (outputMode)
129 {
130 case modeType::point:
131 {
132 rawFilePtrs_.setSize(observerPositions_.size());
133 forAll(rawFilePtrs_, filei)
134 {
135 rawFilePtrs_.set
136 (
137 filei,
138 newFileAtStartTime("observer" + Foam::name(filei))
139 );
140
141 if (rawFilePtrs_.set(filei))
142 {
143 OFstream& os = rawFilePtrs_[filei];
144
145 writeHeaderValue(os, "Observer", filei);
146 writeHeaderValue(os, "Position", observerPositions_[filei]);
147 writeCommented(os, "Time");
148 writeTabbed(os, "p(Curle)");
149 os << endl;
150 }
151 }
152 break;
153 }
154 case modeType::surface:
155 {
156 if (inputMode != modeType::surface)
157 {
158 FatalIOErrorInFunction(dict)
159 << "Surface output is only available when input mode is "
160 << "type '" << modeTypeNames_[modeType::surface] << "'"
161 << abort(FatalIOError);
162 }
163
164 const word writerType = dict.get<word>("surfaceType");
165
166 surfaceWriterPtr_ = surfaceWriter::New
167 (
168 writerType,
169 surfaceWriter::formatOptions(dict, writerType)
170 );
171
172 // Use outputDir/TIME/surface-name
173 surfaceWriterPtr_->useTimeDir(true);
174 }
175 }
176
177 // Read the reference speed of sound
178 dict.readEntry("c0", c0_);
179
180 if (c0_ < VSMALL)
181 {
182 FatalErrorInFunction
183 << "Reference speed of sound = " << c0_
184 << " cannot be negative or zero."
185 << abort(FatalError);
186 }
187
188 return true;
189}

192bool Foam::functionObjects::Curle::execute()
193{
194 if (!foundObject<volScalarField>(pName_))
195 {
196 return false;
197 }
198
199 const volScalarField& p = lookupObject<volScalarField>(pName_);
200 const volScalarField::Boundary& pBf = p.boundaryField();
201 const volScalarField dpdt(scopedName("dpdt"), fvc::ddt(p));
202 const volScalarField::Boundary& dpdtBf = dpdt.boundaryField();
203 const surfaceVectorField::Boundary& SfBf = mesh_.Sf().boundaryField();
204 const surfaceVectorField::Boundary& CfBf = mesh_.Cf().boundaryField();
205
206 scalarField pDash(observerPositions_.size(), 0);
207
208 for (auto patchi : patchSet_)
209 {
210 const scalarField& pp = pBf[patchi];
211 const scalarField& dpdtp = dpdtBf[patchi];
212 const vectorField& Sfp = SfBf[patchi];
213 const vectorField& Cfp = CfBf[patchi];
214
215 forAll(observerPositions_, pointi)
216 {
217 const vectorField r(observerPositions_[pointi] - Cfp);
218 const scalarField invMagR(1/(mag(r) + ROOTVSMALL));
219
220 pDash[pointi] +=
221 sum((pp*sqr(invMagR) + invMagR/c0_*dpdtp)*(Sfp & (r*invMagR)));
222 }
223 }
224
225 pDash /= 4*mathematical::pi;
226
227 Pstream::listCombineReduce(pDash, plusEqOp<scalar>());
228
229 if (surfaceWriterPtr_)
230 {
231 if (Pstream::master())
232 {
233 // Time-aware, with time spliced into the output path
234 surfaceWriterPtr_->beginTime(time_);
235
236 surfaceWriterPtr_->open
237 (
238 inputSurface_.points(),
239 inputSurface_.surfFaces(),
240 (baseFileDir()/name()/"surface"),
241 false // serial - already merged
242 );
243
244 surfaceWriterPtr_->write("p(Curle)", pDash);
245
246 surfaceWriterPtr_->endTime();
247 surfaceWriterPtr_->clear();
248 }
249 }
250 else
251 {
252 forAll(observerPositions_, pointi)
253 {
254 if (rawFilePtrs_.set(pointi))
255 {
256 OFstream& os = rawFilePtrs_[pointi];
257 writeCurrentTime(os);
258 os << pDash[pointi] << endl;
259 }
260 }
261 }
262
263 return true;
264}
