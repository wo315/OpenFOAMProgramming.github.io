107 Info<< "Selecting LES " << lookupName << " type " << deltaType << endl;

109 // First any additional ones

112
113 if (ctorIter.good())

116 }
117 }
118

120
121 if (!ctorPtr)

123 FatalIOErrorInLookup

126 "LESdelta",
127 deltaType,
128 *dictionaryConstructorTablePtr_
129 );
130
131 if (!additionalConstructors.empty())

133 FatalIOError
