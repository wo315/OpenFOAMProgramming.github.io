552//- Test for equality of begin/size values
553template<class IntType>
554inline constexpr bool operator==
555(
556 const IntRange<IntType>& a,
557 const IntRange<IntType>& b
558) noexcept
559{
560 return (a.start() == b.start() && a.size() == b.size());
561}
562
563
564//- Comparison function for sorting, compares the start.
565// If the start values are equal, also compares the size.
566template<class IntType>
567inline constexpr bool operator<
568(
569 const IntRange<IntType>& a,
570 const IntRange<IntType>& b
571) noexcept
572{
573 return
574 (
575 a.start() < b.start()
576 ||
577 (
578 !(b.start() < a.start())
579 && a.size() < b.size()
580 )
581 );
582}
