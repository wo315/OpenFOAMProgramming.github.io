33Foam::labelHashSet Foam::HashSetOps::used(const bitSet& select)
34{
35 labelHashSet output(0);
36
37 if (select.any())
38 {
39 output.resize(2*select.count());
40
41 for (label i = select.find_first(); i >= 0; i = select.find_next(i))
42 {
43 output.insert(i);
44 }
45 }
46
47 return output;
48}

51Foam::labelHashSet Foam::HashSetOps::used(const UList<bool>& select)
52{
53 const label len = select.size();

79Foam::List<bool> Foam::HashSetOps::bools(const labelHashSet& locations)
80{
81 auto const max = std::max_element(locations.begin(), locations.end());
82 const label len = (max != locations.end() ? (1 + *max) : 0);
83
84 if (len <= 0)
85 {
86 return List<bool>();
87 }
88
89 List<bool> output(len, false);
90
91 for (const label i : locations)
92 {
93 if (i >= 0)
94 {
95 output[i] = true;
96 }
97 }
98
99 return output;
100}
