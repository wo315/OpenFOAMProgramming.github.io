144template<class Type>
145Foam::tmp<Foam::Field<Type>>
146Foam::partialSlipFvPatchField<Type>::snGrad() const
147{
148 tmp<vectorField> nHat = this->patch().nf();
149 const Field<Type> pif(this->patchInternalField());
150
151 return
152 (
153 lerp
154 (
155 transform(I - sqr(nHat), pif),
156 refValue_,
157 valueFraction_
158 ) - pif
159 )*this->patch().deltaCoeffs();
160}

163template<class Type>
164void Foam::partialSlipFvPatchField<Type>::evaluate
165(
166 const Pstream::commsTypes
167)
168{
169 if (!this->updated())
170 {
171 this->updateCoeffs();
172 }
173
174 tmp<vectorField> nHat = this->patch().nf();
175
176 Field<Type>::operator=
177 (
178 lerp
179 (
180 transform(I - sqr(nHat), this->patchInternalField()),
181 refValue_,
182 valueFraction_
183 )
184 );
185
186 parent_bctype::evaluate();
187}

190template<class Type>
191Foam::tmp<Foam::Field<Type>>
192Foam::partialSlipFvPatchField<Type>::snGradTransformDiag() const
193{
194 tmp<vectorField> diag(cmptMag(this->patch().nf()));
195
196 return
197 valueFraction_*pTraits<Type>::one
198 + (1.0 - valueFraction_)
199 *transformFieldMask<Type>(pow<vector, pTraits<Type>::rank>(diag));
200}
