64template<class ReactionRate, class FallOffFunction>
65inline Foam::scalar
66Foam::FallOffReactionRate<ReactionRate, FallOffFunction>::operator()
67(
68 const scalar p,
69 const scalar T,
70 const scalarField& c
71) const
72{
73 const scalar k0 = k0_(p, T, c);
74 const scalar kInf = kInf_(p, T, c);
75
76 const scalar Pr = k0*thirdBodyEfficiencies_.M(c)/kInf;
77
78 return kInf*(Pr/(1 + Pr))*F_(T, Pr);
79}
