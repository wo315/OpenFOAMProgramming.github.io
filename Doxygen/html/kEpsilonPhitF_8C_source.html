41template<class BasicTurbulenceModel>
42void kEpsilonPhitF<BasicTurbulenceModel>::correctNut()
43{
44 // (LUU:p. 173)
45 this->nut_ = Cmu_*phit_*k_*T_;
46 this->nut_.correctBoundaryConditions();
47 fv::options::New(this->mesh_).correct(this->nut_);
48
49 BasicTurbulenceModel::correctNut();
50}

349{
350 if (eddyViscosity<RASModel<BasicTurbulenceModel>>::read())

356 Ceps1c_.readIfPresent(this->coeffDict());
357 Ceps2_.readIfPresent(this->coeffDict());
358 Cf1_.readIfPresent(this->coeffDict());
359 Cf2_.readIfPresent(this->coeffDict());
360 CL_.readIfPresent(this->coeffDict());
361 Ceta_.readIfPresent(this->coeffDict());
362 CT_.readIfPresent(this->coeffDict());
363 sigmaK_.readIfPresent(this->coeffDict());
364 sigmaEps_.readIfPresent(this->coeffDict());
365 sigmaPhit_.readIfPresent(this->coeffDict());
366
367 return true;
368 }
369
370 return false;
371}

374template<class BasicTurbulenceModel>
375void kEpsilonPhitF<BasicTurbulenceModel>::correct()
376{
377 if (!this->turbulence_)
378 {
379 return;
380 }
381
382 // Construct local convenience references
383 const alphaField& alpha = this->alpha_;
384 const rhoField& rho = this->rho_;
385 const surfaceScalarField& alphaRhoPhi = this->alphaRhoPhi_;
386 const volVectorField& U = this->U_;
387 volScalarField& nut = this->nut_;
388 fv::options& fvOptions(fv::options::New(this->mesh_));
389
390 eddyViscosity<RASModel<BasicTurbulenceModel>>::correct();
391
392 const volScalarField::Internal divU
393 (
394 fvc::div(fvc::absolute(this->phi(), U))
395 );
396
397 tmp<volSymmTensorField> tS(symm(fvc::grad(U)));
398 volScalarField G(this->GName(), nut*(2.0*(dev(tS()) && tS())));
399 tS.clear();
400
401 T_ = Ts();
402 bound(T_, TMin_);
403
404 const volScalarField L2(type() + "L2", sqr(Ls()) + L2Min_);
405
406 const volScalarField::Internal Ceps1
407 (
408 "Ceps1",
409 Ceps1a_*(Ceps1b_ + Ceps1c_*sqrt(1.0/phit_()))
410 );
411
412
413 // Update epsilon and G at the wall
414 epsilon_.boundaryFieldRef().updateCoeffs();
415 // Push any changed cell values to coupled neighbours
416 epsilon_.boundaryFieldRef().template evaluateCoupled<coupledFvPatch>();
417
418 // Turbulent kinetic energy dissipation rate equation (LUU:Eq. 4)
419 // k/T ~ epsilon
420 tmp<fvScalarMatrix> epsEqn
421 (
422 fvm::ddt(alpha, rho, epsilon_)
423 + fvm::div(alphaRhoPhi, epsilon_)
424 - fvm::laplacian(alpha*rho*DepsilonEff(), epsilon_)
425 ==
426 alpha()*rho()*Ceps1*G()/T_()
427 - fvm::SuSp
428 (
429 (2.0/3.0*Ceps1)*(alpha()*rho()*divU),
430 epsilon_
431 )
432 - fvm::Sp(alpha()*rho()*Ceps2_/T_(), epsilon_)
433 + fvOptions(alpha, rho, epsilon_)
434 );
435
436 epsEqn.ref().relax();
437 fvOptions.constrain(epsEqn.ref());
438 epsEqn.ref().boundaryManipulate(epsilon_.boundaryFieldRef());
439 solve(epsEqn);
440 fvOptions.correct(epsilon_);
441 bound(epsilon_, this->epsilonMin_);
442
443
444 // Turbulent kinetic energy equation (LUU:Eq. 3)
445 // epsilon/k ~ 1/Ts
446 tmp<fvScalarMatrix> kEqn
447 (
448 fvm::ddt(alpha, rho, k_)
449 + fvm::div(alphaRhoPhi, k_)
450 - fvm::laplacian(alpha*rho*DkEff(), k_)
451 ==
452 alpha()*rho()*G()
453 - fvm::SuSp(2.0/3.0*alpha()*rho()*divU, k_)
454 - fvm::Sp(alpha()*rho()*(1.0/T_()), k_)
455 + fvOptions(alpha, rho, k_)
456 );
457
458 kEqn.ref().relax();
459 fvOptions.constrain(kEqn.ref());
460 solve(kEqn);
461 fvOptions.correct(k_);
462 bound(k_, this->kMin_);
463
464
465 // Elliptic relaxation function equation (LUU:Eq. 18)
466 // All source terms are non-negative functions (LUU:p. 176)
467 tmp<fvScalarMatrix> fEqn
468 (
469 - fvm::laplacian(f_)
470 ==
471 - fvm::Sp(1.0/L2(), f_)
472 - (
473 (Cf1_ - 1.0)*(phit_() - 2.0/3.0)/T_()
474 -(Cf2_*G())/k_()
475 +(Cf2_*(2.0/3.0)*divU)
476 -(2.0*this->nu()*(fvc::grad(phit_) & fvc::grad(k_)))()/k_()
477 -(this->nu()*fvc::laplacian(phit_))()
478 )/L2()
479 );
480
481 fEqn.ref().relax();
482 solve(fEqn);
483 bound(f_, fMin_);
484
485
486 // Normalised wall-normal fluctuating velocity scale equation (LUU:Eq. 17)
487 // All source terms are non-negative functions (LUU:p. 176)
488 tmp<fvScalarMatrix> phitEqn
489 (
490 fvm::ddt(alpha, rho, phit_)
491 + fvm::div(alphaRhoPhi, phit_)
492 - fvm::laplacian(alpha*rho*DphitEff(), phit_)
493 ==
494 alpha()*rho()*f_()
495 - fvm::SuSp
496 (
497 alpha()*rho()*
498 (
499 G()/k_()
500 - (2.0/3.0)*divU
501 - (2.0*nut*(fvc::grad(phit_) & fvc::grad(k_)))()
502 /(k_()*sigmaPhit_*phit_())
503 )
504 , phit_
505 )
506 + fvOptions(alpha, rho, phit_)
507 );
508
509 phitEqn.ref().relax();
510 fvOptions.constrain(phitEqn.ref());
511 solve(phitEqn);
512 fvOptions.correct(phit_);
513 bound(phit_, phitMin_);
514
515 correctNut();
516}
