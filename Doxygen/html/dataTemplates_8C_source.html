34template<class Type>
35void Foam::data::setSolverPerformance
36(
37 const word& name,
38 const SolverPerformance<Type>& sp
39) const
40{
41 dictionary& dict = const_cast<dictionary&>(solverPerformanceDict());
42
43 List<SolverPerformance<Type>> perfs;
44
45 if (prevTimeIndex_ != this->time().timeIndex())
46 {
47 // Reset solver performance between iterations
48 prevTimeIndex_ = this->time().timeIndex();
49 dict.clear();
50 }
51 else
52 {
53 dict.readIfPresent(name, perfs);
54 }
55
56 perfs.push_back(sp);
57
58 dict.set(name, perfs);
59}
