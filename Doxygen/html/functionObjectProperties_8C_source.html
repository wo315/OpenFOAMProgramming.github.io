232Foam::wordList Foam::functionObjects::properties::objectResultEntries
233(
234 const word& objectName
235) const
236{
237 DynamicList<word> result;
238
239 if (found(resultsName_))
240 {
241 const dictionary& resultsDict = subDict(resultsName_);
242
243 if (resultsDict.found(objectName))
244 {
245 const dictionary& objectDict = resultsDict.subDict(objectName);
246
247 for (const entry& dEntry : objectDict)
248 {
249 const dictionary& dict = dEntry.dict();
250
251 result.append(dict.toc());
252 }
253 }
254 }
255
256 wordList entries;
