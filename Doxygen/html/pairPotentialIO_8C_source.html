31bool Foam::pairPotential::writeEnergyAndForceTables(Ostream& os) const
32{
33 Info<< "Writing energy and force tables to file for potential "
34 << name_ << endl;
35
36 List<Pair <scalar>> eTab(energyTable());
37
38 List<Pair <scalar>> fTab(forceTable());
39
40 forAll(eTab, e)
41 {
42 os << eTab[e].first()
43 << token::SPACE
44 << eTab[e].second()
45 << token::SPACE
46 << fTab[e].second()
47 << nl;
48 }
49
50 return os.good();
51}
