69template<class CompType, class SolidThermo>
70inline Foam::tmp<Foam::DimensionedField<Foam::scalar, Foam::volMesh>>
71Foam::solidChemistryModel<CompType, SolidThermo>::RRs() const
72{
73 tmp<volScalarField::Internal> tRRs
74 (
75 new volScalarField::Internal
76 (
77 IOobject
78 (
79 "RRs",
80 this->time().timeName(),
81 this->mesh(),
82 IOobject::NO_READ,
83 IOobject::NO_WRITE
84 ),
85 this->mesh(),
86 dimensionedScalar(dimMass/dimVolume/dimTime, Zero)
87 )
88 );
89
90 if (this->chemistry_)
91 {
92 volScalarField::Internal& RRs = tRRs.ref();
93 for (label i=0; i < nSolids_; i++)
94 {
95 RRs += RRs_[i];
96 }
97 }
98 return tRRs;

102template<class CompType, class SolidThermo>
103inline Foam::tmp<Foam::DimensionedField<Foam::scalar, Foam::volMesh>>
104Foam::solidChemistryModel<CompType, SolidThermo>::RRsHs() const
105{
106 tmp<DimensionedField<scalar, volMesh>> tRRsHs
107 (
108 new DimensionedField<scalar, volMesh>
109 (
110 IOobject
111 (
112 "RRsHs",
113 this->time().timeName(),
114 this->mesh(),
115 IOobject::NO_READ,
116 IOobject::NO_WRITE
117 ),
118 this->mesh(),
119 dimensionedScalar(dimEnergy/dimVolume/dimTime, Zero)
120 )
121 );
122
123 if (this->chemistry_)
124 {
125 DimensionedField<scalar, volMesh>& RRs = tRRsHs.ref();
126
127 const volScalarField& T = this->solidThermo().T();
128 const volScalarField& p = this->solidThermo().p();
129
130 for (label i=0; i < nSolids_; i++)
131 {
132 forAll(RRs, cellI)
133 {
134 RRs[cellI] +=
135 RRs_[i][cellI]*solidThermo_[i].Hs(p[cellI], T[cellI]);
136 }
137 }
138 }
139 return tRRsHs;
140}
