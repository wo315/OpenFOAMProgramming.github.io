44void Foam::PatchInteractionFields<CloudType>::clearOrReset
45(
46 autoPtr<volScalarField>& fieldPtr,
47 const word& fieldName,
48 const dimensionSet& dims
49) const
50{
51 if (fieldPtr)
52 {
53 fieldPtr->primitiveFieldRef() = 0.0;

59 fieldPtr.reset

67 mesh,

69 IOobject::NO_WRITE
70 ),

75 }

172template<class CloudType>
173void Foam::PatchInteractionFields<CloudType>::postPatch
174(
175 const parcelType& p,
176 const polyPatch& pp,
177 bool&
178)
179{
