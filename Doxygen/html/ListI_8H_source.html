152template<class T>
153inline T& Foam::List<T>::newElmt(const label i)
154{
155 label n = this->size();
156
157 if (i >= n)
158 {
159 if (!n) n = 1; // Avoid dead-lock when starting from zero-sized
160
161 do
162 {
163 n *= 2;
164 }
165 while (i >= n);
166
167 resize(n);
168 }
169
170 return UList<T>::operator[](i);
171}

174template<class T>
175template<class... Args>
176inline T& Foam::List<T>::emplace_back(Args&&... args)
177{
178 // This could/should be better with inplace construction
179 // (as per std::vector), but currently lacking the methods for that
180 // so resize and move assign
181
182 const label idx = this->size();
183 resize(idx + 1);
184
185 // move assign element
186 this->operator[](idx) = T(std::forward<Args>(args)...);
187 return this->back();
188}

211template<class T>
212inline void Foam::List<T>::push_back(const UList<T>& list)
213{
214 if (this == &list)
215 {
216 FatalErrorInFunction
217 << "Attempted push_back to self" << abort(FatalError);
218 }
219
220 label idx = this->size();
221 const label n = list.size();
222
223 resize(idx + n);
224
225 for (label i=0; i<n; ++i)
226 {
227 this->operator[](idx++) = list[i]; // copy element
228 }
229}

265{
