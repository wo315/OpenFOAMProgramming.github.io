44Foam::profilingInformation* Foam::profiling::create()
45{
46 // Top-level entry: reset everything
47 pool_.clear();
48 children_.clear();
49 stack_.clear();
50 times_.clear();
51
52 Information* info = new Information;
53
54 pool_.push_back(info);
55 children_.resize(pool_.size());
56 children_.back().clear(); // safety
57
58 return info;
59}

62Foam::profilingInformation* Foam::profiling::create
63(
64 profilingInformation *parent,
65 const string& descr
66)
67{
68 const label parentId = parent->id();
69
70 for (Information* child : children_[parentId])
71 {
72 if (descr == child->description())
73 {
74 return child; // Found existing
75 }
76 }
77
78 Information* info = new Information(parent, descr, pool_.size());
79
80 pool_.push_back(info);
81 children_.resize(pool_.size());
82 children_.back().clear(); // safety
83 children_[parentId].push_back(info);
84
85 return info;
86}

89void Foam::profiling::beginTimer(profilingInformation *info)
90{
91 stack_.push_back(info);

93 info->setActive(true); // Mark as on stack

97Foam::profilingInformation* Foam::profiling::endTimer()
98{
99 Information *info = stack_.back();
100 clockValue clockval = times_.back();
101 stack_.pop_back();
102 times_.pop_back();
103
104 info->update(clockval.elapsed()); // Update elapsed time
105 info->setActive(false); // Mark as off stack
106
107 return info;
108}

183Foam::profilingInformation* Foam::profiling::New(const string& descr)
184{
185 Information *info = nullptr;
186
187 if (active())
188 {
189 Information *parent = singleton_->stack_.back();
190
191 info = singleton_->create(parent, descr);
192 singleton_->beginTimer(info);
193
194 if (singleton_->memInfo_)
195 {
196 info->maxMem_ = Foam::max
197 (
198 info->maxMem_,
199 singleton_->memInfo_->update().size()
200 );
201 }
202 }
203
204 return info;
205}

208void Foam::profiling::unstack(const profilingInformation *info)
209{
210 if (active() && info)
211 {
212 Information *top = singleton_->endTimer();
213
214 if (info->id() != top->id())
215 {
216 FatalErrorInFunction
217 << "Profiling information to unstack has different id than"
218 << " the top of the profiling stack" << nl
219 << " info: " << info->id() << " (" << info->description()
220 << ")\n"
221 << " top: " << top->id() << " (" << top->description()
222 << ")\n" << endl
223 << abort(FatalError);
224 }
225 }
226}

311bool Foam::profiling::writeData(Ostream& os) const
312{
313 static DynamicList<scalar> elapsed;
314
315 const clockValue now(clockValue::now());
316
317 const label nstack = stack_.size();
318
319 elapsed.resize(nstack+1); // extend for last entry, which has no child.
320
321 for (label stacki=0; stacki < nstack; ++stacki)
322 {
323 elapsed[stacki] = (now - times_[stacki]);
324 }
325 elapsed.back() = 0;
326
327 os.beginBlock("profiling");
328
329 // Active items
330 for (label stacki=0; stacki < nstack; ++stacki)
331 {
332 if (stacki) os << nl; // Extra line between entries
333
334 stack_[stacki]->write
335 (
336 os,
337 true,
338 elapsed[stacki], // elapsedTime
339 elapsed[stacki+1] // childTimes
340 );
341 }
342
343 // Non-active items
344 for (const Information& info : pool_)
345 {
346 if (!info.active())
347 {
348 os << nl;
349 info.write(os);
350 }
351 }
352
353 os.endBlock();
354
355 if (sysInfo_)
356 {
357 os << nl;
358 os.beginBlock("sysInfo");
359 sysInfo_->write(os);
360 os.endBlock();
361 }
362
363 if (cpuInfo_)
364 {
365 os << nl;
366 os.beginBlock("cpuInfo");
367 cpuInfo_->write(os);
368 os.endBlock();
369 }
370
371 if (memInfo_)
372 {
373 memInfo_->update();
374
375 os << nl;
376 os.beginBlock("memInfo");
377 memInfo_->write(os);
378 os.writeEntry("units", "kB");
379 os.endBlock();
380 }
381
382 return os.good();
383}
