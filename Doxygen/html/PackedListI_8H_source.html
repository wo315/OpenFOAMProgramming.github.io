82template<unsigned Width>
83inline void Foam::PackedList<Width>::clear_trailing_bits()
84{
85 // Mask off any partial rubbish in final block
86 const unsigned int blk = size() / elem_per_block;
87 const unsigned int off = size() % elem_per_block;
88
89 if (off)
90 {
91 blocks_[blk] &= mask_lower(off);
92 }
93}

96template<unsigned Width>
97inline bool Foam::PackedList<Width>::trim(label minpos)
98{
99 if (empty())
100 {
101 return false; // Trivial case
102 }
103
104 const label orig = size();
105 if (orig < minpos)
106 {
107 minpos = orig; // Don't allow allow accidental growth!
108 }
109
110 for (label blocki = num_blocks(size())-1; blocki >= 0; --blocki)
111 {
112 // Truncate to the block begin
113 size_ = blocki * elem_per_block;
114
115 unsigned int blockval = blocks_[blocki];
116
117 // Some bits were found in the block, increment size again
118 if (blockval)
119 {
120 for (; blockval; ++size_)
121 {
122 blockval >>= Width;
123 }
124 break;
125 }
126 else if (size_ < minpos)
127 {
128 break;
129 }
130 }
131
132 if (size_ < minpos)
133 {
134 size_ = minpos;
135 }
136
137 return (size() != orig);
138}

308template<unsigned Width>
309inline bool Foam::PackedList<Width>::reference::set(const unsigned int val)
310{
311 const unsigned int mask = (max_value << shift_);
312 const unsigned int prev = ref_;
313
314 if (val >= max_value)
315 {
316 ref_ |= mask; // Overflow is max_value, so fill entirely
317 }
318 else
319 {
320 ref_ &= ~mask;
321 ref_ |= mask & (val << shift_);
322 }
323
324 return (prev != ref_);
325}

393template<unsigned Width>
394inline void Foam::PackedList<Width>::resize
395(
396 const label newSize,
397 const unsigned int val
398)
399{
400 reserve(newSize);
401
402 const label oldSize = size();
403 size_ = newSize;
404
405 if (oldSize < newSize)
406 {
407 // Fill new elements or newly exposed elements
408 if (val)
409 {
410 // Fill value for complete blocks
411 const unsigned int blockval = repeated_value(val);
412
413 // Fill complete blocks
414 const label oldLen = num_blocks(oldSize);
415 const label newLen = num_blocks(size());
416 for (label blocki = oldLen; blocki < newLen; ++blocki)
417 {
418 blocks_[blocki] = blockval;
419 }
420
421 // Finish previous partial block, preserve existing value
422 {
423 const unsigned int blk = oldSize / elem_per_block;
424 const unsigned int off = oldSize % elem_per_block;
425 if (off)
426 {
427 const unsigned int mask = mask_lower(off);
428
429 blocks_[blk] &= mask;
430 blocks_[blk] |= ~mask & blockval;
431 }
432 }
433
434 clear_trailing_bits();
435 }
436 }
437 else if (newSize < oldSize)
438 {
439 // The list is now shorter than before, so we zero assign the unused
440 // blocks and any trailing junk. This costs slightly here, but make
441 // things much simpler elsewhere.
442
443 // Clear complete blocks
444 const label oldLen = num_blocks(oldSize);
445 const label newLen = num_blocks(size());
446 for (label blocki = newLen; blocki < oldLen; ++blocki)
447 {
448 blocks_[blocki] = 0u;
449 }
450
451 clear_trailing_bits();
452 }
453}

732template<unsigned Width>
733inline void Foam::PackedList<Width>::fill(const unsigned int val)
734{
735 if (empty())
736 {
737 return; // Trivial case
738 }
739
740 const label nblocks = num_blocks(size());
741
742 // Fill value for complete blocks
743 const unsigned int blockval = (val ? repeated_value(val) : 0u);
744
745 for (label blocki=0; blocki < nblocks; ++blocki)
746 {
747 blocks_[blocki] = blockval;
748 }
749
750 if (val)
751 {
752 clear_trailing_bits();
753 }
754}
