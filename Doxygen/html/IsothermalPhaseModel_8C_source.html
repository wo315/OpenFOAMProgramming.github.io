47template<class BasePhaseModel>
48void Foam::IsothermalPhaseModel<BasePhaseModel>::correctThermo()
49{
50 BasePhaseModel::correctThermo();
51
52 // Correct the thermo, but make sure that the temperature remains the same
53 tmp<volScalarField> TCopy
54 (
55 volScalarField::New
56 (
57 this->thermo().T().name() + ":Copy",
58 this->thermo().T()
59 )
60 );
61 this->thermo_->he() = this->thermo().he(this->thermo().p(), TCopy);
62 this->thermo_->correct();
63 this->thermo_->T() = TCopy;
64}
