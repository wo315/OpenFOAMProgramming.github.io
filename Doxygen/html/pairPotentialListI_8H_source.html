30inline Foam::label Foam::pairPotentialList::pairPotentialIndex
31(
32 const label a,
33 const label b
34) const
35{
36 label index;
37
38 if (a < b)
39 {
40 index = a*(2*nIds_ - a - 1)/2 + b;
41 }
42 else
43 {
44 index = b*(2*nIds_ - b - 1)/2 + a;
45 }
46
47 if (index > size() - 1)
48 {
49 FatalErrorInFunction
50 << "Attempting to access a pairPotential with too high an index."
51 << nl << "a = " << a << ", b = " << b << ", index = " << index
52 << nl << "max index = " << size() - 1
53 << nl << abort(FatalError);
54 }
55
56 return index;
57}
