68Foam::tmp<Foam::scalarField> Foam::compressible::
69turbulentTemperatureTwoPhaseRadCoupledMixedFvPatchScalarField::
70kappa
71(
72 const scalarField& Tp
73) const
74{
75 const polyMesh& mesh = patch().boundaryMesh().mesh();
76 const label patchi = patch().index();
77
78 switch (method_)
79 {
80 case mtSolidThermo:
81 {
82 const solidThermo& thermo =
83 mesh.lookupObject<solidThermo>(basicThermo::dictName);
84
85 return thermo.kappa(patchi);
86 break;
87 }
88
89 case mtLookup:
90 {
91 {
92 const auto* ptr =
93 mesh.cfindObject<volScalarField>(kappaName_);
94
95 if (ptr)
96 {
97 return patch().patchField(*ptr);
98 }
99 }
100 {
101 const auto* ptr =
102 mesh.cfindObject<volSymmTensorField>(kappaName_);
103
104 if (ptr)
105 {
106 const symmTensorField& KWall = patch().patchField(*ptr);
107
108 const vectorField n(patch().nf());
109
110 return n & KWall & n;
111 }
112 }
113
114 {
115 FatalErrorInFunction
116 << "Did not find field " << kappaName_
117 << " on mesh " << mesh.name() << " patch " << patch().name()
118 << nl
119 << " Please set 'kappa' to the name of a volScalarField"
120 << " or volSymmTensorField."
121 << exit(FatalError);
122 }
123 break;
124 }
125
126 case mtPhaseSystem:
127 {
128 // Lookup the fluid model
129 const phaseSystem& fluid =
130 (
131 mesh.lookupObject<phaseSystem>("phaseProperties")
132 );
133
134 auto tkappaEff = tmp<scalarField>::New(patch().size(), Zero);
135 auto& kappaEff = tkappaEff.ref();
136
137 forAll(fluid.phases(), phasei)
138 {
139 const phaseModel& phase = fluid.phases()[phasei];
140
141 const fvPatchScalarField& alpha = phase.boundaryField()[patchi];
142
143 kappaEff += alpha*phase.kappaEff(patchi)();
144 }
145
146 return tkappaEff;
147
148 break;
149 }
150
151 default:
152 {
153 FatalErrorInFunction
154 << "Unimplemented method " << KMethodTypeNames_[method_] << nl
155 << "Please set 'kappaMethod' to one of "
156 << flatOutput(KMethodTypeNames_.sortedToc()) << nl
157 << "and 'kappa' to the name of the volScalar"
158 << exit(FatalError);
159 }
160 }
161
162 // Return zero-sized (not nullptr)
163 return tmp<scalarField>::New();
164}

283void turbulentTemperatureTwoPhaseRadCoupledMixedFvPatchScalarField::
284updateCoeffs()
285{
286 if (updated())
287 {
288 return;
289 }
290
291 const polyMesh& mesh = patch().boundaryMesh().mesh();
292
293 // Since we're inside initEvaluate/evaluate there might be processor
294 // comms underway. Change the tag we use.
295 int oldTag = UPstream::msgType();
296 UPstream::msgType() = oldTag+1;
297
298 // Get the coupling information from the mappedPatchBase
299 const label patchi = patch().index();
300 const mappedPatchBase& mpp =
301 refCast<const mappedPatchBase>(patch().patch());
302 const polyMesh& nbrMesh = mpp.sampleMesh();
303 const label samplePatchi = mpp.samplePolyPatch().index();
304 const fvPatch& nbrPatch =
305 refCast<const fvMesh>(nbrMesh).boundary()[samplePatchi];
306
307 scalarField& Tp = *this;
308
309 const auto& nbrField =
310 refCast
311 <
312 const turbulentTemperatureTwoPhaseRadCoupledMixedFvPatchScalarField
313 >(nbrPatch.lookupPatchField<volScalarField>(TnbrName_));
314
315 // Swap to obtain full local values of neighbour internal field
316 scalarField TcNbr(nbrField.patchInternalField());
317 mpp.distribute(TcNbr);
318
319
320 // Swap to obtain full local values of neighbour K*delta
321 scalarField KDeltaNbr;
322 KDeltaNbr = nbrField.kappa(nbrField)*nbrPatch.deltaCoeffs();
323 mpp.distribute(KDeltaNbr);
324
325 scalarField KDelta(kappa(Tp)*patch().deltaCoeffs());
326
327 scalarField qr(Tp.size(), 0.0);
328 if (qrName_ != "none")
329 {
330 qr = patch().lookupPatchField<volScalarField>(qrName_);
331 }
332
333 scalarField qrNbr(Tp.size(), 0.0);
334 if (qrNbrName_ != "none")
335 {
336 qrNbr = nbrPatch.lookupPatchField<volScalarField>(qrNbrName_);
337 mpp.distribute(qrNbr);
338 }
339
340
341 if (regionType_ == solid)
342 {
343 // Lookup the fluid model in the nbrFvMesh
344 const phaseSystem& fluid =
345 (
346 nbrMesh.lookupObject<phaseSystem>("phaseProperties")
347 );
348
349 // The BC is applied to the liquid phase of the fluid
350 const phaseModel& liquid
351 (
352 fluid.phases()[nbrField.internalField().group()]
353 );
354
355 const phaseModel& vapor(fluid.phases()[otherPhaseName_]);
356
357
358 scalarField KDeltaLiqNbr;
359 const fvPatchScalarField& alphal = liquid.boundaryField()[samplePatchi];
360 KDeltaLiqNbr =
361 alphal*(liquid.kappaEff(samplePatchi))*nbrPatch.deltaCoeffs();
362 mpp.distribute(KDeltaLiqNbr);
363
364 scalarField KDeltaVapNbr;
365 const fvPatchScalarField& alphav = vapor.boundaryField()[samplePatchi];
366 KDeltaVapNbr =
367 alphav*(vapor.kappaEff(samplePatchi))*nbrPatch.deltaCoeffs();
368 mpp.distribute(KDeltaVapNbr);
369
370 scalarField TvNbr;
371 const fvPatchScalarField& Tv =
372 vapor.thermo().T().boundaryField()[samplePatchi];
373 TvNbr = Tv.patchInternalField();
374 mpp.distribute(TvNbr);
375
376 // TcNbr: liquid Tp
377 // TvNbr: vapour Tp
378 scalarField c(TcNbr*KDeltaLiqNbr + TvNbr*KDeltaVapNbr);
379
380 //valueFraction() = KDeltaNbr/(KDeltaNbr + KDelta);
381 //refValue() = c/KDeltaNbr;
382 scalarField KDeltaLiqVapNbr(KDeltaLiqNbr + KDeltaVapNbr);
383 valueFraction() = KDeltaLiqVapNbr/(KDeltaLiqVapNbr + KDelta);
384 refValue() = c/KDeltaLiqVapNbr;
385 refGrad() = (qr + qrNbr)/kappa(Tp);
386
387 if (debug)
388 {
389 scalar Q = gSum(kappa(Tp)*patch().magSf()*snGrad());
390
391 Info<< "T solid : " << nl << endl;
392
393 Info
394 << " heat transfer rate from solid:" << Q
395 << " walltemperature "
396 << " min:" << gMin(Tp)
397 << " max:" << gMax(Tp)
398 << " avg:" << gAverage(Tp) << nl
399 << endl;
400 }
401 }
402 else if (regionType_ == fluid)
403 {
404 const phaseSystem& fluid =
405 (
406 mesh.lookupObject<phaseSystem>("phaseProperties")
407 );
408
409 const phaseModel& liquid
410 (
411 fluid.phases()[internalField().group()]
412 );
413
414 const phaseModel& vapor(fluid.phases()[otherPhaseName_]);
415
416 const fvPatchScalarField& Tv =
417 vapor.thermo().T().boundaryField()[patchi];
418
419 const fvPatchScalarField& alphav = vapor.boundaryField()[patchi];
420
421 const scalarField KdeltaVap
422 (
423 alphav*(vapor.kappaEff(patchi))*patch().deltaCoeffs()
424 );
425
426 const fvPatchScalarField& alphal = liquid.boundaryField()[patchi];
427
428 const scalarField KdeltaLiq
429 (
430 alphal*(liquid.kappaEff(patchi))*patch().deltaCoeffs()
431 );
432
433 // TcNbr: solid Tp
434 // Tv: vapour Tp
435 const scalarField c(TcNbr*KDeltaNbr + Tv.patchInternalField()*KdeltaVap);
436
437 const scalarField a(KdeltaVap + KDeltaNbr);
438
439 valueFraction() = a/(a + KdeltaLiq);
440 refValue() = c/a;
441 refGrad() = (qr + qrNbr)/kappa(Tp);
442
443 if (debug)
444 {
445 scalarField Tc(patchInternalField());
446 scalarField qLiq((Tp - Tc)*KdeltaLiq);
447 scalarField qVap((Tp - Tv.patchInternalField())*KdeltaVap);
448
449 Info<< "T flow : " << nl << endl;
450
451 Info<< " qLiq: " << gMin(qLiq) << " - " << gMax(qLiq) << endl;
452 Info<< " qVap: " << gMin(qVap) << " - " << gMax(qVap) << endl;
453
454 scalar QLiq = gSum(qLiq*patch().magSf());
455 scalar QVap = gSum(qVap*patch().magSf());
456
457 Info<< " Heat transfer to Liq: " << QLiq << endl;
458 Info<< " Heat transfer to Vap: " << QVap << endl;
459
460 Info<< " walltemperature "
461 << " min:" << gMin(Tp)
462 << " max:" << gMax(Tp)
463 << " avg:" << gAverage(Tp)
464 << endl;
465 }
466 }
467 else
468 {
469 FatalErrorInFunction
470 << "Unknown phase type. Valid types are: "
471 << regionTypeNames_ << nl << exit(FatalError);
472 }
473
474 mixedFvPatchScalarField::updateCoeffs();
475
476 // Restore tag
477 UPstream::msgType() = oldTag;
478}
