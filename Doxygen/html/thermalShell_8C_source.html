93void thermalShell::solveEnergy()
94{
95 if (debug)
96 {
97 InfoInFunction << endl;
98 }
99
100 const areaScalarField rhoCph(Cp()*rho()*h_);
101
102 faScalarMatrix TEqn
103 (
104 fam::ddt(rhoCph, T_)
105 - fam::laplacian(kappa()*h_, T_)
106 ==
107 qs_
108 + qr()
109 + faOptions()(h_, rhoCph, T_)
110 );
111
112 TEqn.relax();
113
114 faOptions().constrain(TEqn);
115
116 TEqn.solve();
117
118 faOptions().correct(T_);
119}

172void thermalShell::evolveRegion()
173{
174 nNonOrthCorr_ = solution().get<label>("nNonOrthCorr");
175
176 for (int nonOrth = 0; nonOrth <= nNonOrthCorr_; ++nonOrth)
177 {
178 solveEnergy();
179 }
180
181 Info<< T_.name() << " min/max = " << gMinMax(T_) << endl;
182}
