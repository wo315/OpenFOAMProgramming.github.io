59Foam::RBD::joints::composite Foam::RBD::joints::floating::sixDoF()
60{
61 PtrList<joint> cj(2);
62 cj.set(0, new joints::Pxyz());
63
64 // The quaternion-based spherical joint could be used
65 // but then w must be set appropriately
66 //cj.set(1, new joints::Rs());
67
68 // Alternatively the Euler-angle joint can be used
69 cj.set(1, new joints::Rzyx());
70
71 return composite(cj);
72}
