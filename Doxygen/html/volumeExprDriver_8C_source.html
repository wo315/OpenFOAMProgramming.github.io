203Foam::autoPtr<Foam::regIOobject>
204Foam::expressions::volumeExpr::parseDriver::dupZeroField() const
205{
206 const auto* regIOobjectPtr = resultField_.get();
207
208 if (!regIOobjectPtr)
209 {
210 return nullptr;
211 }
212
213 autoPtr<regIOobject> zField;
214
215 switch (fieldGeoType_)
216 {
217 #undef doLocalCode
218 #define doLocalCode(GeoField) \
219 { \
220 const auto* ptr = dynamic_cast<const GeoField*>(regIOobjectPtr); \
221 typedef typename GeoField::value_type Type; \
222 \
223 if (ptr) \
224 { \
225 zField.reset \
226 ( \
227 GeoField::New \
228 ( \
229 word(pTraits<Type>::typeName) + word("(zero)"), \
230 (*ptr).mesh(), \
231 dimensioned<Type>(Zero), \
232 /* zeroGradient (volume) or calculated (other) */ \
233 defaultBoundaryType(*ptr) \
234 ).ptr() \
235 ); \
236 break; \
237 } \
238 }
239
240 case FieldAssociation::VOLUME_DATA:
241 {
242 doLocalCode(volScalarField);
243 doLocalCode(volVectorField);
244 doLocalCode(volTensorField);
245 doLocalCode(volSymmTensorField);
246 doLocalCode(volSphericalTensorField);
247 break;
248 }
249 case FieldAssociation::FACE_DATA:
250 {
251 doLocalCode(surfaceScalarField);
252 doLocalCode(surfaceVectorField);
253 doLocalCode(surfaceTensorField);
254 doLocalCode(surfaceSymmTensorField);
255 doLocalCode(surfaceSphericalTensorField);
256 break;
257 }
258 case FieldAssociation::POINT_DATA:
259 {
260 doLocalCode(pointScalarField);
261 doLocalCode(pointVectorField);
262 doLocalCode(pointTensorField);
263 doLocalCode(pointSymmTensorField);
264 doLocalCode(pointSphericalTensorField);
265 break;
266 }
267 default: break;
268 #undef doLocalCode
269 }
270
271 // if (!zField)
272 // {
273 // // Report
274 // }
275
276 return zField;
277}
