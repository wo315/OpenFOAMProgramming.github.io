74bool Foam::functionObjects::writeCellCentres::write()
75{
76 const volVectorField C
77 (
78 IOobject
79 (
80 mesh_.C().name(),
81 time_.timeName(),
82 mesh_,
83 IOobject::NO_READ,
84 IOobject::NO_WRITE,
85 IOobject::NO_REGISTER
86 ),
87 mesh_.C(),
88 fvPatchFieldBase::calculatedType()
89 );
90
91 Log << type() << " " << name() << " write:" << nl
92 << " writing cell-centres field " << C.name()
93 << " to " << time_.timeName() << endl;
94
95 C.write();
96
97 for (direction i = 0; i < vector::nComponents; ++i)
98 {
99 const volScalarField Ci
100 (
101 IOobject
102 (
103 mesh_.C().name() + vector::componentNames[i],
104 time_.timeName(),
105 mesh_,
106 IOobject::NO_READ,
107 IOobject::NO_WRITE,
108 IOobject::NO_REGISTER
109 ),
110 mesh_.C().component(i)
111 );
112
113 Log << " Writing the "
114 << vector::componentNames[i]
115 << " component field of the cell-centres " << Ci.name()
116 << " to " << time_.timeName() << endl;
117
118 Ci.write();
119 }
120
121 return true;
122}
