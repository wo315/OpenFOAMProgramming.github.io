44template<class Type, class CheckType = Type>
45inline bool writeMeshObject
46(
47 const word& name,
48 const fileName& meshDir,
49 Time& runTime,
50 const bool strictTypeChecking = true,
51 const bool disableHeaderChecking = false
52)
53{
54 IOobject io
55 (
56 name,
57 runTime.timeName(),
58 meshDir,
59 runTime,
60 IOobject::MUST_READ,
61 IOobject::NO_WRITE,
62 IOobject::NO_REGISTER
63 );
64
65 bool writeOk = false;
66
67 if (io.typeHeaderOk<CheckType>(strictTypeChecking, true, false))
68 {
69 Info<< " Reading " << io.headerClassName()
70 << " : " << name << endl;
71
72 // Switch off type checking (for reading e.g. faceZones as
73 // generic list of dictionaries).
74 word oldTypeName;
75 if (disableHeaderChecking)
76 {
77 oldTypeName = Type::typeName;
78 const_cast<word&>(Type::typeName) = word::null;
79 }
80
81 Type meshObject(io);
82
83 if (disableHeaderChecking)
84 {
85 const_cast<word&>(Type::typeName) = oldTypeName;
86 // Fake type back to what was in field
87 const_cast<word&>(meshObject.type()) = io.headerClassName();
88 }
89
90 Info<< " Writing " << name << endl;
91 writeOk = meshObject.regIOobject::write();
92 }
93
94 return writeOk;
95}
