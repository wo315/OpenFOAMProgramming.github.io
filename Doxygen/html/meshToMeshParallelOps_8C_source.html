41Foam::label Foam::meshToMesh::calcDistribution
42(
43 const polyMesh& src,
44 const polyMesh& tgt
45) const
46{
47 label proci = 0;
48
49 if (Pstream::parRun())
50 {
51 const bitSet hasMesh
52 (
53 UPstream::listGatherValues<bool>
54 (
55 src.nCells() > 0 || tgt.nCells() > 0
56 )
57 );
58
59 const auto nHaveMesh = hasMesh.count();
60
61 if (nHaveMesh == 1)
62 {
63 proci = hasMesh.find_first();
64 DebugInFunction
65 << "Meshes local to processor" << proci << endl;
66 }
67 else if (nHaveMesh > 1)
68 {
69 proci = -1;
70 DebugInFunction
71 << "Meshes split across multiple processors" << endl;
72 }
73
74 Pstream::broadcast(proci);
75 }
76
77 return proci;
78}

81Foam::label Foam::meshToMesh::calcOverlappingProcs
82(
83 const List<treeBoundBoxList>& procBb,
84 const boundBox& bb,
85 boolList& overlaps
86) const
87{
88 overlaps = false;
89
90 label nOverlaps = 0;
91
92 forAll(procBb, proci)
93 {
94 const treeBoundBoxList& bbp = procBb[proci];
95
96 for (const treeBoundBox& b : bbp)
97 {
98 if (b.overlaps(bb))
99 {
100 overlaps[proci] = true;
101 ++nOverlaps;
102 break;
103 }
104 }
105 }
106
107 return nOverlaps;
108}

111Foam::autoPtr<Foam::mapDistribute> Foam::meshToMesh::calcProcMap
112(
113 const polyMesh& src,
114 const polyMesh& tgt
115) const
116{
117 switch (procMapMethod_)
118 {
119 case procMapMethod::pmLOD:
120 {
121 Info<< "meshToMesh: Using processorLOD method" << endl;
122
123 // Create processor map of overlapping faces. This map gets
124 // (possibly remote) cells from the tgt mesh such that they
125 // (together) cover all of the src mesh
126 const label nGlobalSrcCells = src.globalData().nTotalCells();
127 // Note: minimum content has to be > 1 since otherwise
128 // would keep on splitting. 16 is fairly random choice.
129 const label cellsPerBox = max(16, 0.001*nGlobalSrcCells);
130 typename processorLODs::cellBox boxLOD
131 (
132 src.cells(),
133 src.faces(),
134 src.points(),
135 tgt.cells(),
136 tgt.faces(),
137 tgt.points(),
138 cellsPerBox,
139 src.nCells()
140 );
141
142 return boxLOD.map();
143 break;
144 }
145 default:
146 {
147 Info<< "meshToMesh: Using AABBTree method" << endl;
148
149 // get decomposition of cells on src mesh
150 List<treeBoundBoxList> procBb(Pstream::nProcs());
151
152 if (src.nCells() > 0)
153 {
154 procBb[Pstream::myProcNo()] = AABBTree<labelList>
155 (
156 src.cellPoints(),
157 src.points(),
158 false
159 ).boundBoxes();
160 }
161 else
162 {
163 procBb[Pstream::myProcNo()] = treeBoundBoxList();
164 }
165
166 Pstream::allGatherList(procBb);
167
168
169 if (debug)
170 {
171 InfoInFunction
172 << "Determining extent of src mesh per processor:" << nl
173 << "\tproc\tbb" << endl;
174 forAll(procBb, proci)
175 {
176 Info<< '\t' << proci << '\t' << procBb[proci] << endl;
177 }
178 }
179
180
181 // Determine which cells of tgt mesh overlaps src mesh per proc
182 labelListList sendMap(Pstream::nProcs());
183
184 {
185 // per processor indices into all segments to send
186 List<DynamicList<label>> dynSendMap(Pstream::nProcs());
187 label iniSize = floor(tgt.nCells()/Pstream::nProcs());
188
189 forAll(dynSendMap, proci)
190 {
191 dynSendMap[proci].setCapacity(iniSize);
192 }
193
194 // work array - whether src processor bb overlaps the tgt cell
195 // bounds
196 boolList procBbOverlaps(Pstream::nProcs());
197
198 for (label celli = 0; celli < tgt.nCells(); ++celli)
199 {
200 // Bounding box of tgt cell
201 boundBox cellBb(tgt.cellBb(celli));
202
203 // find the overlapping tgt cells on each src processor
204 (void)calcOverlappingProcs(procBb, cellBb, procBbOverlaps);
205
206 forAll(procBbOverlaps, proci)
207 {
208 if (procBbOverlaps[proci])
209 {
210 dynSendMap[proci].append(celli);
211 }
212 }
213 }
214
215 if (debug)
216 {
217 Pout<< "Of my " << tgt.nCells()
218 << " target cells I need to send to:" << nl
219 << "\tproc\tcells" << endl;
220 forAll(dynSendMap, proci)
221 {
222 Pout<< '\t' << proci << '\t'
223 << dynSendMap[proci].size() << endl;
224 }
225 }
226
227 // Convert DynamicList -> List
228 forAll(sendMap, proci)
229 {
230 sendMap[proci].transfer(dynSendMap[proci]);
231 }
232 }
233
234
235 labelList recvSizes;
236 Pstream::exchangeSizes(sendMap, recvSizes, UPstream::worldComm);
237
238 // Uses linear receive order
239 labelListList constructMap(UPstream::nProcs());
240
241 label constructSize = 0;
242 forAll(constructMap, proci)
243 {
244 const label len = recvSizes[proci];
245 constructMap[proci] = identity(len, constructSize);
246 constructSize += len;
247 }
248
249 return autoPtr<mapDistribute>::New
250 (
251 constructSize,
252 std::move(sendMap),
253 std::move(constructMap)
254 );
255 break;
256 }
257 }
258}

261void Foam::meshToMesh::distributeCells
262(
263 const mapDistribute& map,
264 const polyMesh& tgtMesh,
265 const globalIndex& globalI,
266 List<pointField>& points,
267 List<label>& nInternalFaces,
268 List<faceList>& faces,
269 List<labelList>& faceOwner,
270 List<labelList>& faceNeighbour,
271 List<labelList>& cellIDs,
272 List<labelList>& nbrProcIDs,
273 List<labelList>& procLocalFaceIDs
274) const
275{
276 points.setSize(Pstream::nProcs());
277 nInternalFaces.setSize(Pstream::nProcs(), 0);
278 faces.setSize(Pstream::nProcs());
279 faceOwner.setSize(Pstream::nProcs());
280 faceNeighbour.setSize(Pstream::nProcs());
281 cellIDs.setSize(Pstream::nProcs());
282
283 nbrProcIDs.setSize(Pstream::nProcs());;
284 procLocalFaceIDs.setSize(Pstream::nProcs());;
285
286
287 PstreamBuffers pBufs(Pstream::commsTypes::nonBlocking);
288
289 for (const int domain : Pstream::allProcs())
290 {
291 const labelList& sendElems = map.subMap()[domain];
292
293 if (sendElems.size())
294 {
295 // reverse cell map
296 labelList reverseCellMap(tgtMesh.nCells(), -1);
297 forAll(sendElems, subCelli)
298 {
299 reverseCellMap[sendElems[subCelli]] = subCelli;
300 }
301
302 DynamicList<face> subFaces(tgtMesh.nFaces());
303 DynamicList<label> subFaceOwner(tgtMesh.nFaces());
304 DynamicList<label> subFaceNeighbour(tgtMesh.nFaces());
305
306 DynamicList<label> subNbrProcIDs(tgtMesh.nFaces());
307 DynamicList<label> subProcLocalFaceIDs(tgtMesh.nFaces());
308
309 label nInternal = 0;
310
311 // internal faces
312 forAll(tgtMesh.faceNeighbour(), facei)
313 {
314 label own = tgtMesh.faceOwner()[facei];
315 label nbr = tgtMesh.faceNeighbour()[facei];
316 label subOwn = reverseCellMap[own];
317 label subNbr = reverseCellMap[nbr];
318
319 if (subOwn != -1 && subNbr != -1)
320 {
321 nInternal++;
322
323 if (subOwn < subNbr)
324 {
325 subFaces.append(tgtMesh.faces()[facei]);
326 subFaceOwner.append(subOwn);
327 subFaceNeighbour.append(subNbr);
328 subNbrProcIDs.append(-1);
329 subProcLocalFaceIDs.append(-1);
330 }
331 else
332 {
333 subFaces.append(tgtMesh.faces()[facei].reverseFace());
334 subFaceOwner.append(subNbr);
335 subFaceNeighbour.append(subOwn);
336 subNbrProcIDs.append(-1);
337 subProcLocalFaceIDs.append(-1);
338 }
339 }
340 }
341
342 // boundary faces for new region
343 forAll(tgtMesh.faceNeighbour(), facei)
344 {
345 label own = tgtMesh.faceOwner()[facei];
346 label nbr = tgtMesh.faceNeighbour()[facei];
347 label subOwn = reverseCellMap[own];
348 label subNbr = reverseCellMap[nbr];
349
350 if (subOwn != -1 && subNbr == -1)
351 {
352 subFaces.append(tgtMesh.faces()[facei]);
353 subFaceOwner.append(subOwn);
354 subFaceNeighbour.append(subNbr);
355 subNbrProcIDs.append(-1);
356 subProcLocalFaceIDs.append(-1);
357 }
358 else if (subOwn == -1 && subNbr != -1)
359 {
360 subFaces.append(tgtMesh.faces()[facei].reverseFace());
361 subFaceOwner.append(subNbr);
362 subFaceNeighbour.append(subOwn);
363 subNbrProcIDs.append(-1);
364 subProcLocalFaceIDs.append(-1);
365 }
366 }
367
368 // boundary faces of existing region
369 forAll(tgtMesh.boundaryMesh(), patchi)
370 {
371 const polyPatch& pp = tgtMesh.boundaryMesh()[patchi];
372 const auto* procPatch = isA<processorPolyPatch>(pp);
373
374 // Store info for faces on processor patches
375 const label nbrProci =
376 (procPatch ? procPatch->neighbProcNo() : -1);
377
378 forAll(pp, i)
379 {
380 label facei = pp.start() + i;
381 label own = tgtMesh.faceOwner()[facei];
382
383 if (reverseCellMap[own] != -1)
384 {
385 subFaces.append(tgtMesh.faces()[facei]);
386 subFaceOwner.append(reverseCellMap[own]);
387 subFaceNeighbour.append(-1);
388 subNbrProcIDs.append(nbrProci);
389 subProcLocalFaceIDs.append(i);
390 }
391 }
392 }
393
394 // reverse point map
395 labelList reversePointMap(tgtMesh.nPoints(), -1);
396 DynamicList<point> subPoints(tgtMesh.nPoints());
397 forAll(subFaces, subFacei)
398 {
399 face& f = subFaces[subFacei];
400 forAll(f, fp)
401 {
402 label pointi = f[fp];
403 if (reversePointMap[pointi] == -1)
404 {
405 reversePointMap[pointi] = subPoints.size();
406 subPoints.append(tgtMesh.points()[pointi]);
407 }
408
409 f[fp] = reversePointMap[pointi];
410 }
411 }
412
413 // tgt cells into global numbering
414 labelList globalElems(globalI.toGlobal(sendElems));
415
416 if (debug > 1)
417 {
418 forAll(sendElems, i)
419 {
420 Pout<< "tgtProc:" << Pstream::myProcNo()
421 << " sending tgt cell " << sendElems[i]
422 << "[" << globalElems[i] << "]"
423 << " to srcProc " << domain << endl;
424 }
425 }
426
427 // pass data
428 if (domain == Pstream::myProcNo())
429 {
430 // allocate my own data
431 points[Pstream::myProcNo()] = subPoints;
432 nInternalFaces[Pstream::myProcNo()] = nInternal;
433 faces[Pstream::myProcNo()] = subFaces;
434 faceOwner[Pstream::myProcNo()] = subFaceOwner;
435 faceNeighbour[Pstream::myProcNo()] = subFaceNeighbour;
436 cellIDs[Pstream::myProcNo()] = globalElems;
437 nbrProcIDs[Pstream::myProcNo()] = subNbrProcIDs;
438 procLocalFaceIDs[Pstream::myProcNo()] = subProcLocalFaceIDs;
439 }
440 else
441 {
442 // send data to other processor domains
443 UOPstream toDomain(domain, pBufs);
444
445 toDomain
446 << subPoints
447 << nInternal
448 << subFaces
449 << subFaceOwner
450 << subFaceNeighbour
451 << globalElems
452 << subNbrProcIDs
453 << subProcLocalFaceIDs;
454 }
455 }
456 }
457
458 // Start receiving
459 pBufs.finishedSends();
460
461 // Consume
462 for (const int domain : Pstream::allProcs())
463 {
464 const labelList& recvElems = map.constructMap()[domain];
465
466 if (domain != Pstream::myProcNo() && recvElems.size())
467 {
468 UIPstream str(domain, pBufs);
469
470 str >> points[domain]
471 >> nInternalFaces[domain]
472 >> faces[domain]
473 >> faceOwner[domain]
474 >> faceNeighbour[domain]
475 >> cellIDs[domain]
476 >> nbrProcIDs[domain]
477 >> procLocalFaceIDs[domain];
478 }
479
480 if (debug)
481 {
482 Pout<< "Target mesh send sizes[" << domain << "]"
483 << ": points="<< points[domain].size()
484 << ", faces=" << faces[domain].size()
485 << ", nInternalFaces=" << nInternalFaces[domain]
486 << ", faceOwn=" << faceOwner[domain].size()
487 << ", faceNbr=" << faceNeighbour[domain].size()
488 << ", cellIDs=" << cellIDs[domain].size() << endl;
489 }
490 }
491}

494void Foam::meshToMesh::distributeAndMergeCells
495(
496 const mapDistribute& map,
497 const polyMesh& tgt,
498 const globalIndex& globalI,
499 pointField& tgtPoints,
500 faceList& tgtFaces,
501 labelList& tgtFaceOwners,
502 labelList& tgtFaceNeighbours,
503 labelList& tgtCellIDs
504) const
505{
506 // Exchange per-processor data
507 List<pointField> allPoints;
508 List<label> allNInternalFaces;
509 List<faceList> allFaces;
510 List<labelList> allFaceOwners;
511 List<labelList> allFaceNeighbours;
512 List<labelList> allTgtCellIDs;
513
514 // Per target mesh face the neighbouring proc and index in
515 // processor patch (all -1 for normal boundary face)
516 List<labelList> allNbrProcIDs;
517 List<labelList> allProcLocalFaceIDs;
518
519 distributeCells
520 (
521 map,
522 tgt,
523 globalI,
524 allPoints,
525 allNInternalFaces,
526 allFaces,
527 allFaceOwners,
528 allFaceNeighbours,
529 allTgtCellIDs,
530 allNbrProcIDs,
531 allProcLocalFaceIDs
532 );
533
534 // Convert lists into format that can be used to generate a valid polyMesh
535 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
536 //
537 // Points and cells are collected into single flat lists:
538 // - i.e. proc0, proc1 ... procN
539 //
540 // Faces need to be sorted after collection to that internal faces are
541 // contiguous, followed by all boundary faces
542 //
543 // Processor patch faces between included cells on neighbouring processors
544 // are converted into internal faces
545 //
546 // Face list structure:
547 // - Per processor:
548 // - internal faces
549 // - processor faces that have been converted into internal faces
550 // - Followed by all boundary faces
551 // - from 'normal' boundary faces
552 // - from singularly-sided processor patch faces
553
554
555 // Number of internal+coupled faces
556 labelList allNIntCoupledFaces(allNInternalFaces);
557
558 // Starting offset for points
559 label nPoints = 0;
560 labelList pointOffset(Pstream::nProcs(), Zero);
561 forAll(allPoints, proci)
562 {
563 pointOffset[proci] = nPoints;
564 nPoints += allPoints[proci].size();
565 }
566
567 // Starting offset for cells
568 label nCells = 0;
569 labelList cellOffset(Pstream::nProcs(), Zero);
570 forAll(allTgtCellIDs, proci)
571 {
572 cellOffset[proci] = nCells;
573 nCells += allTgtCellIDs[proci].size();
574 }
575
576 // Count any coupled faces
577 typedef FixedList<label, 3> label3;
578 typedef HashTable<label, label3> procCoupleInfo;
579 procCoupleInfo procFaceToGlobalCell;
580
581 forAll(allNbrProcIDs, proci)
582 {
583 const labelList& nbrProci = allNbrProcIDs[proci];
584 const labelList& localFacei = allProcLocalFaceIDs[proci];
585
586 forAll(nbrProci, i)
587 {
588 if (nbrProci[i] != -1 && localFacei[i] != -1)
589 {
590 label3 key;
591 key[0] = min(proci, nbrProci[i]);
592 key[1] = max(proci, nbrProci[i]);
593 key[2] = localFacei[i];
594
595 const auto fnd = procFaceToGlobalCell.cfind(key);
596
597 if (!fnd.good())
598 {
599 procFaceToGlobalCell.insert(key, -1);
600 }
601 else
602 {
603 if (debug > 1)
604 {
605 Pout<< "Additional internal face between procs:"
606 << key[0] << " and " << key[1]
607 << " across local face " << key[2] << endl;
608 }
609
610 allNIntCoupledFaces[proci]++;
611 }
612 }
613 }
614 }
615
616
617 // Starting offset for internal faces
618 label nIntFaces = 0;
619 label nFacesTotal = 0;
620 labelList internalFaceOffset(Pstream::nProcs(), Zero);
621 forAll(allNIntCoupledFaces, proci)
622 {
623 label nCoupledFaces =
624 allNIntCoupledFaces[proci] - allNInternalFaces[proci];
625
626 internalFaceOffset[proci] = nIntFaces;
627 nIntFaces += allNIntCoupledFaces[proci];
628 nFacesTotal += allFaceOwners[proci].size() - nCoupledFaces;
629 }
630
631 tgtPoints.setSize(nPoints);
632 tgtFaces.setSize(nFacesTotal);
633 tgtFaceOwners.setSize(nFacesTotal);
634 tgtFaceNeighbours.setSize(nFacesTotal);
635 tgtCellIDs.setSize(nCells);
636
637 // Insert points
638 forAll(allPoints, proci)
639 {
640 const pointField& pts = allPoints[proci];
641 SubList<point>(tgtPoints, pts.size(), pointOffset[proci]) = pts;
642 }
643
644 // Insert cellIDs
645 forAll(allTgtCellIDs, proci)
646 {
647 const labelList& cellIDs = allTgtCellIDs[proci];
648 SubList<label>(tgtCellIDs, cellIDs.size(), cellOffset[proci]) = cellIDs;
649 }
650
651
652 // Insert internal faces (from internal faces)
653 forAll(allFaces, proci)
654 {
655 const faceList& fcs = allFaces[proci];
656 const labelList& faceOs = allFaceOwners[proci];
657 const labelList& faceNs = allFaceNeighbours[proci];
658
659 SubList<face> slice
660 (
661 tgtFaces,
662 allNInternalFaces[proci],
663 internalFaceOffset[proci]
664 );
665 slice = SubList<face>(fcs, allNInternalFaces[proci]);
666 forAll(slice, i)
667 {
668 add(slice[i], pointOffset[proci]);
669 }
670
671 SubField<label> ownSlice
672 (
673 tgtFaceOwners,
674 allNInternalFaces[proci],
675 internalFaceOffset[proci]
676 );
677 ownSlice = SubField<label>(faceOs, allNInternalFaces[proci]);
678 add(ownSlice, cellOffset[proci]);
679
680 SubField<label> nbrSlice
681 (
682 tgtFaceNeighbours,
683 allNInternalFaces[proci],
684 internalFaceOffset[proci]
685 );
686 nbrSlice = SubField<label>(faceNs, allNInternalFaces[proci]);
687 add(nbrSlice, cellOffset[proci]);
688
689 internalFaceOffset[proci] += allNInternalFaces[proci];
690 }
691
692
693 // Insert internal faces (from coupled face-pairs)
694 forAll(allNbrProcIDs, proci)
695 {
696 const labelList& nbrProci = allNbrProcIDs[proci];
697 const labelList& localFacei = allProcLocalFaceIDs[proci];
698 const labelList& faceOs = allFaceOwners[proci];
699 const faceList& fcs = allFaces[proci];
700
701 forAll(nbrProci, i)
702 {
703 if (nbrProci[i] != -1 && localFacei[i] != -1)
704 {
705 label3 key;
706 key[0] = min(proci, nbrProci[i]);
707 key[1] = max(proci, nbrProci[i]);
708 key[2] = localFacei[i];
709
710 auto fnd = procFaceToGlobalCell.find(key);
711
712 if (fnd.good())
713 {
714 label tgtFacei = fnd();
715 if (tgtFacei == -1)
716 {
717 // on first visit store the new cell on this side
718 fnd() = cellOffset[proci] + faceOs[i];
719 }
720 else
721 {
722 // get owner and neighbour in new cell numbering
723 label newOwn = cellOffset[proci] + faceOs[i];
724 label newNbr = fnd();
725 label tgtFacei = internalFaceOffset[proci]++;
726
727 if (debug > 1)
728 {
729 Pout<< " proc " << proci
730 << "\tinserting face:" << tgtFacei
731 << " connection between owner " << newOwn
732 << " and neighbour " << newNbr
733 << endl;
734 }
735
736 if (newOwn < newNbr)
737 {
738 // we have correct orientation
739 tgtFaces[tgtFacei] = fcs[i];
740 tgtFaceOwners[tgtFacei] = newOwn;
741 tgtFaceNeighbours[tgtFacei] = newNbr;
742 }
743 else
744 {
745 // reverse orientation
746 tgtFaces[tgtFacei] = fcs[i].reverseFace();
747 tgtFaceOwners[tgtFacei] = newNbr;
748 tgtFaceNeighbours[tgtFacei] = newOwn;
749 }
750
751 add(tgtFaces[tgtFacei], pointOffset[proci]);
752
753 // mark with unique value
754 fnd() = -2;
755 }
756 }
757 }
758 }
759 }
760
761
762 forAll(allNbrProcIDs, proci)
763 {
764 const labelList& nbrProci = allNbrProcIDs[proci];
765 const labelList& localFacei = allProcLocalFaceIDs[proci];
766 const labelList& faceOs = allFaceOwners[proci];
767 const labelList& faceNs = allFaceNeighbours[proci];
768 const faceList& fcs = allFaces[proci];
769
770 forAll(nbrProci, i)
771 {
772 // coupled boundary face
773 if (nbrProci[i] != -1 && localFacei[i] != -1)
774 {
775 label3 key;
776 key[0] = min(proci, nbrProci[i]);
777 key[1] = max(proci, nbrProci[i]);
778 key[2] = localFacei[i];
779
780 label tgtFacei = procFaceToGlobalCell[key];
781
782 if (tgtFacei == -1)
783 {
784 FatalErrorInFunction
785 << "Unvisited " << key
786 << abort(FatalError);
787 }
788 else if (tgtFacei != -2)
789 {
790 label newOwn = cellOffset[proci] + faceOs[i];
791 label tgtFacei = nIntFaces++;
792
793 if (debug > 1)
794 {
795 Pout<< " proc " << proci
796 << "\tinserting boundary face:" << tgtFacei
797 << " from coupled face " << key
798 << endl;
799 }
800
801 tgtFaces[tgtFacei] = fcs[i];
802 add(tgtFaces[tgtFacei], pointOffset[proci]);
803
804 tgtFaceOwners[tgtFacei] = newOwn;
805 tgtFaceNeighbours[tgtFacei] = -1;
806 }
807 }
808 // normal boundary face
809 else
810 {
811 label own = faceOs[i];
812 label nbr = faceNs[i];
813 if ((own != -1) && (nbr == -1))
814 {
815 label newOwn = cellOffset[proci] + faceOs[i];
816 label tgtFacei = nIntFaces++;
817
