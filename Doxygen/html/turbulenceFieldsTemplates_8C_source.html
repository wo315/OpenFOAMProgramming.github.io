33template<class Type>
34void Foam::functionObjects::turbulenceFields::processField
35(
36 const word& fieldName,
37 const tmp<GeometricField<Type, fvPatchField, volMesh>>& tvalue
38)
39{
40 typedef GeometricField<Type, fvPatchField, volMesh> FieldType;
41
42 const word localName(IOobject::scopedName(prefix_, fieldName));
43
44 FieldType* fldPtr = obr_.getObjectPtr<FieldType>(localName);
45
46 if (fldPtr)
47 {
48 (*fldPtr) == tvalue();
49 }
50 else
51 {
52 fldPtr = new FieldType
53 (
54 IOobject
55 (
56 localName,
57 obr_.time().timeName(),
58 obr_,
59 IOobject::LAZY_READ,
60 IOobject::NO_WRITE,
61 IOobject::REGISTER
62 ),
63 tvalue
64 );
65
66 obr_.store(fldPtr);
67 }
68}

88template<class Model>
89Foam::tmp<Foam::volScalarField>
90Foam::functionObjects::turbulenceFields::L
91(
92 const Model& model
93) const
94{
95 // (P:Eq. 10.37)
96 const scalar Cmu = 0.09;
97 const dimensionedScalar eps0(sqr(dimVelocity)/dimTime, SMALL);
98
99 return tmp<volScalarField>::New
100 (
101 "L.tmp",
102 pow(Cmu, 0.75)*pow(model.k(), 1.5)/(model.epsilon() + eps0)
103 );
104}

107template<class Model>
108Foam::tmp<Foam::volScalarField>
109Foam::functionObjects::turbulenceFields::I
110(
111 const Model& model
112) const
113{
114 // (P:p. 183)
115 // root-mean-square of velocity fluctuations - isotropic turbulence
116 const volScalarField uPrime(sqrt((2.0/3.0)*model.k()));
117 const dimensionedScalar U0(dimVelocity, SMALL);
118
119 return tmp<volScalarField>::New
120 (
121 "I.tmp",
122 uPrime/max(max(uPrime, mag(model.U())), U0)
123 );
124}
