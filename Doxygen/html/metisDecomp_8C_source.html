73Foam::label Foam::metisDecomp::decomposeSerial
74(
75 const labelList& adjncy,
76 const labelList& xadj,
77 const List<scalar>& cWeights,
78 labelList& decomp
79) const
80{
81 // Method of decomposition
82 // recursive: multi-level recursive bisection (default)
83 // k-way: multi-level k-way
84 word method("recursive");
85
86 const dictionary* coeffsDictPtr = decompDict_.findDict("metisCoeffs");
87
88 idx_t numCells = max(0, (xadj.size()-1));
89
90 // Decomposition options
91 List<idx_t> options(METIS_NOPTIONS);
92 METIS_SetDefaultOptions(options.data());
93
94 // Processor weights initialised with no size, only used if specified in
95 // a file
96 Field<real_t> procWeights;
97
98 // Cell weights (so on the vertices of the dual)
99 List<idx_t> cellWeights;
100
101 // Face weights (so on the edges of the dual)
102 List<idx_t> faceWeights;
103
104 // Check for externally provided cellweights and if so initialise weights
105 // Note: min, not gMin since routine runs on master only.
106 const scalar minWeights = min(cWeights);
107
108 if (!cWeights.empty())
109 {
110 if (minWeights <= 0)
111 {
112 WarningInFunction
113 << "Illegal minimum weight " << minWeights
114 << endl;
115 }
116
117 if (cWeights.size() != numCells)
118 {
119 FatalErrorInFunction
120 << "Number of cell weights " << cWeights.size()
121 << " does not equal number of cells " << numCells
122 << exit(FatalError);
123 }
124
125 // Convert to integers.
126 cellWeights.setSize(cWeights.size());
127 forAll(cellWeights, i)
128 {
129 cellWeights[i] = idx_t(cWeights[i]/minWeights);
130 }
131 }
132
133
134 // Check for user supplied weights and decomp options
135 if (coeffsDictPtr)
136 {
137 const dictionary& coeffDict = *coeffsDictPtr;
138
139 word weightsFile;
140
141 if (coeffDict.readIfPresent("method", method))
142 {
143 if (method != "recursive" && method != "k-way")
144 {
145 FatalErrorInFunction
146 << "Method " << method << " in metisCoeffs in dictionary : "
147 << decompDict_.name()
148 << " should be 'recursive' or 'k-way'"
149 << exit(FatalError);
150 }
151
152 Info<< "metisDecomp : Using Metis method " << method
153 << nl << endl;
154 }
155
156 if (coeffDict.readIfPresent("options", options))
157 {
158 if (options.size() != METIS_NOPTIONS)
159 {
160 FatalErrorInFunction
161 << "Number of options in metisCoeffs in dictionary : "
162 << decompDict_.name()
163 << " should be " << METIS_NOPTIONS
164 << exit(FatalError);
165 }
166
167 Info<< "metisDecomp : Using Metis options " << options
168 << nl << endl;
169 }
170
171 if (coeffDict.readIfPresent("processorWeights", procWeights))
172 {
173 if (procWeights.size() != nDomains_)
174 {
175 FatalIOErrorInFunction(coeffDict)
176 << "processorWeights (" << procWeights.size()
177 << ") != number of domains (" << nDomains_ << ")" << nl
178 << exit(FatalIOError);
179 }
180
181 procWeights /= sum(procWeights);
182 }
183 }
184
185 idx_t ncon = 1;
186 idx_t nProcs = nDomains_;
187
188 // Addressing
189 ConstPrecisionAdaptor<idx_t, label, List> xadj_param(xadj);
190 ConstPrecisionAdaptor<idx_t, label, List> adjncy_param(adjncy);
191
192 // Output: cell -> processor addressing
193 decomp.resize(numCells);
194 PrecisionAdaptor<idx_t, label, List> decomp_param(decomp, false);
195
196 // Avoid potential nullptr issues with zero-sized arrays
197 labelList adjncy_dummy, xadj_dummy, decomp_dummy;
198 if (!numCells)
199 {
200 adjncy_dummy.resize(1, 0);
201 adjncy_param.set(adjncy_dummy);
202
203 xadj_dummy.resize(2, 0);
204 xadj_param.set(xadj_dummy);
205
206 decomp_dummy.resize(1, 0);
207 decomp_param.clear(); // Avoid propagating spurious values
208 decomp_param.set(decomp_dummy);
209 }
210
211
212 //
213 // Decompose
214 //
215
216 // Output: number of cut edges
217 idx_t edgeCut = 0;
218
219 if (method == "recursive")
220 {
221 METIS_PartGraphRecursive
222 (
223 &numCells, // num vertices in graph
224 &ncon, // num balancing constraints
225 xadj_param.constCast().data(), // indexing into adjncy
226 adjncy_param.constCast().data(), // neighbour info
227 cellWeights.data(), // vertex wts
228 nullptr, // vsize: total communication vol
229 faceWeights.data(), // edge wts
230 &nProcs, // nParts
231 procWeights.data(), // tpwgts
232 nullptr, // ubvec: processor imbalance (default)
233 options.data(),
234 &edgeCut,
235 decomp_param.ref().data()
236 );
237 }
238 else
239 {
240 METIS_PartGraphKway
241 (
242 &numCells, // num vertices in graph
243 &ncon, // num balancing constraints
244 xadj_param.constCast().data(), // indexing into adjncy
245 adjncy_param.constCast().data(), // neighbour info
246 cellWeights.data(), // vertex wts
247 nullptr, // vsize: total communication vol
248 faceWeights.data(), // edge wts
249 &nProcs, // nParts
250 procWeights.data(), // tpwgts
251 nullptr, // ubvec: processor imbalance (default)
252 options.data(),
253 &edgeCut,
254 decomp_param.ref().data()
255 );
256 }
257
258 return edgeCut;
259}
