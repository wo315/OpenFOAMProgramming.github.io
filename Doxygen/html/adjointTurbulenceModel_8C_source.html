76autoPtr<adjointTurbulenceModel> adjointTurbulenceModel::New
77(
78 incompressibleVars& primalVars,
79 incompressibleAdjointMeanFlowVars& adjointVars,
80 objectiveManager& objManager,
81 const word& adjointTurbulenceModelName
82)
83{
84 const word modelType
85 (
86 IOdictionary
87 (
88 IOobject
89 (
90 "turbulenceProperties",
91 primalVars.U().time().constant(),
92 primalVars.U().db(),
93 IOobject::MUST_READ,
94 IOobject::NO_WRITE,
95 IOobject::NO_REGISTER
96 )
97 ).get<word>("simulationType")
98 );
99
100 Info<< "Selecting turbulence model type " << modelType << endl;
101
102 auto* ctorPtr = adjointTurbulenceModelConstructorTable(modelType);
103
104 if (!ctorPtr)
105 {
106 FatalErrorInLookup
107 (
108 "adjointTurbulenceModel",
109 modelType,
110 *adjointTurbulenceModelConstructorTablePtr_
111 ) << exit(FatalError);
112 }
113
114 return autoPtr<adjointTurbulenceModel>
115 (
116 ctorPtr
117 (
118 primalVars,
119 adjointVars,
120 objManager,
121 adjointTurbulenceModelName
122 )
123 );
124}
