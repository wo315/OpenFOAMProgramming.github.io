31if (args.dryRun() || args.found("dry-run-write"))
32{
33 Foam::Info

53 runTime.stopAt(Foam::Time::saWriteNow);

67}
68else
69{
70 Foam::Info << "Create mesh";

79 new Foam::fvMesh

84 runTime.timeName(),

88 false
