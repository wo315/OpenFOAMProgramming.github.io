294inline void Foam::FixedList<T, N>::checkStart(const label start) const
295{
296 if (start < 0 || (start && unsigned(start) >= N))
297 {
298 // Note: always accept start=0, even for zero-sized lists
299 FatalErrorInFunction
300 << "start " << start << " out of range [0," << N << ")"
301 << abort(FatalError);
302 }

330template<class T, unsigned N>
331inline bool Foam::FixedList<T, N>::uniform() const
332{
333 if (empty()) return false; // <- Compile-time disabled anyhow
334
335 for (unsigned i=1; i<N; ++i)
336 {
337 if (v_[0] != v_[i])
338 {
339 return false;
340 }
341 }
342
343 return true;
344}
