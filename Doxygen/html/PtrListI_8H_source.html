147template<class T>
148inline void Foam::PtrList<T>::push_back(PtrList<T>&& other)
149{
150 if (this == &other)
151 {
152 FatalErrorInFunction
153 << "Attempted push_back to self"
154 << abort(FatalError);
155 }
156
157 const label idx = this->size();
158 const label len = other.size();
159
160 resize(idx + len);
161
162 for (label i = 0; i < len; ++i)
163 {
164 set(idx + i, other.release(i)); // Take pointer ownership
165 }
166
167 other.clear();
168}

192inline Foam::autoPtr<T> Foam::PtrList<T>::set(const label i, T* ptr)
193{
194 // UPtrList::set returns a nullptr if trying to set with the same
195 // pointer (again). This prevents the autoPtr from managing the

246template<class T>
247inline Foam::autoPtr<T> Foam::PtrList<T>::release(const label i)
248{
249 if (i < 0 || i >= this->size())
250 {
251 return nullptr;
252 }
253
254 return autoPtr<T>(UPtrList<T>::set(i, nullptr));
255}

258template<class T>
259inline void Foam::PtrList<T>::transfer(PtrList<T>& list)
260{
261 if (this == &list)
262 {
263 return; // Self-assignment is a no-op
264 }
265
266 (this->ptrs_).free(); // Free and nullify old pointers
267 UPtrList<T>::transfer(list);
268}
