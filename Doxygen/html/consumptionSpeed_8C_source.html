59Foam::scalar Foam::consumptionSpeed::omega0Sigma
60(
61 scalar sigma,
62 scalar a
63) const
64{
65 scalar omega0 = 0.0;
66
67 if (sigma < sigmaExt_)
68 {
69 omega0 = max
70 (
71 a*omega0_*(1.0 - exp(eta_*max(sigma, 0.0))),
72 omegaMin_
73 ) ;
74 }
75
76 return omega0;
77}
