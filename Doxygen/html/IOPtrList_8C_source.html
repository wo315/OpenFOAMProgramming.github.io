121template<class T>
122Foam::PtrList<T> Foam::IOPtrList<T>::readContents(const IOobject& io)
123{
124 IOobject rio(io, IOobjectOption::NO_REGISTER);
125 if (rio.readOpt() == IOobjectOption::MUST_READ_IF_MODIFIED)
126 {
127 rio.readOpt(IOobjectOption::MUST_READ);
128 }
129
130 IOPtrList<T> reader(rio);
131
132 return PtrList<T>(std::move(static_cast<PtrList<T>&>(reader)));
133}
