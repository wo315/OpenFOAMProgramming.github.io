66void Foam::porosityModelList::reset(const dictionary& dict)
67{
68 label count = 0;
69 for (const entry& dEntry : dict)
70 {
71 if (dEntry.isDict())
72 {
73 ++count;
74 }
75 }
76
77 this->resize(count);
78
79 count = 0;
80 for (const entry& dEntry : dict)
81 {
82 if (dEntry.isDict())
83 {
84 const word& name = dEntry.keyword();
85 const dictionary& modelDict = dEntry.dict();
86
87 this->set
88 (
89 count++,
90 porosityModel::New(name, mesh_, modelDict)
91 );
92 }
93 }
94}
