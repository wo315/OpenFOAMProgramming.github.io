74Foam::tmp<Foam::volScalarField>
75Foam::multiphaseEuler::dragModels::GidaspowErgunWenYu::K
76(
77 const volScalarField& Ur
78) const
79{
80 volScalarField alpha2(max(phase2_, scalar(1.0e-6)));
81 volScalarField d(phase1_.d());
82 volScalarField bp(pow(alpha2, -2.65));
83 volScalarField Re(max(Ur*d/phase2_.nu(), scalar(1.0e-3)));
84
85 volScalarField Cds
86 (
87 neg(Re - 1000)*(24.0*(1.0 + 0.15*pow(Re, 0.687))/Re)
88 + pos0(Re - 1000)*0.44
89 );
90
91 // Wen and Yu (1966)
92 return
93 (
94 pos0(alpha2 - 0.8)
95 *(0.75*Cds*phase2_.rho()*Ur*bp/d)
96 + neg(alpha2 - 0.8)
97 *(
98 150.0*phase1_*phase2_.nu()*phase2_.rho()/(sqr(alpha2*d))
99 + 1.75*phase2_.rho()*Ur/(alpha2*d)
100 )
101 );
102}
