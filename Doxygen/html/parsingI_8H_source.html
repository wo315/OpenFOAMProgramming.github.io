28inline Foam::parsing::errorType Foam::parsing::checkConversion
29(
30 const char* buf,
31 char* endptr
32)
33{
34 if (errno || endptr == buf)
35 {
36 // Some type of error OR no conversion
37 return (errno == ERANGE ? errorType::RANGE : errorType::GENERAL);
38 }
39
40 // Trailing spaces are permitted
41 while (isspace(*endptr))
42 {
43 ++endptr;
44 }
45
46 if (*endptr != '\0')
47 {
48 // Trailing content
49 return errorType::TRAILING;
50 }
51
52 // Valid conversion
53 return errorType::NONE;
54}
