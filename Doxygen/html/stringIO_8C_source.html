43Foam::Istream& Foam::operator>>(Istream& is, string& val)
44{
45 token tok(is);
46
47 if (tok.isString())
48 {
49 val = tok.stringToken();
50 }
51 else
52 {
53 FatalIOErrorInFunction(is);
54 if (tok.good())
55 {
56 FatalIOError
57 << "Wrong token type - expected string, found "
58 << tok.info();
59 }
60 else
61 {
62 FatalIOError
63 << "Bad token - could not get string";
64 }
65 FatalIOError << exit(FatalIOError);
66 is.setBad();
67 return is;
68 }
69
70 is.check(FUNCTION_NAME);
71 return is;
72}
