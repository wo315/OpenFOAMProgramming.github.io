39// Return the position with word boundary alignment
40inline static label byteAlign(const label pos, const size_t align)
41{
42 return
43 (
44 (align > 1)
45 ? (align + ((pos - 1) & ~(align - 1)))
46 : pos
47 );
48}

55inline void Foam::UOPstreamBase::prepareBuffer
56(
57 const size_t count,
58 const size_t align
59)
60{
61 if (!count)
62 {
63 return;
64 }
65
66 // Align for the next output position
67 const label pos = byteAlign(sendBuf_.size(), align);
68
69 // Extend buffer (as required)
70 sendBuf_.reserve(max(1000, label(pos + count)));
71
72 // Move to the aligned output position. Fill any gap with nul char.
73 sendBuf_.resize(pos, '\0');
74}

84inline void Foam::UOPstreamBase::writeToBuffer
85(
86 const void* data,
87 const size_t count,
88 const size_t align
89)
90{
91 if (!count)
92 {
93 return;
94 }
95
96 prepareBuffer(count, align);
97
98 // The aligned output position
99 const label pos = sendBuf_.size();
100
101 // Extend the addressable range for direct pointer access
102 sendBuf_.resize(pos + count);
103
104 char* const __restrict__ buf = (sendBuf_.data() + pos);
105 const char* const __restrict__ input = reinterpret_cast<const char*>(data);
106
107 for (size_t i = 0; i < count; ++i)
108 {
109 buf[i] = input[i];
110 }
111}

201bool Foam::UOPstreamBase::write(const token& tok)
202{
203 // Direct token handling only for some types
204
205 switch (tok.type())
206 {
207 case token::tokenType::FLAG :
208 {
209 putChar(token::tokenType::FLAG);
210 putChar(tok.flagToken());
211
212 return true;
213 }
214
215 // The word-variants
216 case token::tokenType::WORD :
217 case token::tokenType::DIRECTIVE :
218 {
219 putChar(tok.type());
220 putString(tok.wordToken());
221
222 return true;
223 }
224
225 // The string-variants
226 case token::tokenType::STRING :
227 case token::tokenType::EXPRESSION :
228 case token::tokenType::VARIABLE :
229 case token::tokenType::VERBATIM :
230 {
231 putChar(tok.type());
232 putString(tok.stringToken());
233
234 return true;
235 }
236
237 default:
238 break;
239 }
240
241 return false;
242}

256Foam::Ostream& Foam::UOPstreamBase::write(const char* str)
257{
258 const word nonWhiteChars(string::validate<word>(str));
259
260 if (nonWhiteChars.size() == 1)
261 {
262 return write(nonWhiteChars[0]);
263 }
264 else if (nonWhiteChars.size())
265 {
266 return write(nonWhiteChars);
267 }
268
269 return *this;

343Foam::Ostream& Foam::UOPstreamBase::write
344(
345 const char* data,
346 std::streamsize count
347)
348{
349 if (format() != BINARY)
350 {
351 FatalErrorInFunction
352 << "stream format not binary"
353 << Foam::abort(FatalError);
354 }
355
356 // Align on word boundary (64-bit)
357 writeToBuffer(data, count, 8);
358
359 return *this;
360}

379bool Foam::UOPstreamBase::beginRawWrite(std::streamsize count)
380{
381 if (format() != BINARY)
382 {
383 FatalErrorInFunction
384 << "stream format not binary"
385 << Foam::abort(FatalError);
386 }
387
388 // Align on word boundary (64-bit)

391
