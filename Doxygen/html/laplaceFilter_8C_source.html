98Foam::tmp<Foam::volScalarField> Foam::laplaceFilter::operator()
99(
100 const tmp<volScalarField>& unFilteredField
101) const
102{
103 correctBoundaryConditions(unFilteredField);
104
105 tmp<volScalarField> filteredField =
