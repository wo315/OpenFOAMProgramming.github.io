60Foam::volScalarField& Foam::functionObjects::scalarTransport::transportedField()
61{
62 if (!foundObject<volScalarField>(fieldName_))
63 {
64 auto tfldPtr = tmp<volScalarField>::New
65 (
66 IOobject
67 (
68 fieldName_,
69 mesh_.time().timeName(),
70 mesh_,
71 IOobject::MUST_READ,
72 IOobject::NO_WRITE,
73 IOobject::REGISTER
74 ),
75 mesh_
76 );
77 store(fieldName_, tfldPtr);
78
79 if (phaseName_ != "none")
80 {
81 mesh_.setFluxRequired(fieldName_);
82 }
83 }
84
85 return lookupObjectRef<volScalarField>(fieldName_);
86}

89Foam::tmp<Foam::volScalarField> Foam::functionObjects::scalarTransport::D
90(
91 const volScalarField& s,
92 const surfaceScalarField& phi
93) const
94{
95 const word Dname("D" + s.name());
96
97 if (constantD_)
98 {
99 return tmp<volScalarField>::New
100 (
101 IOobject
102 (
103 Dname,
104 mesh_.time().timeName(),
105 mesh_.time(),
106 IOobject::NO_READ,
107 IOobject::NO_WRITE
108 ),
109 mesh_,
110 dimensionedScalar(Dname, phi.dimensions()/dimLength, D_)
111 );
112 }
113
114 if (nutName_ != "none")
115 {
116 const volScalarField& nutMean =
117 mesh_.lookupObject<volScalarField>(nutName_);
118
119 return tmp<volScalarField>::New(Dname, nutMean);
120 }
121
122 // Incompressible
123 {
124 const auto* turb =
125 findObject<incompressible::turbulenceModel>
126 (
127 turbulenceModel::propertiesName
128 );
129
130 if (turb)
131 {
132 return tmp<volScalarField>::New
133 (
134 Dname,
135 alphaD_ * turb->nu() + alphaDt_ * turb->nut()
136 );
137 }
138 }
139
140 // Compressible
141 {
142 const auto* turb =
143 findObject<compressible::turbulenceModel>
144 (
145 turbulenceModel::propertiesName
146 );
147
148 if (turb)
149 {
150 return tmp<volScalarField>::New
151 (
152 Dname,
153 alphaD_ * turb->mu() + alphaDt_ * turb->mut()
154 );
155 }
156 }
157
158
159 return tmp<volScalarField>::New
160 (
161 IOobject
162 (
163 Dname,
164 mesh_.time().timeName(),
165 mesh_.time(),
166 IOobject::NO_READ,
167 IOobject::NO_WRITE
168 ),
169 mesh_,
170 dimensionedScalar(phi.dimensions()/dimLength, Zero)
171 );
172}

184 fvMeshFunctionObject(name, runTime, dict),

195 constantD_(false),

198 schemesField_("unknown-schemesField"),
199 fvOptions_(mesh_),
200 bounded01_(dict.getOrDefault("bounded01", true))

202 read(dict);
203

223bool Foam::functionObjects::scalarTransport::read(const dictionary& dict)
224{
225 fvMeshFunctionObject::read(dict);
226
227 dict.readIfPresent("phi", phiName_);
228 dict.readIfPresent("rho", rhoName_);
229 dict.readIfPresent("nut", nutName_);
230 dict.readIfPresent("phase", phaseName_);
231 dict.readIfPresent("bounded01", bounded01_);
232
233 schemesField_ = dict.getOrDefault("schemesField", fieldName_);
234 constantD_ = dict.readIfPresent("D", D_);
235 alphaD_ = dict.getOrDefault<scalar>("alphaD", 1);
236 alphaDt_ = dict.getOrDefault<scalar>("alphaDt", 1);
237
238 dict.readIfPresent("nCorr", nCorr_);
239 dict.readIfPresent("resetOnStartUp", resetOnStartUp_);
240
241 if (dict.found("fvOptions"))
242 {
243 fvOptions_.reset(dict.subDict("fvOptions"));
244 }
245
246 return true;
247}

250bool Foam::functionObjects::scalarTransport::execute()
251{
252 volScalarField& s = transportedField();
253
254 Log << type() << " execute: " << s.name() << endl;
255
256 const surfaceScalarField& phi =
257 mesh_.lookupObject<surfaceScalarField>(phiName_);
258
259 // Calculate the diffusivity
260 volScalarField D(this->D(s, phi));
261
262 word divScheme("div(phi," + schemesField_ + ")");
263 word laplacianScheme("laplacian(" + D.name() + "," + schemesField_ + ")");
264
265 // Set under-relaxation coeff
266 scalar relaxCoeff = 0.0;
267 if (mesh_.relaxEquation(schemesField_))
268 {
269 relaxCoeff = mesh_.equationRelaxationFactor(schemesField_);
270 }
271
272 // Two phase scalar transport
273 if (phaseName_ != "none")
274 {
275 const volScalarField& alpha =
276 mesh_.lookupObject<volScalarField>(phaseName_);
277
278 const surfaceScalarField& limitedPhiAlpha =
279 mesh_.lookupObject<surfaceScalarField>(phasePhiCompressedName_);
280
281 D *= pos(alpha - 0.99);
282
283 // Reset D dimensions consistent with limitedPhiAlpha
284 D.dimensions().reset(limitedPhiAlpha.dimensions()/dimLength);
285
286 // Solve
287 tmp<surfaceScalarField> tTPhiUD;
288 for (label i = 0; i <= nCorr_; i++)
289 {
290 fvScalarMatrix sEqn
291 (
292 fvm::ddt(s)
293 + fvm::div(limitedPhiAlpha, s, divScheme)
294 - fvm::laplacian(D, s, laplacianScheme)
295 ==
296 alpha*fvOptions_(s)
297 );
298
299 sEqn.relax(relaxCoeff);
300 fvOptions_.constrain(sEqn);
301 sEqn.solve(mesh_.solverDict(schemesField_));
302
303 tTPhiUD = sEqn.flux();
304 }
305
306 if (bounded01_)
307 {
308 MULES::explicitSolve
309 (
310 geometricOneField(),
311 s,
312 phi,
313 tTPhiUD.ref(),
314 oneField(),
315 zeroField()
316 );
317 }
318 }
319 else if (phi.dimensions() == dimMass/dimTime)
320 {
321 const volScalarField& rho = lookupObject<volScalarField>(rhoName_);
322
323 for (label i = 0; i <= nCorr_; i++)
324 {
325
326 fvScalarMatrix sEqn
327 (
328 fvm::ddt(rho, s)
329 + fvm::div(phi, s, divScheme)
330 - fvm::laplacian(D, s, laplacianScheme)
331 ==
332 fvOptions_(rho, s)
333 );
334
335 sEqn.relax(relaxCoeff);
336
337 fvOptions_.constrain(sEqn);
338
339 sEqn.solve(mesh_.solverDict(schemesField_));
340 }
341 }
342 else if (phi.dimensions() == dimVolume/dimTime)
343 {
344 for (label i = 0; i <= nCorr_; i++)
345 {
346 fvScalarMatrix sEqn
347 (
348 fvm::ddt(s)
349 + fvm::div(phi, s, divScheme)
350 - fvm::laplacian(D, s, laplacianScheme)
351 ==
352 fvOptions_(s)
353 );
354
355 sEqn.relax(relaxCoeff);
356
357 fvOptions_.constrain(sEqn);
358
359 sEqn.solve(mesh_.solverDict(schemesField_));
360 }
361 }
362 else
363 {
364 FatalErrorInFunction
365 << "Incompatible dimensions for phi: " << phi.dimensions() << nl
366 << "Dimensions should be " << dimMass/dimTime << " or "
367 << dimVolume/dimTime << exit(FatalError);
368 }
369
370 Log << endl;
371
372 return true;
373}
