65Foam::tmp<Foam::volScalarField> Foam::dragModels::Lain::CdRe() const
66{
67 volScalarField Re(pair_.Re());
68
69 return
70 neg(Re - 1.5)*16.0
71 + pos0(Re - 1.5)*neg(Re - 80.0)*14.9*pow(Re, 0.22)
72 + pos0(Re - 80.0)*neg(Re - 1500.0)*48*(1.0 - 2.21/sqrt(max(Re, SMALL)))
73 + pos0(Re - 1500.0)*2.61*Re;
74}
