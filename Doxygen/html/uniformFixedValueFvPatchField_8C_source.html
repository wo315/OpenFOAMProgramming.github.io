131void Foam::uniformFixedValueFvPatchField<Type>::autoMap
132(
133 const fvPatchFieldMapper& mapper
134)
135{
136 fixedValueFvPatchField<Type>::autoMap(mapper);
137
138 if (refValueFunc_)
139 {
140 refValueFunc_().autoMap(mapper);

142 if (refValueFunc_().constant())
143 {
