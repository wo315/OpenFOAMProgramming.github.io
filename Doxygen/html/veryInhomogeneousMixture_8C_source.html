65template<class ThermoType>
66const ThermoType& Foam::veryInhomogeneousMixture<ThermoType>::mixture
67(
68 const scalar ft,
69 const scalar fu
70) const
71{
72 if (ft < 0.0001)
73 {
74 return oxidant_;
75 }
76 else
77 {
78 scalar ox = 1 - ft - (ft - fu)*stoicRatio().value();
79 scalar pr = 1 - fu - ox;
80
81 mixture_ = fu*fuel_;
82 mixture_ += ox*oxidant_;
83 mixture_ += pr*products_;
84
85 return mixture_;
86 }
87}

102template<class ThermoType>
103const ThermoType& Foam::veryInhomogeneousMixture<ThermoType>::getLocalThermo
104(
105 const label speciei
106) const
107{
108 if (speciei == 0)
109 {
110 return fuel_;
111 }
112 else if (speciei == 1)
113 {
114 return oxidant_;
115 }
116 else if (speciei == 2)
117 {
118 return products_;
119 }
120 else
121 {
122 FatalErrorInFunction
123 << "Unknown specie index " << speciei << ". Valid indices are 0..2"
124 << abort(FatalError);
125
126 return fuel_;
127 }
128}
