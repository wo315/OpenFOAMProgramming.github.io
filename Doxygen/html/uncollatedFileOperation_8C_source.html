71Foam::fileName Foam::fileOperations::uncollatedFileOperation::filePathInfo
72(
73 const bool checkGlobal,
74 const bool isFile,
75 const IOobject& io,
76 const bool search
77) const
78{
79 if (io.instance().isAbsolute())
80 {
81 fileName objectPath(io.instance()/io.name());
82
83 if (isFileOrDir(isFile, objectPath))
84 {
85 return objectPath;
86 }
87 }
88 else
89 {
90 fileName path(io.path());
91 fileName objectPath(path/io.name());
92
93 if (isFileOrDir(isFile, objectPath))
94 {
95 return objectPath;
96 }
97 else
98 {
99 if
100 (
101 checkGlobal
102 && io.time().processorCase()
103 && (
104 io.instance() == io.time().system()
105 || io.instance() == io.time().constant()
106 )
107 )
108 {
109 // Constant & system can come from global case
110
111 fileName parentObjectPath =
112 io.rootPath()/io.time().globalCaseName()
113 /io.instance()/io.db().dbDir()/io.local()/io.name();
114
115 if (isFileOrDir(isFile, parentObjectPath))
116 {
117 return parentObjectPath;
118 }
119 }
120
121 // Check if parallel "procesors" directory
122 if (io.time().processorCase())
123 {
124 refPtr<dirIndexList> pDirs
125 (
126 fileOperation::lookupAndCacheProcessorsPath
127 (
128 io.objectPath(),
129 false // No additional parallel synchronisation
130 )
131 );
132
133 for (const dirIndex& dirIdx : pDirs())
134 {
135 const fileName& pDir = dirIdx.first();
136 fileName objPath =
137 processorsPath(io, io.instance(), pDir)
138 /io.name();
139 if (objPath != objectPath && isFileOrDir(isFile, objPath))
140 {
141 return objPath;
142 }
143 }
144 }
145
146
147 // Check for approximately same time. E.g. if time = 1e-2 and
148 // directory is 0.01 (due to different time formats)
149 if (search && !Foam::isDir(path))
150 {
151 word newInstancePath = io.time().findInstancePath
152 (
153 instant(io.instance())
154 );
155
156 if (newInstancePath.size())
157 {
158 fileName fName
159 (
160 io.rootPath()/io.caseName()
161 /newInstancePath/io.db().dbDir()/io.local()/io.name()
162 );
163
164 if (isFileOrDir(isFile, fName))
165 {
166 return fName;
167 }
168 }
169 }
170 }
171 }
172
173 return fileName();
174}

193// Construction helper: self/world/local communicator and IO ranks
194static Tuple2<label, labelList> getCommPattern()
195{
196 // Default is COMM_SELF (only involves itself)
197 Tuple2<label, labelList> commAndIORanks
198 (
199 UPstream::commSelf(),
200 fileOperation::getGlobalIORanks()
201 );
202
203 if (UPstream::parRun() && commAndIORanks.second().size() > 1)
204 {
205 // Multiple masters: ranks for my IO range
206 commAndIORanks.first() = UPstream::allocateCommunicator
207 (
208 UPstream::worldComm,
209 fileOperation::subRanks(commAndIORanks.second())
210 );
211 }
212
213 return commAndIORanks;
214}

564bool Foam::fileOperations::uncollatedFileOperation::readHeader
565(
566 IOobject& io,
567 const fileName& fName,
568 const word& typeName
569) const
570{
571 if (debug)
572 {
573 Pout<< "uncollatedFileOperation::readHeader :"

575 << " typeName:" << typeName << endl;
576 }
577 if (fName.empty())
578 {
579 if (IOobject::debug)
580 {
581 InfoInFunction

584 }
585
586 return false;

588
589 autoPtr<ISstream> isPtr(NewIFstream(fName));
590
591 if (!isPtr || !isPtr->good())
592 {

596 // Regular header or from decomposed data
597 bool ok = decomposedBlockData::readHeader(io, *isPtr);
598
599 if (debug)
600 {
601 Pout<< "uncollatedFileOperation::readHeader :"
602 << " for fName:" << fName
603 << " ok:" << ok
604 << " headerClassName:" << io.headerClassName() << endl;
605 }
606
607 return ok;

611Foam::autoPtr<Foam::ISstream>
612Foam::fileOperations::uncollatedFileOperation::readStream
613(
614 regIOobject& io,
615 const fileName& fName,
616 const word& typeName,
617 const bool readOnProc
618) const
619{
620 if (!readOnProc)
621 {
622 return autoPtr<ISstream>(new dummyISstream());
623 }
624
625 if (fName.empty())
626 {
627 FatalErrorInFunction
628 << "cannot find file " << io.objectPath()
629 << exit(FatalError);
630 }
631
632 autoPtr<ISstream> isPtr = NewIFstream(fName);
633
634 if (!isPtr || !isPtr->good())
635 {
636 FatalIOError
637 (
638 "uncollatedFileOperation::readStream()",
639 __FILE__,
640 __LINE__,
641 fName,
642 0
643 ) << "cannot open file"
644 << exit(FatalIOError);
645 }
646 else if (!io.readHeader(*isPtr))
647 {
648 FatalIOErrorInFunction(*isPtr)
649 << "problem while reading header for object " << io.name()
650 << exit(FatalIOError);
651 }
652
653 if (!decomposedBlockData::isCollatedType(io))
654 {
655 // Short circuit: non-collated format.
656 return isPtr;
657 }
658 else
659 {
660 // Analyse the objectpath to find out the processor we're trying
661 // to access
662 label proci = detectProcessorPath(io.objectPath());
663
664 if (proci == -1)
665 {
666 FatalIOErrorInFunction(*isPtr)
667 << "could not detect processor number"
668 << " from objectPath:" << io.objectPath()
669 << " fName:" << fName
670 << exit(FatalIOError);
671 }
672
673 // Analyse the fileName for any processor subset. Note: this
674 // should really be part of filePath() which should return
675 // both file and index in file.
676 fileName path, procDir, local;
677 procRangeType group;
678 label nProcs;
679 splitProcessorPath(fName, path, procDir, local, group, nProcs);
680
681 // The local rank (offset)
682 if (!group.empty())
683 {
684 proci = proci - group.start();
685 }
686
687 // Read data and return as stream
688 return decomposedBlockData::readBlock(proci, *isPtr, io);
689 }
690}

693bool Foam::fileOperations::uncollatedFileOperation::read
694(
695 regIOobject& io,
696 const bool masterOnly,
697 const IOstreamOption::streamFormat format,
698 const word& typeName
699) const
700{
701 bool ok = false;
702
703 if (!masterOnly || Pstream::master(UPstream::worldComm))
704 {
705 if (debug)
706 {
707 Pout<< "uncollatedFileOperation::read :"
708 << " Reading object " << io.objectPath()
709 << " from file " << endl;
710 }
711
712 // Set flag for e.g. codeStream
713 const bool oldGlobal = io.globalObject(masterOnly);
714
715 // If codeStream originates from dictionary which is
716 // not IOdictionary we have a problem so use global
717 const bool oldMasterOnly = regIOobject::masterOnlyReading;
718 regIOobject::masterOnlyReading = masterOnly;
719
720 // Read file
721 ok = io.readData(io.readStream(typeName));
722 io.close();
723
724 // Restore flags
725 io.globalObject(oldGlobal);
726 regIOobject::masterOnlyReading = oldMasterOnly;
727
728 if (debug)
729 {
730 Pout<< "uncollatedFileOperation::read :"
731 << " Done reading object " << io.objectPath()
732 << " from file " << endl;
733 }
734 }
735
736 if (masterOnly && Pstream::parRun())
737 {
738 Pstream::broadcasts
739 (
740 UPstream::worldComm,
741 io.headerClassName(),
742 io.note()
743 );
744
745 if (UPstream::master(UPstream::worldComm))
746 {
747 OPBstream toAll
748 (
749 UPstream::masterNo(),
750 UPstream::worldComm,
751 format
752 );
753 bool okWrite = io.writeData(toAll);
754 ok = ok && okWrite;
755 }
756 else
757 {
758 IPBstream fromMaster
759 (
760 UPstream::masterNo(),
761 UPstream::worldComm,
762 format
763 );
764 ok = io.readData(fromMaster);
765 }
766 }
767
768 return ok;
769}
