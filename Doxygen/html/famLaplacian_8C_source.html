272tmp<faMatrix<Type>>
273laplacian
274(
275 const areaTensorField& gamma,
276 const GeometricField<Type, faPatchField, areaMesh>& vf
277)
278{
279 const faMesh& mesh = vf.mesh();
280
281 return fam::laplacian
282 (
283 (mesh.Le() & fac::interpolate(gamma) & mesh.Le())
284 /sqr(mesh.magLe()),
285 vf
286 );
287}
