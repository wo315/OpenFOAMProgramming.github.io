63void Foam::fv::cellSetOption::setSelection(const dictionary& dict)
64{
65 selectionNames_.clear();
66
67 switch (selectionMode_)
68 {
69 case smAll:
70 {
71 break;
72 }
73 case smGeometric:
74 {
75 geometricSelection_ = dict.subDict("selection");
76 break;
77 }
78 case smPoints:
79 {
80 dict.readEntry("points", points_);
81 break;
82 }
83 case smCellSet:
84 {
85 selectionNames_.resize(1);
86 dict.readEntry("cellSet", selectionNames_.first());
87 break;
88 }
89 case smCellZone:
90 {
91 if
92 (
93 !dict.readIfPresent("cellZones", selectionNames_)
94 || selectionNames_.empty()
95 )
96 {
97 selectionNames_.resize(1);
98 dict.readEntry("cellZone", selectionNames_.first());
99 }
100 break;
101 }
102 case smCellType:
103 {
104 break;
105 }
106 default:
107 {
108 FatalErrorInFunction
109 << "Unknown selectionMode "
110 << selectionModeTypeNames_[selectionMode_]
111 << ". Valid selectionMode types : "
112 << selectionModeTypeNames_
113 << exit(FatalError);
114 }
115 }
116}

119void Foam::fv::cellSetOption::setVol()
120{
121 // Set volume information
122
123 scalar sumVol = 0;
124 for (const label celli : cells_)
125 {
126 sumVol += mesh_.V()[celli];
127 }
128 reduce(sumVol, sumOp<scalar>());
129
130 const scalar old(V_);
131 V_ = sumVol;
132
133 // Compare volume values, stringified using current write precision
134 if
135 (
136 Time::timeName(old, IOstream::defaultPrecision())
137 != Time::timeName(V_, IOstream::defaultPrecision())
138 )
139 {
140 Info<< indent
141 << "- selected " << returnReduce(cells_.size(), sumOp<label>())
142 << " cell(s) with volume " << V_ << endl;
143 }
144}

147void Foam::fv::cellSetOption::setCellSelection()
148{
149 switch (selectionMode_)
150 {
151 case smAll:
152 {
153 Info<< indent << "- selecting all cells" << endl;
154
155 cells_ = identity(mesh_.nCells());
156 break;
157 }
158 case smGeometric:
159 {
160 Info<< indent << "- selecting cells geometrically" << endl;
161
162 bitSet selectedCells
163 (
164 // verbosity = true
165 cellBitSet::select(mesh_, geometricSelection_, true)
166 );
167
168 // From bitSet -> labels
169 cells_ = selectedCells.sortedToc();
170 break;
171 }
172 case smPoints:
173 {
174 Info<< indent << "- selecting cells using points" << endl;
175
176 labelHashSet selectedCells;
177
178 for (const point& p : points_)
179 {
180 const label celli = mesh_.findCell(p);
181
182 const bool found = (celli >= 0);
183
184 if (found)
185 {
186 selectedCells.insert(celli);
187 }
188
189 if (!returnReduceOr(found))
190 {
191 WarningInFunction
192 << "No owner cell found for point " << p << endl;
193 }
194 }
195
196 cells_ = selectedCells.sortedToc();
197 break;
198 }
199 case smCellSet:
200 {
201 Info<< indent
202 << "- selecting cells using cellSet "
203 << zoneName() << endl;
204
205 cells_ = cellSet(mesh_, zoneName()).sortedToc();
206 break;
207 }
208 case smCellZone:
209 {
210 Info<< indent
211 << "- selecting cells using cellZones "
212 << flatOutput(selectionNames_) << nl;
213
214 const auto& zones = mesh_.cellZones();
215
216 // Also handles groups, multiple zones etc ...
217 labelList zoneIDs = zones.indices(selectionNames_);
218
219 if (zoneIDs.empty())
220 {
221 FatalErrorInFunction
222 << "No matching cellZones: "
223 << flatOutput(selectionNames_) << nl
224 << "Valid zones : "
225 << flatOutput(zones.names()) << nl
226 << "Valid groups: "
227 << flatOutput(zones.groupNames())
228 << nl
229 << exit(FatalError);
230 }
231
232 if (zoneIDs.size() == 1)
233 {
234 cells_ = zones[zoneIDs.first()];
235 // TBD: Foam::sort(cells_);
236 }
237 else
238 {
239 cells_ = zones.selection(zoneIDs).sortedToc();
240 }
241 break;
242 }
243 case smCellType:
244 {
245 labelHashSet selectedCells;
246 const cellCellStencilObject& overlap = Stencil::New(mesh_);
247 const labelList& cellTypes = overlap.cellTypes();
248 forAll(cellTypes, celli)
249 {
250 if (cellTypes[celli] == cellCellStencil::POROUS)
251 {
252 selectedCells.insert(celli);
253 }
254 cells_ = selectedCells.sortedToc();
255 }
256 break;
257 }
258 default:
259 {
260 FatalErrorInFunction
261 << "Unknown selectionMode "
262 << selectionModeTypeNames_[selectionMode_]
263 << ". Valid selectionMode types are "
264 << selectionModeTypeNames_
265 << exit(FatalError);
266 }
267 }
268
269 if (smAll != selectionMode_ && returnReduceAnd(cells_.empty()))
270 {
271 WarningInFunction
272 << "No cells selected!" << endl;
273 }
274}

307bool Foam::fv::cellSetOption::isActive()
308{
309 if (fv::option::isActive() && inTimeLimits(mesh_.time().value()))
310 {
311 // Update the cell set if the mesh is changing
312 if (mesh_.changing())
313 {
314 if (mesh_.topoChanging())
315 {
316 setCellSelection();
317 // Force printing of new set volume
318 V_ = -GREAT;
319 }
320 else if
321 (
322 selectionMode_ == smGeometric
323 || selectionMode_ == smPoints
324 || selectionMode_ == smCellType
325 )
326 {
327 // Geometric selection mode(s)
328 setCellSelection();
329 }
330
331 // Report new volume (if changed)
332 setVol();
333 }
334
335 return true;
336 }
337
338 return false;
339}
