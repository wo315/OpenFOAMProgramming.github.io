52const Foam::volScalarField&
53Foam::radiation::localDensityAbsorptionEmission::alpha(word alphaName) const
54{
55 if (!mesh_.foundObject<volScalarField>(alphaName))
56 {
57 FatalErrorInFunction
58 << "Unable to retrieve density field " << alphaName << " from "
59 << "database. Available objects:" << mesh_.sortedNames()
60 << exit(FatalError);
61 }
62
63 return mesh_.lookupObject<volScalarField>(alphaName);
64}

86Foam::tmp<Foam::volScalarField>
87Foam::radiation::localDensityAbsorptionEmission::aCont(const label bandI) const
88{
89 tmp<volScalarField> ta
90 (
91 new volScalarField
92 (
93 IOobject
94 (
95 "a",
96 mesh_.time().timeName(),
97 mesh_,
98 IOobject::NO_READ,
99 IOobject::NO_WRITE,
100 IOobject::NO_REGISTER
101 ),
102 mesh_,
103 dimensionedScalar(inv(dimLength), Zero)
104 )
105 );
106
107 volScalarField& a = ta.ref();
108
109 forAll(alphaNames_, i)
110 {
111 dimensionedScalar aPhase("a", dimless/dimLength, aCoeff_[i]);
112 a += max(alpha(alphaNames_[i]), scalar(0))*aPhase;
113 }
114
115 return ta;
116}
