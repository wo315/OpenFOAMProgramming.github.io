Usage
Minimal example by using constant/<CloudProperties>: 
subModels
{
    injectionModels
    {
        <name>
        {
            ...

            sizeDistribution
            {
                type        RosinRammler;
                RosinRammlerDistribution
                {
                    lambda      <scaleParameterValue>;
                    n           <shapeParameterValue>;
                    minValue    <minValue>;
                    maxValue    <maxValue>;
                }
            }
        }
    }
}
