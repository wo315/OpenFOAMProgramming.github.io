33template<class Type>
34void Foam::particleTracksSampler::createTrackField
35(
36 const UList<Type>& values,
37 List<DynamicList<Type>>& trackValues
38) const
39{
40 List<Type> allValues(cloudGather_.gather(values));
41
42 const label nTracks = trackValues.size();
43
44 forAll(allValues, i)
45 {
46 const label globalId =
47 origParcelAddr_.toGlobal(origProcIds_[i], origParcelIds_[i]);
48
49 if (globalId % stride_ == 0)
50 {
51 const label trackId = globalId/stride_;
52
53 if
54 (
55 trackId < nTracks
56 && trackValues[trackId].size() < maxPositions_
57 )
58 {
59 trackValues[trackId].append(allValues[i]);
60 }
61 }
62 }
63}

66template<class Type>
67Foam::label Foam::particleTracksSampler::setTrackFields
68(
69 const objectRegistry& obr,
70 HashTable<List<DynamicList<Type>>>& fieldTable
71) const
72{
73 // First time - obtain field names and populate storage locations
74 if (fieldTable.empty())
75 {
76 wordList fieldNames = obr.names<IOField<Type>>();
77
78 if (Pstream::parRun())
79 {
80 Pstream::combineReduce(fieldNames, ListOps::uniqueEqOp<word>());
81 }
82
83 for (const word& fieldName : fieldNames)
84 {
85 fieldTable(fieldName).resize(nTracks());
86 }
87 }
88
89 // Process in parallel-consistent order
90 for (const word& fieldName : fieldTable.sortedToc())
91 {
92 auto& output = fieldTable[fieldName];
93
94 const auto* ptr = obr.cfindObject<IOField<Type>>(fieldName);
95
96 this->createTrackField
97 (
98 ptr ? static_cast<const UList<Type>&>(*ptr) : UList<Type>::null(),
99 output
100 );
101 }
102
103 return fieldTable.size();
104}
