105template<class T>
106void Foam::UList<T>::deepCopy(const UList<T>& list)
107{
108 const label len = this->size_;
109
110 if (len != list.size_)
111 {
112 FatalErrorInFunction
113 << "Lists have different sizes: "
114 << len << " != " << list.size() << nl
115 << abort(FatalError);
116 }
117 else if (len)
118 {
119 #ifdef USEMEMCPY
120 if (is_contiguous<T>::value)
121 {
122 std::memcpy
123 (
124 static_cast<void*>(this->v_), list.v_, this->size_bytes()
125 );
126 }
127 else
128 #endif
129 {
130 List_ACCESS(T, (*this), lhs);
131 List_CONST_ACCESS(T, list, rhs);
132 for (label i = 0; i < len; ++i)
133 {
134 lhs[i] = rhs[i];
135 }
136 }
137 }
138}

210template<class T>
211Foam::label Foam::UList<T>::find(const T& val, label pos) const
212{
213 const label len = this->size();
214
215 if (pos >= 0 && len)
216 {
217 List_CONST_ACCESS(T, (*this), list);
218
219 while (pos < len)
220 {
221 if (list[pos] == val)
222 {
223 return pos;
224 }
225
226 ++pos;
227 }
228 }
229
230 return -1;

234template<class T>
235Foam::label Foam::UList<T>::rfind(const T& val, label pos) const
236{
237 // pos == -1 has same meaning as std::string::npos - search from end
238 if (pos < 0 || pos >= this->size())

240 pos = this->size()-1;
241 }
242
243 List_CONST_ACCESS(T, (*this), list);

250 }

261template<class T>
262bool Foam::UList<T>::operator==(const UList<T>& list) const
263{
268 // && std::equal(cbegin(), cend(), list.cbegin())
270
271 const label len = this->size_;
272 if (len != list.size_)
273 {
274 return false;
275 }
276
277 List_CONST_ACCESS(T, (*this), lhs);
278 List_CONST_ACCESS(T, (list), rhs);
279
280 for (label i = 0; i < len; ++i)
281 {
282 if (!(lhs[i] == rhs[i])) return false;
283 }
284
285 return true;
286}

296template<class T>
297bool Foam::UList<T>::operator<(const UList<T>& list) const
298{
305
306 const const_iterator last1 = cend();
307 const const_iterator last2 = list.cend();
308
309 for
310 (
311 const_iterator lhs = cbegin(), rhs = list.cbegin();
312 (lhs != last1) && (rhs != last2);
313 ++lhs, (void) ++rhs
314 )
315 {
316 if (*lhs < *rhs)
317 {
318 return true;
319 }
320 else if (*rhs < *lhs)
321 {
322 return false;
323 }
324 }
325
326 // Contents look to be identical, or lists have different sizes
327 return (this->size_ < list.size_);
328}
