65Foam::tmp<Foam::volScalarField>
66Foam::dragModels::IshiiZuber::CdRe() const
67{
68 volScalarField Re(pair_.Re());
69 volScalarField Eo(pair_.Eo());
70
71 volScalarField mud(pair_.dispersed().mu());
72 volScalarField muc(pair_.continuous().mu());
73
74 volScalarField muStar((mud + 0.4*muc)/(mud + muc));
75
76 volScalarField muMix
77 (
78 muc
79 *pow(max(1 - pair_.dispersed(), scalar(1e-3)), -2.5*muStar)
80 );
81
82 volScalarField ReM(Re*muc/muMix);
83 volScalarField CdRe
84 (
85 pos0(1000 - ReM)*24.0*(scalar(1) + 0.15*pow(ReM, 0.687))
86 + neg(1000 - ReM)*0.44*ReM
87 );
88
89 volScalarField F((muc/muMix)*sqrt(1 - pair_.dispersed()));
90 F.clamp_min(1e-3);
91
92 volScalarField Ealpha((1 + 17.67*pow(F, 0.8571428))/(18.67*F));
93
94 volScalarField CdReEllipse(Ealpha*0.6666*sqrt(Eo)*Re);
95
96 return
97 pos0(CdReEllipse - CdRe)
98 *min(CdReEllipse, Re*sqr(1 - pair_.dispersed())*2.66667)
99 + neg(CdReEllipse - CdRe)*CdRe;
100}
