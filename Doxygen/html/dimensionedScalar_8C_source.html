243#define transFunc(func) \
244dimensionedScalar func(const dimensionedScalar& ds) \
245{ \
246 if (dimensionSet::checking() && !ds.dimensions().dimensionless()) \
247 { \
248 FatalErrorInFunction \
249 << "scalar is not dimensionless: " << ds.dimensions() << nl \
250 << abort(FatalError); \
251 } \
252 \
253 return dimensionedScalar \
254 ( \
255 #func "(" + ds.name() + ')', \
256 dimless, \
257 ::func(ds.value()) \
258 ); \
259}
