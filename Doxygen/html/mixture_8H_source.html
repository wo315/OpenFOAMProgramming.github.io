73 {
74 scalar volTot = 0;
75
76 for (label i = 0; i < size(); i++)
77 {
78 volTot += operator[](i).volFrac();
79 }

83 FatalErrorInFunction
84 << "Sum of volume fractions for Mixture " << name_
