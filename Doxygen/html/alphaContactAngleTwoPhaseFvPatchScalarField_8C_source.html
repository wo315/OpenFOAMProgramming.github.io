126void Foam::alphaContactAngleTwoPhaseFvPatchScalarField::evaluate
127(
128 const Pstream::commsTypes
129)
130{
131 if (limit_ == lcGradient)
132 {
133 gradient() =
134 patch().deltaCoeffs()
135 *(
136 clamp(*this + gradient()/patch().deltaCoeffs(), zero_one{})
137 - *this
138 );
139 }
140 else if (limit_ == lcZeroGradient)
141 {
142 gradient() = Zero;
143 }
144
145 fixedGradientFvPatchScalarField::evaluate();
146
147 if (limit_ == lcAlpha)
148 {
149 scalarField::operator=(clamp(*this, zero_one{}));
150 }
151}
