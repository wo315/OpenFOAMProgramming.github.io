115template<class Type>
116bool Foam::cyclicACMIFvsPatchField<Type>::coupled() const
117{

119 (
120 Pstream::parRun()

122 this->cyclicACMIPatch_.size()
123 && this->cyclicACMIPatch_.cyclicACMIPatch().neighbPatch().size()

125 )
126 {

128 }

130 return false;
