112
113void Foam::boundaryToFace::applyToSet
114(
115 const topoSetSource::setAction action,
116 topoSet& set
117) const
118{
119 if (action == topoSetSource::ADD || action == topoSetSource::NEW)
120 {
121 if (verbose_)
122 {
123 Info<< " Adding all boundary faces ..." << endl;
124 }
125
126 combine(set, true);
127 }
128 else if (action == topoSetSource::SUBTRACT)
129 {
130 if (verbose_)
131 {
132 Info<< " Removing all boundary faces ..." << endl;
133 }
134
135 combine(set, false);
136 }
137}
