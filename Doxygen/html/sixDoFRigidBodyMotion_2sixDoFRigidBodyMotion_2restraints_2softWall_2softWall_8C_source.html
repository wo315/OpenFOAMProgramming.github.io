71void Foam::sixDoFRigidBodyMotionRestraints::softWall::restrain
72(
73 const sixDoFRigidBodyMotion& motion,
74 vector& restraintPosition,
75 vector& restraintForce,
76 vector& restraintMoment
77) const
78{
79 restraintPosition = motion.transform(refAttachmentPt_);
80 restraintForce = Zero;
81 restraintMoment = Zero;
82
83 const vector r(restraintPosition - anchor_);
84
85 const vector v(motion.velocity(restraintPosition));
86
87 const scalar d = (wallNormal_/mag(wallNormal_)) & r;
88
89 const vector rDir(r/(mag(r) + VSMALL));
90
91 const scalar m = motion.mass();
92 const scalar wn = 3.14/C_;
93 const scalar damping = psi_*2*wn*m;
94 const scalar stiffness = sqr(wn)*m;
95
96 if (d < 0)
97 {
98 restraintForce = (-damping*(rDir & v) + stiffness*d)*rDir;
99 restraintMoment = restraintPosition^restraintForce;
100 }
101
102 if (motion.report())
103 {
104 Info<< " restraintPosition: " << restraintPosition
105 << " restraintForce: " << restraintForce
106 << " restraintMoment: " << restraintMoment
107 << endl;
108 }
109}
